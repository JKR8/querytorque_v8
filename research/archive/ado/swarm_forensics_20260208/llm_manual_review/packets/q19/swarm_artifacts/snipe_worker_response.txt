```sql
WITH filtered_date AS (
    SELECT d_date_sk
    FROM date_dim
    WHERE d_moy = 12
      AND d_year = 1999
), filtered_item AS (
    SELECT i_item_sk, i_brand_id, i_brand, i_manufact_id, i_manufact
    FROM item
    WHERE i_manager_id = 2
), filtered_store_sales AS (
    SELECT ss_ext_sales_price, ss_item_sk, ss_customer_sk, ss_store_sk
    FROM store_sales
    INNER JOIN filtered_date ON store_sales.ss_sold_date_sk = filtered_date.d_date_sk
)
SELECT
    filtered_item.i_brand_id AS brand_id,
    filtered_item.i_brand AS brand,
    filtered_item.i_manufact_id,
    filtered_item.i_manufact,
    SUM(filtered_store_sales.ss_ext_sales_price) AS ext_price
FROM filtered_store_sales
INNER JOIN filtered_item ON filtered_store_sales.ss_item_sk = filtered_item.i_item_sk
INNER JOIN customer ON filtered_store_sales.ss_customer_sk = customer.c_customer_sk
INNER JOIN customer_address ON customer.c_current_addr_sk = customer_address.ca_address_sk
INNER JOIN store ON filtered_store_sales.ss_store_sk = store.s_store_sk
WHERE SUBSTRING(customer_address.ca_zip, 1, 5) <> SUBSTRING(store.s_zip, 1, 5)
GROUP BY
    filtered_item.i_brand,
    filtered_item.i_brand_id,
    filtered_item.i_manufact_id,
    filtered_item.i_manufact
ORDER BY
    ext_price DESC,
    filtered_item.i_brand,
    filtered_item.i_brand_id,
    filtered_item.i_manufact_id,
    filtered_item.i_manufact
LIMIT 100
```

**Changes**: Applied early filtering CTEs for date_dim and item to create small hash tables, then joined with store_sales in a dedicated CTE to reduce the large fact table early. This isolates dimension filtering from the complex multi-table join.

**Expected speedup**: 2.0x+ by reducing the store_sales scan through early join with filtered dimensions, creating smaller working sets for subsequent joins.
