```sql
WITH filtered_date AS (
    SELECT d_date_sk
    FROM date_dim
    WHERE d_dow IN (6, 0)
      AND d_year IN (1999, 1999 + 1, 1999 + 2)
), filtered_store AS (
    SELECT s_store_sk
    FROM store
    WHERE s_city IN ('Five Points', 'Centerville', 'Oak Grove', 'Fairview', 'Liberty')
), filtered_hd AS (
    SELECT hd_demo_sk
    FROM household_demographics
    WHERE hd_dep_count = 6 OR hd_vehicle_count = 0
), sales_aggregation AS (
    SELECT
        ss_ticket_number,
        ss_customer_sk,
        customer_address.ca_city AS bought_city,
        SUM(ss_coupon_amt) AS amt,
        SUM(ss_net_profit) AS profit
    FROM store_sales
    JOIN filtered_date ON store_sales.ss_sold_date_sk = filtered_date.d_date_sk
    JOIN filtered_store ON store_sales.ss_store_sk = filtered_store.s_store_sk
    JOIN filtered_hd ON store_sales.ss_hdemo_sk = filtered_hd.hd_demo_sk
    JOIN customer_address ON store_sales.ss_addr_sk = customer_address.ca_address_sk
    GROUP BY
        ss_ticket_number,
        ss_customer_sk,
        ss_addr_sk,
        customer_address.ca_city
)
SELECT
    c_last_name,
    c_first_name,
    current_addr.ca_city AS ca_city,
    sales_aggregation.bought_city,
    sales_aggregation.ss_ticket_number,
    sales_aggregation.amt,
    sales_aggregation.profit
FROM sales_aggregation
JOIN customer ON sales_aggregation.ss_customer_sk = customer.c_customer_sk
JOIN customer_address AS current_addr ON customer.c_current_addr_sk = current_addr.ca_address_sk
WHERE current_addr.ca_city <> sales_aggregation.bought_city
ORDER BY
    c_last_name,
    c_first_name,
    ca_city,
    bought_city,
    ss_ticket_number
LIMIT 100
```

**Changes**: Pre-filtered three dimension tables (date_dim, store, household_demographics) into CTEs before joining with store_sales and customer_address, reducing the fact table scan early. The aggregation is performed in a single pass after all selective filters are applied.
