```sql
WITH filtered_dates AS (
  SELECT d_date_sk
  FROM date_dim
  WHERE d_dom BETWEEN 1 AND 2
    AND d_year IN (1998, 1999, 2000)
),
filtered_stores AS (
  SELECT s_store_sk, s_city
  FROM store
  WHERE s_city IN ('Pleasant Hill', 'Five Points')
),
sales_hd_dep AS (
  SELECT
    ss_ticket_number,
    ss_customer_sk,
    ss_addr_sk,
    ca_city AS bought_city,
    SUM(ss_ext_sales_price) AS extended_price,
    SUM(ss_ext_list_price) AS list_price,
    SUM(ss_ext_tax) AS extended_tax
  FROM store_sales
  JOIN filtered_dates ON ss_sold_date_sk = d_date_sk
  JOIN filtered_stores ON ss_store_sk = s_store_sk
  JOIN household_demographics ON ss_hdemo_sk = hd_demo_sk
  JOIN customer_address ON ss_addr_sk = ca_address_sk
  WHERE hd_dep_count = 8
  GROUP BY ss_ticket_number, ss_customer_sk, ss_addr_sk, ca_city
),
sales_hd_veh AS (
  SELECT
    ss_ticket_number,
    ss_customer_sk,
    ss_addr_sk,
    ca_city AS bought_city,
    SUM(ss_ext_sales_price) AS extended_price,
    SUM(ss_ext_list_price) AS list_price,
    SUM(ss_ext_tax) AS extended_tax
  FROM store_sales
  JOIN filtered_dates ON ss_sold_date_sk = d_date_sk
  JOIN filtered_stores ON ss_store_sk = s_store_sk
  JOIN household_demographics ON ss_hdemo_sk = hd_demo_sk
  JOIN customer_address ON ss_addr_sk = ca_address_sk
  WHERE hd_vehicle_count = -1
  GROUP BY ss_ticket_number, ss_customer_sk, ss_addr_sk, ca_city
),
unified_sales AS (
  SELECT * FROM sales_hd_dep
  UNION ALL
  SELECT * FROM sales_hd_veh
)
SELECT
  c_last_name,
  c_first_name,
  current_addr.ca_city,
  unified_sales.bought_city,
  unified_sales.ss_ticket_number,
  unified_sales.extended_price,
  unified_sales.extended_tax,
  unified_sales.list_price
FROM unified_sales
JOIN customer ON unified_sales.ss_customer_sk = c_customer_sk
JOIN customer_address AS current_addr ON customer.c_current_addr_sk = current_addr.ca_address_sk
WHERE current_addr.ca_city <> unified_sales.bought_city
ORDER BY c_last_name, unified_sales.ss_ticket_number
LIMIT 100
```

**Changes**: Transformed OR condition on different household_demographics columns into UNION ALL of two pre-aggregated CTEs (hd_dep_count=8 and hd_vehicle_count=-1), with shared date and store filters isolated into separate CTEs for reuse.

**Expected speedup**: 2-3x from parallelizable UNION branches and reduced join complexity.
