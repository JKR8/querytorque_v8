```sql
WITH date_filtered AS (
  SELECT d_date_sk
  FROM date_dim
  WHERE d_year = 2002 AND d_moy BETWEEN 10 AND 10+2
),
store_customers AS MATERIALIZED (
  SELECT ss_customer_sk
  FROM store_sales
  JOIN date_filtered ON store_sales.ss_sold_date_sk = date_filtered.d_date_sk
  WHERE ss_list_price BETWEEN 80 AND 169
  GROUP BY ss_customer_sk
),
web_customers AS MATERIALIZED (
  SELECT ws_bill_customer_sk
  FROM web_sales
  JOIN date_filtered ON web_sales.ws_sold_date_sk = date_filtered.d_date_sk
  WHERE ws_list_price BETWEEN 80 AND 169
  GROUP BY ws_bill_customer_sk
),
catalog_customers AS MATERIALIZED (
  SELECT cs_ship_customer_sk
  FROM catalog_sales
  JOIN date_filtered ON catalog_sales.cs_sold_date_sk = date_filtered.d_date_sk
  WHERE cs_list_price BETWEEN 80 AND 169
  GROUP BY cs_ship_customer_sk
)
SELECT 
  cd_gender,
  cd_marital_status,
  cd_education_status,
  count(*) cnt1,
  cd_purchase_estimate,
  count(*) cnt2,
  cd_credit_rating,
  count(*) cnt3
FROM customer c
JOIN customer_address ca ON c.c_current_addr_sk = ca.ca_address_sk
JOIN customer_demographics ON cd_demo_sk = c.c_current_cdemo_sk
JOIN store_customers sc ON c.c_customer_sk = sc.ss_customer_sk
LEFT JOIN web_customers wc ON c.c_customer_sk = wc.ws_bill_customer_sk
LEFT JOIN catalog_customers cc ON c.c_customer_sk = cc.cs_ship_customer_sk
WHERE ca.ca_state IN ('CO','NC','TX')
  AND cd_marital_status IN ('S', 'M', 'U')
  AND cd_education_status IN ('Primary', 'College')
  AND wc.ws_bill_customer_sk IS NULL
  AND cc.cs_ship_customer_sk IS NULL
GROUP BY cd_gender,
         cd_marital_status,
         cd_education_status,
         cd_purchase_estimate,
         cd_credit_rating
ORDER BY cd_gender,
         cd_marital_status,
         cd_education_status,
         cd_purchase_estimate,
         cd_credit_rating
LIMIT 100;
```
