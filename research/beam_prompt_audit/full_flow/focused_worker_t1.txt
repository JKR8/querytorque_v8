```sql
WITH date_filtered AS (
  SELECT d_date_sk
  FROM date_dim
  WHERE d_year = 2002 AND d_moy BETWEEN 10 AND 12
)
SELECT 
  cd_gender,
  cd_marital_status,
  cd_education_status,
  count(*) cnt1,
  cd_purchase_estimate,
  count(*) cnt2,
  cd_credit_rating,
  count(*) cnt3
FROM
  customer c, customer_address ca, customer_demographics
WHERE
  c.c_current_addr_sk = ca.ca_address_sk AND
  ca_state in ('CO','NC','TX') AND
  cd_demo_sk = c.c_current_cdemo_sk AND
  cd_marital_status in ('S', 'M', 'U') AND
  cd_education_status in ('Primary', 'College') AND
  EXISTS (
    SELECT *
    FROM store_sales, date_filtered
    WHERE c.c_customer_sk = store_sales.ss_customer_sk AND
          store_sales.ss_sold_date_sk = date_filtered.d_date_sk AND
          store_sales.ss_list_price between 80 and 169
  ) AND
  (NOT EXISTS (
    SELECT *
    FROM web_sales, date_filtered
    WHERE c.c_customer_sk = web_sales.ws_bill_customer_sk AND
          web_sales.ws_sold_date_sk = date_filtered.d_date_sk AND
          web_sales.ws_list_price between 80 and 169
  ) AND
   NOT EXISTS (
    SELECT *
    FROM catalog_sales, date_filtered
    WHERE c.c_customer_sk = catalog_sales.cs_ship_customer_sk AND
          catalog_sales.cs_sold_date_sk = date_filtered.d_date_sk AND
          catalog_sales.cs_list_price between 80 and 169
  ))
GROUP BY 
  cd_gender,
  cd_marital_status,
  cd_education_status,
  cd_purchase_estimate,
  cd_credit_rating
ORDER BY 
  cd_gender,
  cd_marital_status,
  cd_education_status,
  cd_purchase_estimate,
  cd_credit_rating
LIMIT 100;
```
