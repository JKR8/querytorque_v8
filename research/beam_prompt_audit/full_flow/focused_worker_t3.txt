```sql
WITH date_filtered AS (
  SELECT d_date_sk
  FROM date_dim
  WHERE d_year = 2002 AND d_moy BETWEEN 10 AND 10+2
)
SELECT 
  cd_gender,
  cd_marital_status,
  cd_education_status,
  count(*) cnt1,
  cd_purchase_estimate,
  count(*) cnt2,
  cd_credit_rating,
  count(*) cnt3
FROM customer c
INNER JOIN customer_address ca ON c.c_current_addr_sk = ca.ca_address_sk
INNER JOIN customer_demographics cd ON cd.cd_demo_sk = c.c_current_cdemo_sk
WHERE ca.ca_state IN ('CO','NC','TX')
  AND cd.cd_marital_status IN ('S','M','U')
  AND cd.cd_education_status IN ('Primary','College')
  AND EXISTS (
    SELECT 1
    FROM store_sales
    INNER JOIN date_filtered ON store_sales.ss_sold_date_sk = date_filtered.d_date_sk
    WHERE store_sales.ss_customer_sk = c.c_customer_sk
      AND store_sales.ss_list_price BETWEEN 80 AND 169
  )
  AND NOT EXISTS (
    SELECT 1
    FROM web_sales
    INNER JOIN date_filtered ON web_sales.ws_sold_date_sk = date_filtered.d_date_sk
    WHERE web_sales.ws_bill_customer_sk = c.c_customer_sk
      AND web_sales.ws_list_price BETWEEN 80 AND 169
  )
  AND NOT EXISTS (
    SELECT 1
    FROM catalog_sales
    INNER JOIN date_filtered ON catalog_sales.cs_sold_date_sk = date_filtered.d_date_sk
    WHERE catalog_sales.cs_ship_customer_sk = c.c_customer_sk
      AND catalog_sales.cs_list_price BETWEEN 80 AND 169
  )
GROUP BY cd_gender,
         cd_marital_status,
         cd_education_status,
         cd_purchase_estimate,
         cd_credit_rating
ORDER BY cd_gender,
         cd_marital_status,
         cd_education_status,
         cd_purchase_estimate,
         cd_credit_rating
LIMIT 100;
```
