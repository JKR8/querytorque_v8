WITH filtered_dates AS (
  SELECT d_date_sk
  FROM date_dim
  WHERE d_year = 2002 AND d_moy BETWEEN 10 AND 10 + 2
),
store_sales_filtered AS (
  SELECT ss_customer_sk
  FROM store_sales
  JOIN filtered_dates ON ss_sold_date_sk = d_date_sk
  WHERE ss_list_price BETWEEN 80 AND 169
),
web_sales_filtered AS (
  SELECT ws_bill_customer_sk
  FROM web_sales
  JOIN filtered_dates ON ws_sold_date_sk = d_date_sk
  WHERE ws_list_price BETWEEN 80 AND 169
),
catalog_sales_filtered AS (
  SELECT cs_ship_customer_sk
  FROM catalog_sales
  JOIN filtered_dates ON cs_sold_date_sk = d_date_sk
  WHERE cs_list_price BETWEEN 80 AND 169
)
SELECT 
  cd_gender,
  cd_marital_status,
  cd_education_status,
  COUNT(*) cnt1,
  cd_purchase_estimate,
  COUNT(*) cnt2,
  cd_credit_rating,
  COUNT(*) cnt3
FROM customer c
JOIN customer_address ca ON c.c_current_addr_sk = ca.ca_address_sk
JOIN customer_demographics cd ON cd.cd_demo_sk = c.c_current_cdemo_sk
WHERE 
  ca_state IN ('CO','NC','TX') AND
  cd_marital_status IN ('S', 'M', 'U') AND
  cd_education_status IN ('Primary', 'College') AND
  EXISTS (SELECT 1 FROM store_sales_filtered WHERE c.c_customer_sk = ss_customer_sk) AND
  NOT EXISTS (SELECT 1 FROM web_sales_filtered WHERE c.c_customer_sk = ws_bill_customer_sk) AND
  NOT EXISTS (SELECT 1 FROM catalog_sales_filtered WHERE c.c_customer_sk = cs_ship_customer_sk)
GROUP BY 
  cd_gender,
  cd_marital_status,
  cd_education_status,
  cd_purchase_estimate,
  cd_credit_rating
ORDER BY 
  cd_gender,
  cd_marital_status,
  cd_education_status,
  cd_purchase_estimate,
  cd_credit_rating
LIMIT 100;
