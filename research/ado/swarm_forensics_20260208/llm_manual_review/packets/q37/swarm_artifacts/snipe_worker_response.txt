```sql
WITH filtered_dates AS (
    SELECT d_date_sk
    FROM date_dim
    WHERE d_date BETWEEN CAST('1999-02-21' AS DATE) 
                    AND (CAST('1999-02-21' AS DATE) + INTERVAL '60' DAY)
), filtered_items AS (
    SELECT i_item_sk, i_item_id, i_item_desc, i_current_price
    FROM item
    WHERE i_current_price BETWEEN 45 AND 45 + 30
      AND i_manufact_id IN (856, 707, 1000, 747)
), filtered_inventory AS (
    SELECT inv_item_sk
    FROM inventory
    WHERE inv_quantity_on_hand BETWEEN 100 AND 500
), inventory_with_dates AS (
    SELECT DISTINCT fi.inv_item_sk
    FROM filtered_inventory fi
    JOIN filtered_dates fd ON fi.inv_item_sk = fd.d_date_sk
), item_inventory_set AS (
    SELECT fi.i_item_sk, fi.i_item_id, fi.i_item_desc, fi.i_current_price
    FROM filtered_items fi
    WHERE EXISTS (
        SELECT 1
        FROM inventory_with_dates iwd
        WHERE iwd.inv_item_sk = fi.i_item_sk
    )
)
SELECT 
    iis.i_item_id,
    iis.i_item_desc,
    iis.i_current_price
FROM item_inventory_set iis
WHERE EXISTS (
    SELECT 1
    FROM catalog_sales cs
    WHERE cs.cs_item_sk = iis.i_item_sk
)
GROUP BY
    iis.i_item_id,
    iis.i_item_desc,
    iis.i_current_price
ORDER BY
    iis.i_item_id
LIMIT 100;
```

Changes: Isolated dimension filters (date, item) into CTEs early, pre-joined filtered inventory with dates, converted the main multi-table join into a sequence of semi-joins (EXISTS) that eliminate rows before expensive joins, and maintained all original filters exactly.
