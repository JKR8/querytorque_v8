```sql
WITH bucket_data AS (
  SELECT 
    ss_list_price,
    CASE 
      WHEN ss_quantity BETWEEN 0 AND 5 
        AND (ss_list_price BETWEEN 131 AND 141
             OR ss_coupon_amt BETWEEN 16798 AND 17798
             OR ss_wholesale_cost BETWEEN 25 AND 45) THEN 1
      WHEN ss_quantity BETWEEN 6 AND 10 
        AND (ss_list_price BETWEEN 145 AND 155
             OR ss_coupon_amt BETWEEN 14792 AND 15792
             OR ss_wholesale_cost BETWEEN 46 AND 66) THEN 2
      WHEN ss_quantity BETWEEN 11 AND 15 
        AND (ss_list_price BETWEEN 150 AND 160
             OR ss_coupon_amt BETWEEN 6600 AND 7600
             OR ss_wholesale_cost BETWEEN 9 AND 29) THEN 3
      WHEN ss_quantity BETWEEN 16 AND 20 
        AND (ss_list_price BETWEEN 91 AND 101
             OR ss_coupon_amt BETWEEN 13493 AND 14493
             OR ss_wholesale_cost BETWEEN 36 AND 56) THEN 4
      WHEN ss_quantity BETWEEN 21 AND 25 
        AND (ss_list_price BETWEEN 0 AND 10
             OR ss_coupon_amt BETWEEN 7629 AND 8629
             OR ss_wholesale_cost BETWEEN 6 AND 26) THEN 5
      WHEN ss_quantity BETWEEN 26 AND 30 
        AND (ss_list_price BETWEEN 89 AND 99
             OR ss_coupon_amt BETWEEN 15257 AND 16257
             OR ss_wholesale_cost BETWEEN 31 AND 51) THEN 6
    END AS bucket
  FROM store_sales
),
bucket_aggs AS (
  SELECT
    bucket,
    AVG(ss_list_price) AS LP,
    COUNT(ss_list_price) AS CNT,
    COUNT(DISTINCT ss_list_price) AS CNTD
  FROM bucket_data
  WHERE bucket IS NOT NULL
  GROUP BY bucket
)
SELECT
  MAX(CASE WHEN bucket = 1 THEN LP END) AS B1_LP,
  COALESCE(MAX(CASE WHEN bucket = 1 THEN CNT END), 0) AS B1_CNT,
  COALESCE(MAX(CASE WHEN bucket = 1 THEN CNTD END), 0) AS B1_CNTD,
  MAX(CASE WHEN bucket = 2 THEN LP END) AS B2_LP,
  COALESCE(MAX(CASE WHEN bucket = 2 THEN CNT END), 0) AS B2_CNT,
  COALESCE(MAX(CASE WHEN bucket = 2 THEN CNTD END), 0) AS B2_CNTD,
  MAX(CASE WHEN bucket = 3 THEN LP END) AS B3_LP,
  COALESCE(MAX(CASE WHEN bucket = 3 THEN CNT END), 0) AS B3_CNT,
  COALESCE(MAX(CASE WHEN bucket = 3 THEN CNTD END), 0) AS B3_CNTD,
  MAX(CASE WHEN bucket = 4 THEN LP END) AS B4_LP,
  COALESCE(MAX(CASE WHEN bucket = 4 THEN CNT END), 0) AS B4_CNT,
  COALESCE(MAX(CASE WHEN bucket = 4 THEN CNTD END), 0) AS B4_CNTD,
  MAX(CASE WHEN bucket = 5 THEN LP END) AS B5_LP,
  COALESCE(MAX(CASE WHEN bucket = 5 THEN CNT END), 0) AS B5_CNT,
  COALESCE(MAX(CASE WHEN bucket = 5 THEN CNTD END), 0) AS B5_CNTD,
  MAX(CASE WHEN bucket = 6 THEN LP END) AS B6_LP,
  COALESCE(MAX(CASE WHEN bucket = 6 THEN CNT END), 0) AS B6_CNT,
  COALESCE(MAX(CASE WHEN bucket = 6 THEN CNTD END), 0) AS B6_CNTD
FROM bucket_aggs
LIMIT 100
```

**Changes**: Restructured six independent table scans into a single scan with bucket assignment, followed by grouped aggregation and pivoting. This reduces full table scans from six to one while preserving exact semantics.
