```sql
WITH filtered_date AS (
    SELECT d_date_sk
    FROM date_dim
    WHERE d_dom BETWEEN 1 AND 2
      AND d_year IN (2000, 2001, 2002)
),
filtered_store AS (
    SELECT s_store_sk
    FROM store
    WHERE s_county IN ('Fairfield County', 'Walker County', 'Daviess County', 'Barrow County')
),
household_filter AS (
    SELECT hd_demo_sk
    FROM household_demographics
    WHERE hd_vehicle_count > 0
      AND hd_dep_count > hd_vehicle_count
      AND hd_buy_potential = '501-1000'
    UNION ALL
    SELECT hd_demo_sk
    FROM household_demographics
    WHERE hd_vehicle_count > 0
      AND hd_dep_count > hd_vehicle_count
      AND hd_buy_potential = 'Unknown'
),
aggregated_sales AS (
    SELECT
        ss_ticket_number,
        ss_customer_sk,
        COUNT(*) AS cnt
    FROM store_sales
    JOIN filtered_date ON store_sales.ss_sold_date_sk = filtered_date.d_date_sk
    JOIN filtered_store ON store_sales.ss_store_sk = filtered_store.s_store_sk
    JOIN household_filter ON store_sales.ss_hdemo_sk = household_filter.hd_demo_sk
    GROUP BY ss_ticket_number, ss_customer_sk
    HAVING COUNT(*) BETWEEN 1 AND 5
)
SELECT
    c.c_last_name,
    c.c_first_name,
    c.c_salutation,
    c.c_preferred_cust_flag,
    a.ss_ticket_number,
    a.cnt
FROM aggregated_sales a
JOIN customer c ON a.ss_customer_sk = c.c_customer_sk
ORDER BY a.cnt DESC, c.c_last_name ASC;
```

Changes: Transformed OR conditions on hd_buy_potential into UNION ALL branches, replaced CASE expression with simplified filterable condition (hd_dep_count > hd_vehicle_count), and pre-filtered dimension tables into CTEs to reduce join volume early.

Expected speedup: 2.5-3x from focused filtering and dimension isolation.
