```sql
WITH filtered_date AS (
  SELECT d_date_sk
  FROM date_dim
  WHERE d_dom BETWEEN 1 AND 2
    AND d_year IN (2000, 2001, 2002)
),
filtered_store AS (
  SELECT s_store_sk
  FROM store
  WHERE s_county IN ('Fairfield County', 'Walker County', 'Daviess County', 'Barrow County')
),
filtered_hd AS (
  SELECT hd_demo_sk
  FROM household_demographics
  WHERE (hd_buy_potential = '501-1000' OR hd_buy_potential = 'Unknown')
    AND hd_vehicle_count > 0
    AND hd_dep_count / hd_vehicle_count > 1
),
filtered_sales AS (
  SELECT
    ss_ticket_number,
    ss_customer_sk
  FROM store_sales
  JOIN filtered_date ON store_sales.ss_sold_date_sk = filtered_date.d_date_sk
  JOIN filtered_store ON store_sales.ss_store_sk = filtered_store.s_store_sk
  JOIN filtered_hd ON store_sales.ss_hdemo_sk = filtered_hd.hd_demo_sk
),
ticket_counts AS (
  SELECT
    ss_ticket_number,
    ss_customer_sk,
    COUNT(*) AS cnt
  FROM filtered_sales
  GROUP BY ss_ticket_number, ss_customer_sk
  HAVING COUNT(*) BETWEEN 1 AND 5
)
SELECT
  c_last_name,
  c_first_name,
  c_salutation,
  c_preferred_cust_flag,
  ticket_counts.ss_ticket_number,
  ticket_counts.cnt
FROM ticket_counts
JOIN customer ON ticket_counts.ss_customer_sk = customer.c_customer_sk
ORDER BY ticket_counts.cnt DESC, customer.c_last_name ASC
```

Changes: Created three filtered dimension CTEs (date_dim, store, household_demographics) with selective predicates, then joined them with store_sales in a single pass CTE, followed by aggregation with early HAVING filter to reduce data before final customer join.

Expected speedup: 2.5-3.5x from reduced fact table scans via dimension pre-filtering and early aggregation pruning.
