```sql
WITH filtered_dimensions AS (
  SELECT 
    d_date_sk,
    d_year,
    d_dom
  FROM date_dim
  WHERE d_dom BETWEEN 1 AND 2
    AND d_year IN (1998, 1999, 2000)
),
filtered_store AS (
  SELECT 
    s_store_sk,
    s_city
  FROM store
  WHERE s_city IN ('Pleasant Hill', 'Five Points')
),
filtered_household AS (
  SELECT 
    hd_demo_sk
  FROM household_demographics
  WHERE hd_dep_count = 8 
     OR hd_vehicle_count = -1
),
prejoined_sales AS (
  SELECT 
    ss.ss_ticket_number,
    ss.ss_customer_sk,
    ss_addr.ca_city AS bought_city,
    ss.ss_ext_sales_price,
    ss.ss_ext_list_price,
    ss.ss_ext_tax
  FROM store_sales ss
  JOIN filtered_dimensions d ON ss.ss_sold_date_sk = d.d_date_sk
  JOIN filtered_store s ON ss.ss_store_sk = s.s_store_sk
  JOIN filtered_household hd ON ss.ss_hdemo_sk = hd.hd_demo_sk
  JOIN customer_address ss_addr ON ss.ss_addr_sk = ss_addr.ca_address_sk
),
aggregated_sales AS (
  SELECT 
    ss_ticket_number,
    ss_customer_sk,
    bought_city,
    SUM(ss_ext_sales_price) AS extended_price,
    SUM(ss_ext_list_price) AS list_price,
    SUM(ss_ext_tax) AS extended_tax
  FROM prejoined_sales
  GROUP BY 
    ss_ticket_number,
    ss_customer_sk,
    bought_city
)
SELECT
  c.c_last_name,
  c.c_first_name,
  cur_addr.ca_city,
  a.bought_city,
  a.ss_ticket_number,
  a.extended_price,
  a.extended_tax,
  a.list_price
FROM aggregated_sales a
JOIN customer c ON a.ss_customer_sk = c.c_customer_sk
JOIN customer_address cur_addr ON c.c_current_addr_sk = cur_addr.ca_address_sk
WHERE cur_addr.ca_city <> a.bought_city
ORDER BY
  c.c_last_name,
  a.ss_ticket_number
LIMIT 100
```

Changes: Isolated filtered dimension tables (date, store, household) into CTEs before joining with store_sales and sale address, enabling early filtering and reducing fact table scan. Performed aggregation only after filtering, then applied city inequality filter via EXISTS-equivalent anti-join (implied by <> condition) with customer's current address.

Expected speedup: 2-3x due to reduced fact table joins and early filtering before aggregation.
