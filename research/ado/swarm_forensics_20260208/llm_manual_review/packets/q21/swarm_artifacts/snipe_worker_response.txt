```sql
WITH filtered_dates AS (
    SELECT d_date_sk, d_date
    FROM date_dim
    WHERE d_date BETWEEN 
        (CAST('2002-02-27' AS DATE) - INTERVAL '30' DAY) AND
        (CAST('2002-02-27' AS DATE) + INTERVAL '30' DAY)
), filtered_items AS (
    SELECT i_item_sk, i_item_id
    FROM item
    WHERE i_current_price BETWEEN 0.99 AND 1.49
), inventory_agg AS (
    SELECT
        w.w_warehouse_name,
        fi.i_item_id,
        SUM(
            CASE
                WHEN fd.d_date < CAST('2002-02-27' AS DATE)
                THEN inv.inv_quantity_on_hand
                ELSE 0
            END
        ) AS inv_before,
        SUM(
            CASE
                WHEN fd.d_date >= CAST('2002-02-27' AS DATE)
                THEN inv.inv_quantity_on_hand
                ELSE 0
            END
        ) AS inv_after
    FROM inventory inv
    JOIN filtered_dates fd ON inv.inv_date_sk = fd.d_date_sk
    JOIN filtered_items fi ON inv.inv_item_sk = fi.i_item_sk
    JOIN warehouse w ON inv.inv_warehouse_sk = w.w_warehouse_sk
    GROUP BY w.w_warehouse_name, fi.i_item_id
)
SELECT *
FROM inventory_agg
WHERE (
    CASE WHEN inv_before > 0 THEN inv_after / inv_before ELSE NULL END
) BETWEEN 2.0 / 3.0 AND 3.0 / 2.0
ORDER BY w_warehouse_name, i_item_id
LIMIT 100
```

**Changes**: Applied multi-dimension prefetch by extracting date_dim and item filters into CTEs before joining with the large fact table (inventory). This reduces the fact table scan to only rows that match both dimension filters.

**Expected speedup**: 2.0-3.0x by reducing the inventory table scan through early filtering of selective dimensions.
