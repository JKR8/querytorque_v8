TABLES:
  catalog_sales
  customer
  customer_address
  customer_demographics
  date_dim
  item

JOINS:
  customer.c_current_addr_sk = customer_address.ca_address_sk
  catalog_sales.cs_bill_customer_sk = customer.c_customer_sk
  catalog_sales.cs_bill_cdemo_sk = customer_demographics.cd_demo_sk
  catalog_sales.cs_sold_date_sk = date_dim.d_date_sk
  catalog_sales.cs_item_sk = item.i_item_sk

FILTERS:
  item.i_category = 'Music'
  catalog_sales.cs_wholesale_cost BETWEEN 5 AND 10
  customer_address.ca_state IN ('AR', 'IN', 'VA')
  customer.c_birth_month = 12
  date_dim.d_year = 1999
  customer_demographics.cd_education_status = '4 yr Degree'
  customer_demographics.cd_gender = 'F'

COLUMNS NEEDED (table.column : type):
  catalog_sales: cs_bill_cdemo_sk (INTEGER), cs_bill_customer_sk (INTEGER), cs_coupon_amt (DECIMAL(7), cs_item_sk (INTEGER), cs_list_price (DECIMAL(7), cs_net_profit (DECIMAL(7), cs_quantity (INTEGER), cs_sales_price (DECIMAL(7), cs_sold_date_sk (INTEGER), cs_wholesale_cost (DECIMAL(7)
  customer: c_birth_month (INTEGER), c_birth_year (INTEGER), c_current_addr_sk (INTEGER), c_customer_sk (INTEGER)
  customer_address: ca_address_sk (INTEGER), ca_country (VARCHAR), ca_county (VARCHAR), ca_state (VARCHAR)
  customer_demographics: cd_demo_sk (INTEGER), cd_education_status (VARCHAR), cd_gender (VARCHAR)
  date_dim: d_date_sk (INTEGER), d_year (INTEGER)
  item: i_category (VARCHAR), i_item_id (VARCHAR), i_item_sk (INTEGER)
