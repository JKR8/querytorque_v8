TABLES:
  catalog_sales
  date_dim
  item

JOINS:
  date_dim.d_date_sk = catalog_sales.cs_sold_date_sk
  item.i_item_sk = catalog_sales.cs_item_sk
  catalog_sales.cs_item_sk = item.i_item_sk

FILTERS:
  date_dim.d_date BETWEEN '1998-01-06' AND CAST('1998-01-06' AS DATE) + INTERVAL '90' DAY
  item.i_manufact_id IN (47, 226, 612, 676, 818)
  item.i_manager_id BETWEEN 71 AND 100
  cs_sales_price / cs_list_price BETWEEN 23 * 0.01 AND 43 * 0.01
  catalog_sales.cs_list_price BETWEEN 108 AND 137

AGGREGATES:
  Correlated: catalog_sales.cs_ext_discount_amt > (SELECT 1.3 * AVG(cs_ext_discount_amt) WHERE date_dim.d_date_sk = catalog_sales.cs_sold_date_sk)

NOTES:
  Division in filter: cs_sales_price / cs_list_price — denominator must be > 0
  Division in filter: cs_sales_price / cs_list_price — denominator must be > 0

COLUMNS NEEDED (table.column : type):
  catalog_sales: cs_ext_discount_amt (DECIMAL(7), cs_item_sk (INTEGER), cs_list_price (DECIMAL(7), cs_sales_price (DECIMAL(7), cs_sold_date_sk (INTEGER)
  date_dim: d_date (DATE), d_date_sk (INTEGER)
  item: i_item_sk (INTEGER), i_manager_id (INTEGER), i_manufact_id (INTEGER)
