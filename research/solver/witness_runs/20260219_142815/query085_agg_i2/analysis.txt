TABLES:
  customer_demographics (alias: cd1)
  customer_demographics (alias: cd2)
  customer_address
  date_dim
  reason
  web_page
  web_returns
  web_sales

JOINS:
  reason.r_reason_sk = web_returns.wr_reason_sk
  customer_address.ca_address_sk = web_returns.wr_refunded_addr_sk
  customer_demographics.cd_demo_sk = web_returns.wr_returning_cdemo_sk
  customer_demographics.cd_demo_sk = web_returns.wr_refunded_cdemo_sk
  web_sales.ws_sold_date_sk = date_dim.d_date_sk
  web_sales.ws_order_number = web_returns.wr_order_number
  web_sales.ws_web_page_sk = web_page.wp_web_page_sk
  web_sales.ws_item_sk = web_returns.wr_item_sk

FILTERS:
  web_sales.ws_net_profit BETWEEN 50 AND 250
  web_sales.ws_sales_price BETWEEN 150.00 AND 200.00
  web_sales.ws_net_profit BETWEEN 100 AND 200
  web_sales.ws_net_profit BETWEEN 150 AND 300
  customer_address.ca_country = 'United States'
  customer_address.ca_state IN ('KS', 'OH', 'VA')
  date_dim.d_year = 2002
  web_sales.ws_sales_price BETWEEN 100.00 AND 150.00
  web_sales.ws_sales_price BETWEEN 50.00 AND 100.00
  customer_address.ca_state IN ('KY', 'NC', 'SD')
  customer_address.ca_state IN ('AR', 'GA', 'IN')
  customer_demographics.cd_education_status = '2 yr Degree'
  customer_demographics.cd_education_status = 'Primary'
  customer_demographics.cd_education_status = 'Secondary'
  customer_demographics.cd_marital_status = 'U'
  customer_demographics.cd_marital_status = 'S'
  customer_demographics.cd_marital_status = 'M'

AGGREGATES:
  GROUP BY: reason.r_reason_desc

COLUMNS NEEDED (table.column : type):
  customer_address: ca_address_sk (INTEGER), ca_country (VARCHAR), ca_state (VARCHAR)
  customer_demographics: cd_demo_sk (INTEGER), cd_education_status (VARCHAR), cd_marital_status (VARCHAR)
  date_dim: d_date_sk (INTEGER), d_year (INTEGER)
  reason: r_reason_desc (VARCHAR), r_reason_sk (INTEGER)
  web_page: wp_web_page_sk (INTEGER)
  web_returns: wr_fee (DECIMAL(7), wr_item_sk (INTEGER), wr_order_number (INTEGER), wr_reason_sk (INTEGER), wr_refunded_addr_sk (INTEGER), wr_refunded_cash (DECIMAL(7), wr_refunded_cdemo_sk (INTEGER), wr_returning_cdemo_sk (INTEGER)
  web_sales: ws_item_sk (INTEGER), ws_net_profit (DECIMAL(7), ws_order_number (INTEGER), ws_quantity (INTEGER), ws_sales_price (DECIMAL(7), ws_sold_date_sk (INTEGER), ws_web_page_sk (INTEGER)
