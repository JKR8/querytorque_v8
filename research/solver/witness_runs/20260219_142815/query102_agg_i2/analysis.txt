TABLES:
  customer
  customer_address
  customer_demographics
  date_dim (alias: d1)
  date_dim (alias: d2)
  household_demographics
  inventory
  item
  store
  store_sales
  warehouse
  web_sales

JOINS:
  customer.c_current_addr_sk = customer_address.ca_address_sk
  customer.c_current_hdemo_sk = household_demographics.hd_demo_sk
  customer.c_current_cdemo_sk = customer_demographics.cd_demo_sk
  store.s_state = warehouse.w_state
  inventory.inv_date_sk = store_sales.ss_sold_date_sk
  inventory.inv_item_sk = store_sales.ss_item_sk
  web_sales.ws_warehouse_sk = warehouse.w_warehouse_sk
  web_sales.ws_warehouse_sk = inventory.inv_warehouse_sk
  web_sales.ws_bill_customer_sk = customer.c_customer_sk
  store_sales.ss_customer_sk = customer.c_customer_sk
  web_sales.ws_sold_date_sk = date_dim.d_date_sk
  store_sales.ss_sold_date_sk = date_dim.d_date_sk
  store_sales.ss_item_sk = item.i_item_sk
  web_sales.ws_item_sk = store_sales.ss_item_sk

FILTERS:
  web_sales.ws_wholesale_cost BETWEEN 76 AND 96
  date_dim.d_year = 1998
  customer_address.ca_state IN ('AR', 'CA', 'KS', 'NY', 'VA')
  item.i_manager_id IN (6, 11, 16, 17, 19, 28, 47, 82, 88, 98)
  item.i_category IN ('Men', 'Shoes', 'Sports')
  inventory.inv_quantity_on_hand >= store_sales.ss_quantity
  date_dim.d_date BETWEEN date_dim.d_date AND (d1.d_date + INTERVAL '30' DAY)

AGGREGATES:
  GROUP BY: customer_demographics.cd_gender, customer_demographics.cd_marital_status, customer_demographics.cd_education_status, household_demographics.hd_vehicle_count

COLUMNS NEEDED (table.column : type):
  customer: c_current_addr_sk (INTEGER), c_current_cdemo_sk (INTEGER), c_current_hdemo_sk (INTEGER), c_customer_sk (INTEGER)
  customer_address: ca_address_sk (INTEGER), ca_state (VARCHAR)
  customer_demographics: cd_demo_sk (INTEGER), cd_education_status (VARCHAR), cd_gender (VARCHAR), cd_marital_status (VARCHAR)
  date_dim: d_date (DATE), d_date_sk (INTEGER), d_year (INTEGER)
  household_demographics: hd_demo_sk (INTEGER), hd_vehicle_count (INTEGER)
  inventory: inv_date_sk (INTEGER), inv_item_sk (INTEGER), inv_quantity_on_hand (INTEGER), inv_warehouse_sk (INTEGER)
  item: i_category (VARCHAR), i_item_sk (INTEGER), i_manager_id (INTEGER)
  store: s_state (VARCHAR)
  store_sales: ss_customer_sk (INTEGER), ss_item_sk (INTEGER), ss_quantity (INTEGER), ss_sold_date_sk (INTEGER)
  warehouse: w_state (VARCHAR), w_warehouse_sk (INTEGER)
  web_sales: ws_bill_customer_sk (INTEGER), ws_item_sk (INTEGER), ws_sold_date_sk (INTEGER), ws_warehouse_sk (INTEGER), ws_wholesale_cost (DECIMAL(7)
