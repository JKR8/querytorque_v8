TABLES:
  call_center
  catalog_sales
  date_dim
  ship_mode
  warehouse

JOINS:
  catalog_sales.cs_call_center_sk = call_center.cc_call_center_sk
  catalog_sales.cs_ship_mode_sk = ship_mode.sm_ship_mode_sk
  catalog_sales.cs_warehouse_sk = warehouse.w_warehouse_sk
  catalog_sales.cs_ship_date_sk = date_dim.d_date_sk

FILTERS:
  warehouse.w_gmt_offset = -5
  call_center.cc_class = 'small'
  ship_mode.sm_type = 'EXPRESS'
  catalog_sales.cs_list_price BETWEEN 86 AND 115
  date_dim.d_month_seq BETWEEN 1192 AND 1192 + 23

AGGREGATES:
  GROUP BY: SUBSTRING(w_warehouse_name, 1, 20), ship_mode.sm_type, call_center.cc_name

COLUMNS NEEDED (table.column : type):
  call_center: cc_call_center_sk (INTEGER), cc_class (VARCHAR), cc_name (VARCHAR)
  catalog_sales: cs_call_center_sk (INTEGER), cs_list_price (DECIMAL(7), cs_ship_date_sk (INTEGER), cs_ship_mode_sk (INTEGER), cs_sold_date_sk (INTEGER), cs_warehouse_sk (INTEGER)
  date_dim: d_date_sk (INTEGER), d_month_seq (INTEGER)
  ship_mode: sm_ship_mode_sk (INTEGER), sm_type (VARCHAR)
  warehouse: w_gmt_offset (DECIMAL(5), w_warehouse_name (VARCHAR), w_warehouse_sk (INTEGER)
