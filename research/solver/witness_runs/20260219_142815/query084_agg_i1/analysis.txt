TABLES:
  customer
  customer_address
  customer_demographics
  household_demographics
  income_band
  store_returns

JOINS:
  store_returns.sr_cdemo_sk = customer_demographics.cd_demo_sk
  household_demographics.hd_demo_sk = customer.c_current_hdemo_sk
  customer_demographics.cd_demo_sk = customer.c_current_cdemo_sk
  income_band.ib_income_band_sk = household_demographics.hd_income_band_sk
  customer.c_current_addr_sk = customer_address.ca_address_sk

FILTERS:
  income_band.ib_upper_bound <= 23567 + 50000
  income_band.ib_lower_bound >= 23567
  customer_address.ca_city = 'Jackson'

COLUMNS NEEDED (table.column : type):
  customer: c_current_addr_sk (INTEGER), c_current_cdemo_sk (INTEGER), c_current_hdemo_sk (INTEGER), c_customer_id (VARCHAR), c_first_name (VARCHAR), c_last_name (VARCHAR)
  customer_address: ca_address_sk (INTEGER), ca_city (VARCHAR)
  customer_demographics: cd_demo_sk (INTEGER)
  household_demographics: hd_demo_sk (INTEGER), hd_income_band_sk (INTEGER)
  income_band: ib_income_band_sk (INTEGER), ib_lower_bound (INTEGER), ib_upper_bound (INTEGER)
  store_returns: sr_cdemo_sk (INTEGER)
