TABLES:
  customer (alias: c)
  customer_address (alias: ca)
  catalog_sales
  customer_demographics
  date_dim
  item
  store_sales
  web_sales

JOINS:
  customer_demographics.cd_demo_sk = customer.c_current_cdemo_sk
  customer.c_current_addr_sk = customer_address.ca_address_sk
  store_sales.ss_item_sk = item.i_item_sk
  web_sales.ws_item_sk = item.i_item_sk
  catalog_sales.cs_item_sk = item.i_item_sk
  customer.c_customer_sk = store_sales.ss_customer_sk
  store_sales.ss_sold_date_sk = date_dim.d_date_sk
  customer.c_customer_sk = web_sales.ws_bill_customer_sk
  web_sales.ws_sold_date_sk = date_dim.d_date_sk
  customer.c_customer_sk = catalog_sales.cs_ship_customer_sk
  catalog_sales.cs_sold_date_sk = date_dim.d_date_sk

FILTERS:
  customer_demographics.cd_gender = 'M'
  customer_demographics.cd_education_status IN ('College', 'Secondary', 'College')
  customer_demographics.cd_marital_status IN ('U', 'S', 'U')
  item.i_manager_id BETWEEN 91 AND 100
  customer.c_birth_month IN (4, 5)
  ss_sales_price / ss_list_price BETWEEN 65 * 0.01 AND 75 * 0.01
  ws_sales_price / ws_list_price BETWEEN 65 * 0.01 AND 75 * 0.01
  cs_sales_price / cs_list_price BETWEEN 65 * 0.01 AND 75 * 0.01
  customer_address.ca_county IN ('Alameda County', 'Lexington city', 'Pender County', 'Petroleum County', 'Walworth County')
  item.i_category IN ('Children', 'Home', 'Women')
  date_dim.d_moy BETWEEN 5 AND 5 + 3
  date_dim.d_year = 1999

AGGREGATES:
  GROUP BY: customer_demographics.cd_gender, customer_demographics.cd_marital_status, customer_demographics.cd_education_status, customer_demographics.cd_purchase_estimate, customer_demographics.cd_credit_rating, customer_demographics.cd_dep_count, customer_demographics.cd_dep_employed_count, customer_demographics.cd_dep_college_count

NOTES:
  Division in filter: ss_sales_price / ss_list_price — denominator must be > 0
  Division in filter: ws_sales_price / ws_list_price — denominator must be > 0
  Division in filter: cs_sales_price / cs_list_price — denominator must be > 0
  Division in filter: ss_sales_price / ss_list_price — denominator must be > 0
  Division in filter: ws_sales_price / ws_list_price — denominator must be > 0
  Division in filter: cs_sales_price / cs_list_price — denominator must be > 0

COLUMNS NEEDED (table.column : type):
  catalog_sales: cs_item_sk (INTEGER), cs_list_price (DECIMAL(7), cs_sales_price (DECIMAL(7), cs_ship_customer_sk (INTEGER), cs_sold_date_sk (INTEGER)
  customer: c_birth_month (INTEGER), c_current_addr_sk (INTEGER), c_current_cdemo_sk (INTEGER), c_customer_sk (INTEGER)
  customer_address: ca_address_sk (INTEGER), ca_county (VARCHAR)
  customer_demographics: cd_credit_rating (VARCHAR), cd_demo_sk (INTEGER), cd_dep_college_count (INTEGER), cd_dep_count (INTEGER), cd_dep_employed_count (INTEGER), cd_education_status (VARCHAR), cd_gender (VARCHAR), cd_marital_status (VARCHAR), cd_purchase_estimate (INTEGER)
  date_dim: d_date_sk (INTEGER), d_moy (INTEGER), d_year (INTEGER)
  item: i_category (VARCHAR), i_item_sk (INTEGER), i_manager_id (INTEGER)
  store_sales: ss_customer_sk (INTEGER), ss_item_sk (INTEGER), ss_list_price (DECIMAL(7), ss_sales_price (DECIMAL(7), ss_sold_date_sk (INTEGER)
  web_sales: ws_bill_customer_sk (INTEGER), ws_item_sk (INTEGER), ws_list_price (DECIMAL(7), ws_sales_price (DECIMAL(7), ws_sold_date_sk (INTEGER)
