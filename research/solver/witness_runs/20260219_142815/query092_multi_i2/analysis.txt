TABLES:
  date_dim
  item
  web_sales

JOINS:
  date_dim.d_date_sk = web_sales.ws_sold_date_sk
  item.i_item_sk = web_sales.ws_item_sk
  web_sales.ws_item_sk = item.i_item_sk

FILTERS:
  web_sales.ws_wholesale_cost BETWEEN 35 AND 55
  date_dim.d_date BETWEEN '1999-01-29' AND CAST('1999-01-29' AS DATE) + INTERVAL '90' DAY
  ws_sales_price / ws_list_price BETWEEN 76 * 0.01 AND 91 * 0.01
  item.i_manufact_id BETWEEN 797 AND 996
  item.i_category IN ('Men', 'Shoes', 'Sports')

AGGREGATES:
  Correlated: web_sales.ws_ext_discount_amt > (SELECT 1.3 * AVG(ws_ext_discount_amt) WHERE date_dim.d_date_sk = web_sales.ws_sold_date_sk)

NOTES:
  Division in filter: ws_sales_price / ws_list_price — denominator must be > 0
  Division in filter: ws_sales_price / ws_list_price — denominator must be > 0

COLUMNS NEEDED (table.column : type):
  date_dim: d_date (DATE), d_date_sk (INTEGER)
  item: i_category (VARCHAR), i_item_sk (INTEGER), i_manufact_id (INTEGER)
  web_sales: ws_ext_discount_amt (DECIMAL(7), ws_item_sk (INTEGER), ws_list_price (DECIMAL(7), ws_sales_price (DECIMAL(7), ws_sold_date_sk (INTEGER), ws_wholesale_cost (DECIMAL(7)
