TABLES:
  customer_address
  customer_demographics
  date_dim
  household_demographics
  store
  store_sales

JOINS:
  store.s_store_sk = store_sales.ss_store_sk
  store_sales.ss_sold_date_sk = date_dim.d_date_sk
  store_sales.ss_addr_sk = customer_address.ca_address_sk
  store_sales.ss_hdemo_sk = household_demographics.hd_demo_sk
  customer_demographics.cd_demo_sk = store_sales.ss_cdemo_sk

FILTERS:
  date_dim.d_year = 2001
  store_sales.ss_net_profit BETWEEN 50 AND 250
  household_demographics.hd_dep_count = 1
  store_sales.ss_net_profit BETWEEN 100 AND 200
  store_sales.ss_net_profit BETWEEN 150 AND 300
  customer_address.ca_state IN ('IA', 'IL', 'NC')
  household_demographics.hd_dep_count = 3
  store_sales.ss_sales_price BETWEEN 150.00 AND 200.00
  customer_address.ca_state IN ('CO', 'NC', 'TX')
  customer_address.ca_state IN ('AR', 'NY', 'TX')
  customer_address.ca_country = 'United States'
  store_sales.ss_sales_price BETWEEN 100.00 AND 150.00
  store_sales.ss_sales_price BETWEEN 50.00 AND 100.00
  customer_demographics.cd_education_status = 'Unknown'
  customer_demographics.cd_education_status = '2 yr Degree'
  customer_demographics.cd_education_status = 'College'
  customer_demographics.cd_marital_status = 'S'
  customer_demographics.cd_marital_status = 'M'
  customer_demographics.cd_marital_status = 'U'

COLUMNS NEEDED (table.column : type):
  customer_address: ca_address_sk (INTEGER), ca_country (VARCHAR), ca_state (VARCHAR)
  customer_demographics: cd_demo_sk (INTEGER), cd_education_status (VARCHAR), cd_marital_status (VARCHAR)
  date_dim: d_date_sk (INTEGER), d_year (INTEGER)
  household_demographics: hd_demo_sk (INTEGER), hd_dep_count (INTEGER)
  store: s_store_sk (INTEGER)
  store_sales: ss_addr_sk (INTEGER), ss_cdemo_sk (INTEGER), ss_ext_sales_price (DECIMAL(7), ss_ext_wholesale_cost (DECIMAL(7), ss_hdemo_sk (INTEGER), ss_net_profit (DECIMAL(7), ss_quantity (INTEGER), ss_sales_price (DECIMAL(7), ss_sold_date_sk (INTEGER), ss_store_sk (INTEGER)
