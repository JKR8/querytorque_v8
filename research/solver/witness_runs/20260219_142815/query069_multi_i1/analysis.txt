TABLES:
  customer (alias: c)
  customer_address (alias: ca)
  catalog_sales
  customer_demographics
  date_dim
  store_sales
  web_sales

JOINS:
  customer_demographics.cd_demo_sk = customer.c_current_cdemo_sk
  customer.c_current_addr_sk = customer_address.ca_address_sk
  customer.c_customer_sk = store_sales.ss_customer_sk
  store_sales.ss_sold_date_sk = date_dim.d_date_sk
  customer.c_customer_sk = web_sales.ws_bill_customer_sk
  web_sales.ws_sold_date_sk = date_dim.d_date_sk
  customer.c_customer_sk = catalog_sales.cs_ship_customer_sk
  catalog_sales.cs_sold_date_sk = date_dim.d_date_sk

FILTERS:
  customer_demographics.cd_education_status IN ('Primary', 'College')
  customer_demographics.cd_marital_status IN ('S', 'M', 'U')
  customer_address.ca_state IN ('CO', 'NC', 'TX')
  store_sales.ss_list_price BETWEEN 80 AND 169
  date_dim.d_moy BETWEEN 10 AND 10 + 2
  date_dim.d_year = 2002
  web_sales.ws_list_price BETWEEN 80 AND 169
  catalog_sales.cs_list_price BETWEEN 80 AND 169

AGGREGATES:
  GROUP BY: customer_demographics.cd_gender, customer_demographics.cd_marital_status, customer_demographics.cd_education_status, customer_demographics.cd_purchase_estimate, customer_demographics.cd_credit_rating

COLUMNS NEEDED (table.column : type):
  catalog_sales: cs_list_price (DECIMAL(7), cs_ship_customer_sk (INTEGER), cs_sold_date_sk (INTEGER)
  customer: c_current_addr_sk (INTEGER), c_current_cdemo_sk (INTEGER), c_customer_sk (INTEGER)
  customer_address: ca_address_sk (INTEGER), ca_state (VARCHAR)
  customer_demographics: cd_credit_rating (VARCHAR), cd_demo_sk (INTEGER), cd_education_status (VARCHAR), cd_gender (VARCHAR), cd_marital_status (VARCHAR), cd_purchase_estimate (INTEGER)
  date_dim: d_date_sk (INTEGER), d_moy (INTEGER), d_year (INTEGER)
  store_sales: ss_customer_sk (INTEGER), ss_list_price (DECIMAL(7), ss_sold_date_sk (INTEGER)
  web_sales: ws_bill_customer_sk (INTEGER), ws_list_price (DECIMAL(7), ws_sold_date_sk (INTEGER)
