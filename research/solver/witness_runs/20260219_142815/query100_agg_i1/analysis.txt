TABLES:
  customer
  customer_address
  customer_demographics
  date_dim
  item (alias: item1)
  item (alias: item2)
  store_sales (alias: s1)
  store_sales (alias: s2)

JOINS:
  date_dim.d_date_sk = store_sales.ss_sold_date_sk
  customer.c_current_cdemo_sk = customer_demographics.cd_demo_sk
  customer.c_current_addr_sk = customer_address.ca_address_sk
  store_sales.ss_customer_sk = customer.c_customer_sk
  store_sales.ss_item_sk = item.i_item_sk

FILTERS:
  store_sales.ss_list_price BETWEEN 108 AND 122
  customer_demographics.cd_education_status = 'College'
  customer_demographics.cd_marital_status = 'S'
  item.i_manager_id BETWEEN 81 AND 100
  item.i_category IN ('Electronics', 'Jewelry')
  date_dim.d_year BETWEEN 1998 AND 1998 + 1
  item.i_item_sk < item.i_item_sk

AGGREGATES:
  GROUP BY: item.i_item_sk, item.i_item_sk

COLUMNS NEEDED (table.column : type):
  customer: c_current_addr_sk (INTEGER), c_current_cdemo_sk (INTEGER), c_customer_sk (INTEGER)
  customer_address: ca_address_sk (INTEGER)
  customer_demographics: cd_demo_sk (INTEGER), cd_education_status (VARCHAR), cd_marital_status (VARCHAR)
  date_dim: d_date_sk (INTEGER), d_year (INTEGER)
  item: i_category (VARCHAR), i_item_sk (INTEGER), i_manager_id (INTEGER)
  store_sales: ss_customer_sk (INTEGER), ss_item_sk (INTEGER), ss_list_price (DECIMAL(7), ss_sold_date_sk (INTEGER), ss_ticket_number (INTEGER)
