TABLES:
  catalog_returns
  customer
  customer_address
  date_dim

JOINS:
  customer_total_return.ctr_customer_sk = customer.c_customer_sk
  customer_address.ca_address_sk = customer.c_current_addr_sk
  catalog_returns.cr_returning_addr_sk = customer_address.ca_address_sk
  catalog_returns.cr_returned_date_sk = date_dim.d_date_sk

FILTERS:
  customer_address.ca_state = 'IA'
  date_dim.d_year = 2000

AGGREGATES:
  Correlated: customer_total_return.ctr_total_return > (SELECT AVG(ctr_total_return) * 1.2 WHERE customer_total_return.ctr_state = customer_total_return.ctr_state)
  GROUP BY: catalog_returns.cr_returning_customer_sk, customer_address.ca_state

CTEs:
  customer_total_return:
    FROM: catalog_returns, date_dim, customer_address
    WHERE: catalog_returns.cr_returning_addr_sk = customer_address.ca_address_sk (JOIN), catalog_returns.cr_returned_date_sk = date_dim.d_date_sk (JOIN), date_dim.d_year = 2000
    GROUP BY: catalog_returns.cr_returning_customer_sk, customer_address.ca_state
    OUTPUT: ctr_customer_sk = catalog_returns.cr_returning_customer_sk, ctr_state = customer_address.ca_state, ctr_total_return = SUM(cr_return_amt_inc_tax)

COLUMNS NEEDED (table.column : type):
  catalog_returns: cr_return_amt_inc_tax (DECIMAL(7), cr_returned_date_sk (INTEGER), cr_returning_addr_sk (INTEGER), cr_returning_customer_sk (INTEGER)
  customer: c_current_addr_sk (INTEGER), c_customer_id (VARCHAR), c_customer_sk (INTEGER), c_first_name (VARCHAR), c_last_name (VARCHAR), c_salutation (VARCHAR)
  customer_address: ca_address_sk (INTEGER), ca_city (VARCHAR), ca_country (VARCHAR), ca_county (VARCHAR), ca_gmt_offset (DECIMAL(5), ca_location_type (VARCHAR), ca_state (VARCHAR), ca_street_name (VARCHAR), ca_street_number (VARCHAR), ca_street_type (VARCHAR), ca_suite_number (VARCHAR), ca_zip (VARCHAR)
  date_dim: d_date_sk (INTEGER), d_year (INTEGER)
