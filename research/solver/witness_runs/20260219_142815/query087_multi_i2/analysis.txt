TABLES:
  catalog_sales
  customer
  date_dim
  store_sales
  web_sales

JOINS:
  web_sales.ws_sold_date_sk = date_dim.d_date_sk
  web_sales.ws_bill_customer_sk = customer.c_customer_sk
  store_sales.ss_sold_date_sk = date_dim.d_date_sk
  store_sales.ss_customer_sk = customer.c_customer_sk
  catalog_sales.cs_sold_date_sk = date_dim.d_date_sk
  catalog_sales.cs_bill_customer_sk = customer.c_customer_sk

FILTERS:
  web_sales.ws_wholesale_cost BETWEEN 76 AND 86
  customer.c_birth_year BETWEEN 1968 AND 1974
  web_sales.ws_list_price BETWEEN 168 AND 197
  date_dim.d_month_seq BETWEEN 1213 AND 1213 + 11
  store_sales.ss_wholesale_cost BETWEEN 76 AND 86
  store_sales.ss_list_price BETWEEN 168 AND 197
  catalog_sales.cs_wholesale_cost BETWEEN 76 AND 86
  catalog_sales.cs_list_price BETWEEN 168 AND 197

COLUMNS NEEDED (table.column : type):
  catalog_sales: cs_bill_customer_sk (INTEGER), cs_list_price (DECIMAL(7), cs_sold_date_sk (INTEGER), cs_wholesale_cost (DECIMAL(7)
  customer: c_birth_year (INTEGER), c_customer_sk (INTEGER)
  date_dim: d_date_sk (INTEGER), d_month_seq (INTEGER)
  store_sales: ss_customer_sk (INTEGER), ss_list_price (DECIMAL(7), ss_sold_date_sk (INTEGER), ss_wholesale_cost (DECIMAL(7)
  web_sales: ws_bill_customer_sk (INTEGER), ws_list_price (DECIMAL(7), ws_sold_date_sk (INTEGER), ws_wholesale_cost (DECIMAL(7)
