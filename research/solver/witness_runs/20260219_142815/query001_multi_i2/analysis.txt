TABLES:
  customer
  customer_demographics
  date_dim
  store
  store_returns

JOINS:
  customer.c_current_cdemo_sk = customer_demographics.cd_demo_sk
  customer_total_return.ctr_customer_sk = customer.c_customer_sk
  store.s_store_sk = customer_total_return.ctr_store_sk
  store_returns.sr_returned_date_sk = date_dim.d_date_sk

FILTERS:
  customer.c_birth_year BETWEEN 1987 AND 1993
  customer.c_birth_month = 4
  customer_demographics.cd_gender = 'M'
  customer_demographics.cd_education_status IN ('4 yr Degree', '4 yr Degree')
  customer_demographics.cd_marital_status IN ('S', 'S')
  store.s_state IN ('IA', 'KY', 'NE')
  customer_total_return.ctr_reason_sk BETWEEN 17 AND 20
  sr_return_amt / sr_return_quantity BETWEEN 115 AND 174
  date_dim.d_year = 2000

AGGREGATES:
  Correlated: customer_total_return.ctr_total_return > (SELECT AVG(ctr_total_return) * 1.2 WHERE customer_total_return.ctr_store_sk = customer_total_return.ctr_store_sk)
  GROUP BY: store_returns.sr_customer_sk, store_returns.sr_store_sk, store_returns.sr_reason_sk

CTEs:
  customer_total_return:
    FROM: store_returns, date_dim
    WHERE: sr_return_amt / sr_return_quantity BETWEEN 115 AND 174, store_returns.sr_returned_date_sk = date_dim.d_date_sk (JOIN), date_dim.d_year = 2000
    GROUP BY: store_returns.sr_customer_sk, store_returns.sr_store_sk, store_returns.sr_reason_sk
    OUTPUT: ctr_customer_sk = store_returns.sr_customer_sk, ctr_store_sk = store_returns.sr_store_sk, ctr_reason_sk = store_returns.sr_reason_sk, ctr_total_return = SUM(SR_RETURN_AMT_INC_TAX)

NOTES:
  Division in filter: sr_return_amt / sr_return_quantity â€” denominator must be > 0
