TABLES:
  customer_demographics
  date_dim
  item
  store
  store_sales

JOINS:
  store_sales.ss_cdemo_sk = customer_demographics.cd_demo_sk
  store_sales.ss_store_sk = store.s_store_sk
  store_sales.ss_sold_date_sk = date_dim.d_date_sk
  store_sales.ss_item_sk = item.i_item_sk

FILTERS:
  item.i_category = 'Music'
  store.s_state = 'OH'
  date_dim.d_year = 1999
  customer_demographics.cd_education_status = 'Secondary'
  customer_demographics.cd_marital_status = 'W'
  customer_demographics.cd_gender = 'M'

COLUMNS NEEDED (table.column : type):
  customer_demographics: cd_demo_sk (INTEGER), cd_education_status (VARCHAR), cd_gender (VARCHAR), cd_marital_status (VARCHAR)
  date_dim: d_date_sk (INTEGER), d_year (INTEGER)
  item: i_category (VARCHAR), i_item_id (VARCHAR), i_item_sk (INTEGER)
  store: s_state (VARCHAR), s_store_sk (INTEGER)
  store_sales: ss_cdemo_sk (INTEGER), ss_coupon_amt (DECIMAL(7), ss_item_sk (INTEGER), ss_list_price (DECIMAL(7), ss_quantity (INTEGER), ss_sales_price (DECIMAL(7), ss_sold_date_sk (INTEGER), ss_store_sk (INTEGER)
