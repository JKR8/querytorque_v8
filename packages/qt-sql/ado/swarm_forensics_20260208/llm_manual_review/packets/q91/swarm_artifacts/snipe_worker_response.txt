```sql
WITH filtered_dates AS (
    SELECT d_date_sk
    FROM date_dim
    WHERE d_year = 2001
      AND d_moy = 11
),
filtered_customer_demo AS (
    SELECT cd_demo_sk, cd_marital_status, cd_education_status
    FROM customer_demographics
    WHERE (cd_marital_status = 'M' AND cd_education_status = 'Unknown')
       OR (cd_marital_status = 'W' AND cd_education_status = 'Advanced Degree')
),
filtered_household_demo AS (
    SELECT hd_demo_sk
    FROM household_demographics
    WHERE hd_buy_potential LIKE '1001-5000%'
),
filtered_address AS (
    SELECT ca_address_sk
    FROM customer_address
    WHERE ca_gmt_offset = -6
),
filtered_customers AS (
    SELECT c_customer_sk, cd_marital_status, cd_education_status
    FROM customer
    JOIN filtered_customer_demo ON cd_demo_sk = c_current_cdemo_sk
    JOIN filtered_household_demo ON hd_demo_sk = c_current_hdemo_sk
    JOIN filtered_address ON ca_address_sk = c_current_addr_sk
)
SELECT
    cc_call_center_id AS Call_Center,
    cc_name AS Call_Center_Name,
    cc_manager AS Manager,
    SUM(cr_net_loss) AS Returns_Loss
FROM catalog_returns
JOIN filtered_dates ON cr_returned_date_sk = d_date_sk
JOIN filtered_customers ON cr_returning_customer_sk = c_customer_sk
JOIN call_center ON cr_call_center_sk = cc_call_center_sk
GROUP BY
    cc_call_center_id,
    cc_name,
    cc_manager,
    cd_marital_status,
    cd_education_status
ORDER BY
    SUM(cr_net_loss) DESC
```

**Changes**: Created CTEs to pre-filter dimension tables (date_dim, customer_demographics, household_demographics, customer_address) before joining with the large fact table. Combined filtered dimensions into a single customer CTE to reduce join fanout early.

**Expected speedup**: 2.0x+ via early reduction of fact table rows through selective dimension filtering and reduced join cardinality.
