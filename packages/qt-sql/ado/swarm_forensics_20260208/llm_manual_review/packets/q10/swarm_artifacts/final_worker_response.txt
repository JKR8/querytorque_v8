```sql
WITH filtered_date AS (
    SELECT d_date_sk
    FROM date_dim
    WHERE d_year = 2001
      AND d_moy BETWEEN 1 AND 4
),
filtered_address AS (
    SELECT ca_address_sk
    FROM customer_address
    WHERE ca_county IN (
        'Storey County',
        'Marquette County',
        'Warren County',
        'Cochran County',
        'Kandiyohi County'
    )
),
candidate_customers AS (
    SELECT 
        c.c_customer_sk,
        cd.cd_gender,
        cd.cd_marital_status,
        cd.cd_education_status,
        cd.cd_purchase_estimate,
        cd.cd_credit_rating,
        cd.cd_dep_count,
        cd.cd_dep_employed_count,
        cd.cd_dep_college_count
    FROM customer c
    INNER JOIN filtered_address fa ON c.c_current_addr_sk = fa.ca_address_sk
    INNER JOIN customer_demographics cd ON cd.cd_demo_sk = c.c_current_cdemo_sk
),
store_customers AS (
    SELECT DISTINCT ss_customer_sk
    FROM store_sales
    INNER JOIN filtered_date fd ON store_sales.ss_sold_date_sk = fd.d_date_sk
),
web_customers AS (
    SELECT DISTINCT ws_bill_customer_sk
    FROM web_sales
    INNER JOIN filtered_date fd ON web_sales.ws_sold_date_sk = fd.d_date_sk
),
catalog_customers AS (
    SELECT DISTINCT cs_ship_customer_sk
    FROM catalog_sales
    INNER JOIN filtered_date fd ON catalog_sales.cs_sold_date_sk = fd.d_date_sk
),
qualified_customers AS (
    SELECT 
        cc.*,
        CASE WHEN sc.ss_customer_sk IS NOT NULL THEN 1 ELSE 0 END AS has_store_sale,
        CASE WHEN wc.ws_bill_customer_sk IS NOT NULL THEN 1 ELSE 0 END AS has_web_sale,
        CASE WHEN cat.cs_ship_customer_sk IS NOT NULL THEN 1 ELSE 0 END AS has_catalog_sale
    FROM candidate_customers cc
    LEFT JOIN store_customers sc ON cc.c_customer_sk = sc.ss_customer_sk
    LEFT JOIN web_customers wc ON cc.c_customer_sk = wc.ws_bill_customer_sk
    LEFT JOIN catalog_customers cat ON cc.c_customer_sk = cat.cs_ship_customer_sk
    WHERE sc.ss_customer_sk IS NOT NULL
      AND (wc.ws_bill_customer_sk IS NOT NULL OR cat.cs_ship_customer_sk IS NOT NULL)
)
SELECT 
    cd_gender,
    cd_marital_status,
    cd_education_status,
    COUNT(*) AS cnt1,
    cd_purchase_estimate,
    COUNT(*) AS cnt2,
    cd_credit_rating,
    COUNT(*) AS cnt3,
    cd_dep_count,
    COUNT(*) AS cnt4,
    cd_dep_employed_count,
    COUNT(*) AS cnt5,
    cd_dep_college_count,
    COUNT(*) AS cnt6
FROM qualified_customers
GROUP BY
    cd_gender,
    cd_marital_status,
    cd_education_status,
    cd_purchase_estimate,
    cd_credit_rating,
    cd_dep_count,
    cd_dep_employed_count,
    cd_dep_college_count
ORDER BY
    cd_gender,
    cd_marital_status,
    cd_education_status,
    cd_purchase_estimate,
    cd_credit_rating,
    cd_dep_count,
    cd_dep_employed_count,
    cd_dep_college_count
LIMIT 100;
```

**Changes:** Created CTEs to filter date_dim and customer_address early, built a small candidate customer set with demographics, computed store/web/catalog sales indicators via distinct semi-joins, filtered to customers meeting both conditions (store + (web OR catalog)), then aggregated once on the qualified set.

**Expected speedup:** ~3-4x by reducing fact table scans to only relevant date periods and pushing customer filters into semi-joins via pre-computed customer sets.
