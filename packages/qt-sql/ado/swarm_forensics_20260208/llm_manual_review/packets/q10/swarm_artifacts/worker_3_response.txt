```sql
WITH filtered_dates AS (
    SELECT d_date_sk
    FROM date_dim
    WHERE d_year = 2001
      AND d_moy BETWEEN 1 AND 1 + 3
),
filtered_addresses AS (
    SELECT ca_address_sk
    FROM customer_address
    WHERE ca_county IN (
        'Storey County',
        'Marquette County',
        'Warren County',
        'Cochran County',
        'Kandiyohi County'
    )
),
customers_with_store_sales AS (
    SELECT DISTINCT ss_customer_sk
    FROM store_sales
    INNER JOIN filtered_dates ON store_sales.ss_sold_date_sk = filtered_dates.d_date_sk
),
customers_with_web_sales AS (
    SELECT DISTINCT ws_bill_customer_sk
    FROM web_sales
    INNER JOIN filtered_dates ON web_sales.ws_sold_date_sk = filtered_dates.d_date_sk
),
customers_with_catalog_sales AS (
    SELECT DISTINCT cs_ship_customer_sk
    FROM catalog_sales
    INNER JOIN filtered_dates ON catalog_sales.cs_sold_date_sk = filtered_dates.d_date_sk
),
filtered_customers AS (
    SELECT
        c.c_customer_sk,
        c.c_current_cdemo_sk
    FROM customer AS c
    INNER JOIN filtered_addresses fa ON c.c_current_addr_sk = fa.ca_address_sk
    INNER JOIN customers_with_store_sales cs ON c.c_customer_sk = cs.ss_customer_sk
    WHERE EXISTS (
        SELECT 1
        FROM customers_with_web_sales
        WHERE ws_bill_customer_sk = c.c_customer_sk
    ) OR EXISTS (
        SELECT 1
        FROM customers_with_catalog_sales
        WHERE cs_ship_customer_sk = c.c_customer_sk
    )
)
SELECT
    cd.cd_gender,
    cd.cd_marital_status,
    cd.cd_education_status,
    COUNT(*) AS cnt1,
    cd.cd_purchase_estimate,
    COUNT(*) AS cnt2,
    cd.cd_credit_rating,
    COUNT(*) AS cnt3,
    cd.cd_dep_count,
    COUNT(*) AS cnt4,
    cd.cd_dep_employed_count,
    COUNT(*) AS cnt5,
    cd.cd_dep_college_count,
    COUNT(*) AS cnt6
FROM filtered_customers fc
INNER JOIN customer_demographics cd ON fc.c_current_cdemo_sk = cd.cd_demo_sk
GROUP BY
    cd.cd_gender,
    cd.cd_marital_status,
    cd.cd_education_status,
    cd.cd_purchase_estimate,
    cd.cd_credit_rating,
    cd.cd_dep_count,
    cd.cd_dep_employed_count,
    cd.cd_dep_college_count
ORDER BY
    cd.cd_gender,
    cd.cd_marital_status,
    cd.cd_education_status,
    cd.cd_purchase_estimate,
    cd.cd_credit_rating,
    cd.cd_dep_count,
    cd.cd_employed_count,
    cd.cd_dep_college_count
LIMIT 100;
```

Changes: Pre-filtered dimension tables (date_dim, customer_address) into CTEs, pre-joined each fact table with filtered dates to get distinct customer keys, then consolidated all filters in a single customer CTE before final aggregation. This reduces repeated scans and allows early filtering.

Expected speedup: 2-3x through reduced fact table scanning and early dimension filtering.
