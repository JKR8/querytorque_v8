```sql
WITH filtered_sales AS (
  SELECT 
    ss_list_price,
    ss_quantity,
    -- Pre-compute column predicates for each bucket
    ss_list_price BETWEEN 131 AND 141 AS lp_b1,
    ss_coupon_amt BETWEEN 16798 AND 17798 AS ca_b1,
    ss_wholesale_cost BETWEEN 25 AND 45 AS wc_b1,
    
    ss_list_price BETWEEN 145 AND 155 AS lp_b2,
    ss_coupon_amt BETWEEN 14792 AND 15792 AS ca_b2,
    ss_wholesale_cost BETWEEN 46 AND 66 AS wc_b2,
    
    ss_list_price BETWEEN 150 AND 160 AS lp_b3,
    ss_coupon_amt BETWEEN 6600 AND 7600 AS ca_b3,
    ss_wholesale_cost BETWEEN 9 AND 29 AS wc_b3,
    
    ss_list_price BETWEEN 91 AND 101 AS lp_b4,
    ss_coupon_amt BETWEEN 13493 AND 14493 AS ca_b4,
    ss_wholesale_cost BETWEEN 36 AND 56 AS wc_b4,
    
    ss_list_price BETWEEN 0 AND 10 AS lp_b5,
    ss_coupon_amt BETWEEN 7629 AND 8629 AS ca_b5,
    ss_wholesale_cost BETWEEN 6 AND 26 AS wc_b5,
    
    ss_list_price BETWEEN 89 AND 99 AS lp_b6,
    ss_coupon_amt BETWEEN 15257 AND 16257 AS ca_b6,
    ss_wholesale_cost BETWEEN 31 AND 51 AS wc_b6
  FROM store_sales
  WHERE ss_quantity BETWEEN 0 AND 30
),
bucket_assignments AS (
  SELECT
    ss_list_price,
    -- Assign rows to buckets using pre-computed flags
    CASE 
      WHEN ss_quantity BETWEEN 0 AND 5 AND (lp_b1 OR ca_b1 OR wc_b1) THEN 1
      WHEN ss_quantity BETWEEN 6 AND 10 AND (lp_b2 OR ca_b2 OR wc_b2) THEN 2
      WHEN ss_quantity BETWEEN 11 AND 15 AND (lp_b3 OR ca_b3 OR wc_b3) THEN 3
      WHEN ss_quantity BETWEEN 16 AND 20 AND (lp_b4 OR ca_b4 OR wc_b4) THEN 4
      WHEN ss_quantity BETWEEN 21 AND 25 AND (lp_b5 OR ca_b5 OR wc_b5) THEN 5
      WHEN ss_quantity BETWEEN 26 AND 30 AND (lp_b6 OR ca_b6 OR wc_b6) THEN 6
    END AS bucket
  FROM filtered_sales
)
SELECT
  AVG(CASE WHEN bucket = 1 THEN ss_list_price END) AS B1_LP,
  COUNT(CASE WHEN bucket = 1 THEN ss_list_price END) AS B1_CNT,
  COUNT(DISTINCT CASE WHEN bucket = 1 THEN ss_list_price END) AS B1_CNTD,
  
  AVG(CASE WHEN bucket = 2 THEN ss_list_price END) AS B2_LP,
  COUNT(CASE WHEN bucket = 2 THEN ss_list_price END) AS B2_CNT,
  COUNT(DISTINCT CASE WHEN bucket = 2 THEN ss_list_price END) AS B2_CNTD,
  
  AVG(CASE WHEN bucket = 3 THEN ss_list_price END) AS B3_LP,
  COUNT(CASE WHEN bucket = 3 THEN ss_list_price END) AS B3_CNT,
  COUNT(DISTINCT CASE WHEN bucket = 3 THEN ss_list_price END) AS B3_CNTD,
  
  AVG(CASE WHEN bucket = 4 THEN ss_list_price END) AS B4_LP,
  COUNT(CASE WHEN bucket = 4 THEN ss_list_price END) AS B4_CNT,
  COUNT(DISTINCT CASE WHEN bucket = 4 THEN ss_list_price END) AS B4_CNTD,
  
  AVG(CASE WHEN bucket = 5 THEN ss_list_price END) AS B5_LP,
  COUNT(CASE WHEN bucket = 5 THEN ss_list_price END) AS B5_CNT,
  COUNT(DISTINCT CASE WHEN bucket = 5 THEN ss_list_price END) AS B5_CNTD,
  
  AVG(CASE WHEN bucket = 6 THEN ss_list_price END) AS B6_LP,
  COUNT(CASE WHEN bucket = 6 THEN ss_list_price END) AS B6_CNT,
  COUNT(DISTINCT CASE WHEN bucket = 6 THEN ss_list_price END) AS B6_CNTD
FROM bucket_assignments
LIMIT 100;
```

**Changes:** Consolidated six separate table scans into a single pass by pre-computing column predicates for all buckets, filtering to the relevant quantity range (0-30) upfront, then using conditional aggregation with efficient bucket assignment via pre-computed boolean flags.

**Expected speedup:** 3-5x by eliminating redundant scans and predicate re-evaluation while maintaining exact semantics.
