```sql
WITH filtered_reason AS (
    SELECT r_reason_sk
    FROM reason
    WHERE r_reason_desc = 'duplicate purchase'
),
filtered_returns AS (
    SELECT 
        sr_item_sk,
        sr_ticket_number,
        sr_return_quantity
    FROM store_returns
    JOIN filtered_reason 
        ON sr_reason_sk = r_reason_sk
)
SELECT
    ss_customer_sk,
    SUM(act_sales) AS sumsales
FROM (
    SELECT
        ss_customer_sk,
        CASE
            WHEN NOT sr_return_quantity IS NULL
            THEN (ss_quantity - sr_return_quantity) * ss_sales_price
            ELSE (ss_quantity * ss_sales_price)
        END AS act_sales
    FROM store_sales
    LEFT OUTER JOIN filtered_returns
        ON sr_item_sk = ss_item_sk 
        AND sr_ticket_number = ss_ticket_number
) AS t
GROUP BY ss_customer_sk
ORDER BY sumsales, ss_customer_sk
LIMIT 100
```

Changes: Isolated the dimension filter on `reason` into a CTE, then pre-joined it with `store_returns` to create a filtered set of returns before joining to `store_sales`. This reduces the join volume early and avoids scanning the entire `reason` table during the main query.

Expected speedup: 1.5-2x by filtering small dimension tables first and reducing the join cardinality before the main fact table join.
