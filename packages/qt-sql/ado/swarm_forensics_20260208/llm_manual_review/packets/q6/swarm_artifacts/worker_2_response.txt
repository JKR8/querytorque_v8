```sql
WITH target_month AS (
  SELECT DISTINCT d_month_seq
  FROM date_dim
  WHERE d_year = 2002 AND d_moy = 3
),
filtered_dates AS (
  SELECT d_date_sk
  FROM date_dim
  JOIN target_month ON date_dim.d_month_seq = target_month.d_month_seq
),
category_avg_price AS (
  SELECT 
    i_category,
    AVG(i_current_price) * 1.2 AS price_threshold
  FROM item
  GROUP BY i_category
),
filtered_sales AS (
  SELECT 
    ss_customer_sk,
    ss_item_sk
  FROM store_sales
  JOIN filtered_dates ON store_sales.ss_sold_date_sk = filtered_dates.d_date_sk
)
SELECT
  a.ca_state AS state,
  COUNT(*) AS cnt
FROM customer_address a
JOIN customer c ON a.ca_address_sk = c.c_current_addr_sk
JOIN filtered_sales s ON c.c_customer_sk = s.ss_customer_sk
JOIN item i ON s.ss_item_sk = i.i_item_sk
JOIN category_avg_price cap ON i.i_category = cap.i_category
WHERE i.i_current_price > cap.price_threshold
GROUP BY a.ca_state
HAVING COUNT(*) >= 10
ORDER BY cnt, a.ca_state
LIMIT 100
```

Changes: Isolated date filtering into `filtered_dates` CTE and decorrelated item price subquery into `category_avg_price` CTE, converting both to efficient hash joins instead of repeated subquery executions.

Expected speedup: 2-3x through elimination of correlated subquery execution and early fact table filtering.
