```sql
WITH store_filtered AS (
  SELECT s_store_sk
  FROM store
  WHERE s_state = 'TN'
), customer_total_return_filtered AS (
  SELECT
    sr_customer_sk AS ctr_customer_sk,
    sr_store_sk AS ctr_store_sk,
    SUM(sr_return_amt) AS ctr_total_return
  FROM store_returns
  JOIN date_dim ON sr_returned_date_sk = d_date_sk
  JOIN store_filtered sf ON sr_store_sk = sf.s_store_sk
  WHERE d_year = 2000
  GROUP BY
    sr_customer_sk,
    sr_store_sk
), store_avg_return AS (
  SELECT
    ctr_store_sk,
    AVG(ctr_total_return) * 1.2 AS avg_threshold
  FROM customer_total_return_filtered
  GROUP BY ctr_store_sk
)
SELECT
  c_customer_id
FROM customer_total_return_filtered ctr1
JOIN store_avg_return sar ON ctr1.ctr_store_sk = sar.ctr_store_sk
JOIN customer ON ctr1.ctr_customer_sk = c_customer_sk
WHERE ctr1.ctr_total_return > sar.avg_threshold
ORDER BY c_customer_id
LIMIT 100
```

Changes: Decorrelated the subquery by computing store averages in a separate CTE, pushed the `s_state='TN'` filter early into the `store_filtered` CTE, and converted implicit joins to explicit JOIN syntax.

Expected speedup: 1.5x-2x due to elimination of correlated subquery execution and early filtering of store rows.
