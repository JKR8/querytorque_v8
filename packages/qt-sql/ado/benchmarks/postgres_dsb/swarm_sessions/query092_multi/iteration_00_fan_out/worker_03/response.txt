```sql
WITH filtered_dates AS (
    SELECT d_date_sk
    FROM date_dim
    WHERE d_date BETWEEN '1998-03-13' AND CAST('1998-03-13' AS DATE) + INTERVAL '90 DAY'
),
filtered_items AS (
    SELECT i_item_sk
    FROM item
    WHERE i_manufact_id BETWEEN 341 AND 540
       OR i_category IN ('Home', 'Men', 'Music')
),
filtered_web_sales AS (
    SELECT ws_item_sk, ws_sold_date_sk, ws_ext_discount_amt,
           ws_sales_price, ws_list_price
    FROM web_sales
    WHERE ws_wholesale_cost BETWEEN 26 AND 46
),
item_thresholds AS (
    SELECT fws.ws_item_sk,
           1.3 * AVG(fws.ws_ext_discount_amt) AS threshold
    FROM filtered_web_sales fws
    JOIN filtered_dates fd ON fws.ws_sold_date_sk = fd.d_date_sk
    WHERE fws.ws_sales_price / fws.ws_list_price BETWEEN 34 * 0.01 AND 49 * 0.01
    GROUP BY fws.ws_item_sk
)
SELECT SUM(fws.ws_ext_discount_amt) AS "Excess Discount Amount"
FROM filtered_web_sales fws
JOIN filtered_dates fd ON fws.ws_sold_date_sk = fd.d_date_sk
JOIN filtered_items fi ON fws.ws_item_sk = fi.i_item_sk
JOIN item_thresholds it ON fws.ws_item_sk = it.ws_item_sk
WHERE fws.ws_ext_discount_amt > it.threshold
ORDER BY SUM(fws.ws_ext_discount_amt)
LIMIT 100;
```

**Changes:** Materialized three filtered dimension/fact CTEs (`filtered_dates`, `filtered_items`, `filtered_web_sales`), then computed the subquery aggregation once per item in a separate CTE (`item_thresholds`), eliminating the correlated execution pattern.

**Expected speedup:** 2-3x due to eliminating per-row correlated subquery execution and reducing join cardinality through early materialization of filtered tables.
