```sql
WITH store_filtered AS (
    SELECT s_store_sk
    FROM store
    WHERE s_state IN ('MI', 'ND', 'TX')
),
customer_total_return AS (
    SELECT
        sr_customer_sk AS ctr_customer_sk,
        sr_store_sk AS ctr_store_sk,
        sr_reason_sk AS ctr_reason_sk,
        SUM(sr_refunded_cash) AS ctr_total_return
    FROM store_returns
    JOIN date_dim ON sr_returned_date_sk = d_date_sk
    JOIN store_filtered sf ON sr_store_sk = sf.s_store_sk
    WHERE
        d_year = 2000
        AND sr_return_amt / sr_return_quantity BETWEEN 16 AND 75
    GROUP BY
        sr_customer_sk,
        sr_store_sk,
        sr_reason_sk
),
store_thresholds AS (
    SELECT
        ctr_store_sk,
        AVG(ctr_total_return) * 1.2 AS avg_limit
    FROM customer_total_return
    GROUP BY ctr_store_sk
),
filtered_customer_demographics AS (
    SELECT cd_demo_sk
    FROM customer_demographics
    WHERE 
        cd_marital_status IN ('M', 'M')
        AND cd_education_status IN ('College', 'College')
        AND cd_gender = 'M'
)
SELECT
    c_customer_id
FROM customer_total_return ctr1
JOIN store_thresholds st ON ctr1.ctr_store_sk = st.ctr_store_sk
JOIN customer ON ctr1.ctr_customer_sk = c_customer_sk
JOIN filtered_customer_demographics fcd ON c_current_cdemo_sk = fcd.cd_demo_sk
JOIN store s ON ctr1.ctr_store_sk = s.s_store_sk
WHERE
    ctr1.ctr_total_return > st.avg_limit
    AND ctr1.ctr_reason_sk BETWEEN 25 AND 28
    AND s.s_state IN ('MI', 'ND', 'TX')
    AND c_birth_month = 9
    AND c_birth_year BETWEEN 1979 AND 1985
ORDER BY
    c_customer_id
LIMIT 100
```

Changes: Applied early filter pushdown by pre-filtering store and customer_demographics tables, decorrelated the subquery by computing store thresholds in a separate CTE, and converted all comma joins to explicit JOIN syntax. This reduces data volume early in the query execution.
