```sql
WITH filtered_date AS (
  SELECT d_date_sk
  FROM date_dim
  WHERE d_month_seq IN (
    SELECT d_month_seq
    FROM date_dim
    WHERE d_date IN ('2002-02-01', '2002-04-11', '2002-07-17', '2002-10-09')
  )
),
filtered_item AS (
  SELECT i_item_sk, i_item_id
  FROM item
  WHERE i_category IN ('Jewelry', 'Music')
    AND i_manager_id BETWEEN 16 AND 25
),
unified_returns AS (
  SELECT
    i.i_item_id AS item_id,
    COALESCE(SUM(CASE WHEN sr.sr_item_sk IS NOT NULL THEN sr.sr_return_quantity END), 0) AS sr_item_qty,
    COALESCE(SUM(CASE WHEN cr.cr_item_sk IS NOT NULL THEN cr.cr_return_quantity END), 0) AS cr_item_qty,
    COALESCE(SUM(CASE WHEN wr.wr_item_sk IS NOT NULL THEN wr.wr_return_quantity END), 0) AS wr_item_qty
  FROM filtered_item i
  LEFT JOIN store_returns sr ON i.i_item_sk = sr.sr_item_sk
    AND sr.sr_returned_date_sk IN (SELECT d_date_sk FROM filtered_date)
    AND sr.sr_return_amt / sr.sr_return_quantity BETWEEN 184 AND 213
    AND sr.sr_reason_sk IN (26, 32, 40, 66, 73)
  LEFT JOIN catalog_returns cr ON i.i_item_sk = cr.cr_item_sk
    AND cr.cr_returned_date_sk IN (SELECT d_date_sk FROM filtered_date)
    AND cr.cr_return_amount / cr.cr_return_quantity BETWEEN 184 AND 213
    AND cr.cr_reason_sk IN (26, 32, 40, 66, 73)
  LEFT JOIN web_returns wr ON i.i_item_sk = wr.wr_item_sk
    AND wr.wr_returned_date_sk IN (SELECT d_date_sk FROM filtered_date)
    AND wr.wr_return_amt / wr.wr_return_quantity BETWEEN 184 AND 213
    AND wr.wr_reason_sk IN (26, 32, 40, 66, 73)
  GROUP BY i.i_item_id
  HAVING COALESCE(SUM(CASE WHEN sr.sr_item_sk IS NOT NULL THEN sr.sr_return_quantity END), 0) > 0
     AND COALESCE(SUM(CASE WHEN cr.cr_item_sk IS NOT NULL THEN cr.cr_return_quantity END), 0) > 0
     AND COALESCE(SUM(CASE WHEN wr.wr_item_sk IS NOT NULL THEN wr.wr_return_quantity END), 0) > 0
)
SELECT
  item_id,
  sr_item_qty,
  sr_item_qty / (sr_item_qty + cr_item_qty + wr_item_qty) / 3.0 * 100 AS sr_dev,
  cr_item_qty,
  cr_item_qty / (sr_item_qty + cr_item_qty + wr_item_qty) / 3.0 * 100 AS cr_dev,
  wr_item_qty,
  wr_item_qty / (sr_item_qty + cr_item_qty + wr_item_qty) / 3.0 * 100 AS wr_dev,
  (sr_item_qty + cr_item_qty + wr_item_qty) / 3.0 AS average
FROM unified_returns
ORDER BY item_id, sr_item_qty
LIMIT 100
```

**Changes**: Applied aggressive_multi_cte_fusion by creating filtered dimension CTEs and a single unified returns CTE that computes all three return quantities in one scan using conditional aggregation with LEFT JOINs and HAVING to ensure items exist in all three channels.

**Expected speedup**: 2.5-3.5x due to reduced table scans, early dimension filtering, and single aggregation pass instead of three separate CTEs.
