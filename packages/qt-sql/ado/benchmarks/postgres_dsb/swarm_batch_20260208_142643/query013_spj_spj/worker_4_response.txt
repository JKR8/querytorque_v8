```sql
WITH date_filter AS (
    SELECT d_date_sk
    FROM date_dim
    WHERE d_year = 2001
),
-- First OR cluster: demographic + price conditions
demo_branch1 AS (
    SELECT ss_quantity, ss_ext_sales_price, ss_ext_wholesale_cost
    FROM store_sales
    JOIN date_filter ON ss_sold_date_sk = d_date_sk
    JOIN store ON s_store_sk = ss_store_sk
    JOIN customer_demographics ON cd_demo_sk = ss_cdemo_sk
    JOIN household_demographics ON ss_hdemo_sk = hd_demo_sk
    WHERE cd_marital_status = 'U'
      AND cd_education_status = 'College'
      AND ss_sales_price BETWEEN 100.00 AND 150.00
      AND hd_dep_count = 3
),
demo_branch2 AS (
    SELECT ss_quantity, ss_ext_sales_price, ss_ext_wholesale_cost
    FROM store_sales
    JOIN date_filter ON ss_sold_date_sk = d_date_sk
    JOIN store ON s_store_sk = ss_store_sk
    JOIN customer_demographics ON cd_demo_sk = ss_cdemo_sk
    JOIN household_demographics ON ss_hdemo_sk = hd_demo_sk
    WHERE cd_marital_status = 'W'
      AND cd_education_status = 'Secondary'
      AND ss_sales_price BETWEEN 50.00 AND 100.00
      AND hd_dep_count = 1
),
demo_branch3 AS (
    SELECT ss_quantity, ss_ext_sales_price, ss_ext_wholesale_cost
    FROM store_sales
    JOIN date_filter ON ss_sold_date_sk = d_date_sk
    JOIN store ON s_store_sk = ss_store_sk
    JOIN customer_demographics ON cd_demo_sk = ss_cdemo_sk
    JOIN household_demographics ON ss_hdemo_sk = hd_demo_sk
    WHERE cd_marital_status = 'D'
      AND cd_education_status = 'Secondary'
      AND ss_sales_price BETWEEN 150.00 AND 200.00
      AND hd_dep_count = 1
),
-- Second OR cluster: address + profit conditions
addr_branch1 AS (
    SELECT ss_quantity, ss_ext_sales_price, ss_ext_wholesale_cost
    FROM store_sales
    JOIN date_filter ON ss_sold_date_sk = d_date_sk
    JOIN store ON s_store_sk = ss_store_sk
    JOIN customer_address ON ss_addr_sk = ca_address_sk
    WHERE ca_country = 'United States'
      AND ca_state IN ('IA', 'MO', 'TX')
      AND ss_net_profit BETWEEN 100 AND 200
),
addr_branch2 AS (
    SELECT ss_quantity, ss_ext_sales_price, ss_ext_wholesale_cost
    FROM store_sales
    JOIN date_filter ON ss_sold_date_sk = d_date_sk
    JOIN store ON s_store_sk = ss_store_sk
    JOIN customer_address ON ss_addr_sk = ca_address_sk
    WHERE ca_country = 'United States'
      AND ca_state IN ('GA', 'LA', 'SD')
      AND ss_net_profit BETWEEN 150 AND 300
),
addr_branch3 AS (
    SELECT ss_quantity, ss_ext_sales_price, ss_ext_wholesale_cost
    FROM store_sales
    JOIN date_filter ON ss_sold_date_sk = d_date_sk
    JOIN store ON s_store_sk = ss_store_sk
    JOIN customer_address ON ss_addr_sk = ca_address_sk
    WHERE ca_country = 'United States'
      AND ca_state IN ('TN', 'TX', 'VA')
      AND ss_net_profit BETWEEN 50 AND 250
),
-- Combine all branches: logical AND between the two OR clusters
all_qualified_rows AS (
    SELECT ss_quantity, ss_ext_sales_price, ss_ext_wholesale_cost
    FROM demo_branch1
    INTERSECT
    SELECT ss_quantity, ss_ext_sales_price, ss_ext_wholesale_cost
    FROM addr_branch1
    
    UNION
    
    SELECT ss_quantity, ss_ext_sales_price, ss_ext_wholesale_cost
    FROM demo_branch1
    INTERSECT
    SELECT ss_quantity, ss_ext_sales_price, ss_ext_wholesale_cost
    FROM addr_branch2
    
    UNION
    
    SELECT ss_quantity, ss_ext_sales_price, ss_ext_wholesale_cost
    FROM demo_branch1
    INTERSECT
    SELECT ss_quantity, ss_ext_sales_price, ss_ext_wholesale_cost
    FROM addr_branch3
    
    UNION
    
    SELECT ss_quantity, ss_ext_sales_price, ss_ext_wholesale_cost
    FROM demo_branch2
    INTERSECT
    SELECT ss_quantity, ss_ext_sales_price, ss_ext_wholesale_cost
    FROM addr_branch1
    
    UNION
    
    SELECT ss_quantity, ss_ext_sales_price, ss_ext_wholesale_cost
    FROM demo_branch2
    INTERSECT
    SELECT ss_quantity, ss_ext_sales_price, ss_ext_wholesale_cost
    FROM addr_branch2
    
    UNION
    
    SELECT ss_quantity, ss_ext_sales_price, ss_ext_wholesale_cost
    FROM demo_branch2
    INTERSECT
    SELECT ss_quantity, ss_ext_sales_price, ss_ext_wholesale_cost
    FROM addr_branch3
    
    UNION
    
    SELECT ss_quantity, ss_ext_sales_price, ss_ext_wholesale_cost
    FROM demo_branch3
    INTERSECT
    SELECT ss_quantity, ss_ext_sales_price, ss_ext_wholesale_cost
    FROM addr_branch1
    
    UNION
    
    SELECT ss_quantity, ss_ext_sales_price, ss_ext_wholesale_cost
    FROM demo_branch3
    INTERSECT
    SELECT ss_quantity, ss_ext_sales_price, ss_ext_wholesale_cost
    FROM addr_branch2
    
    UNION
    
    SELECT ss_quantity, ss_ext_sales_price, ss_ext_wholesale_cost
    FROM demo_branch3
    INTERSECT
    SELECT ss_quantity, ss_ext_sales_price, ss_ext_wholesale_cost
    FROM addr_branch3
)
SELECT
    MIN(ss_quantity),
    MIN(ss_ext_sales_price),
    MIN(ss_ext_wholesale_cost),
    MIN(ss_ext_wholesale_cost)
FROM all_qualified_rows;
```

**Changes**: Transformed the two OR clusters (demographic+price and address+profit) into separate sets of UNION branches, then used INTERSECT to get rows satisfying both conditions, implementing the logical AND between the OR clusters. This allows PostgreSQL to optimize each branch independently with appropriate index usage.

**Expected speedup**: 2-3x by eliminating the complex OR condition evaluation and allowing per-branch optimization paths. Each branch can use indexes on the specific filtered columns, and the date filter is pushed down early via the CTE.
