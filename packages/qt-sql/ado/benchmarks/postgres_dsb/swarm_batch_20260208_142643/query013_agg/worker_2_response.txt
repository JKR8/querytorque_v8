```sql
WITH filtered_date AS (
    SELECT d_date_sk
    FROM date_dim
    WHERE d_year = 2001
),
filtered_cd_hd AS (
    SELECT 
        cd_demo_sk,
        hd_demo_sk,
        cd_marital_status,
        cd_education_status,
        hd_dep_count
    FROM customer_demographics
    JOIN household_demographics ON true
    WHERE 
        (cd_marital_status = 'U' AND cd_education_status = 'College' AND hd_dep_count = 3)
        OR (cd_marital_status = 'W' AND cd_education_status = 'Secondary' AND hd_dep_count = 1)
        OR (cd_marital_status = 'D' AND cd_education_status = 'Secondary' AND hd_dep_count = 1)
),
filtered_ca AS (
    SELECT 
        ca_address_sk,
        ca_state
    FROM customer_address
    WHERE 
        ca_country = 'United States'
        AND (
            ca_state IN ('IA', 'MO', 'TX')
            OR ca_state IN ('GA', 'LA', 'SD')
            OR ca_state IN ('TN', 'TX', 'VA')
        )
)
SELECT
    AVG(ss_quantity),
    AVG(ss_ext_sales_price),
    AVG(ss_ext_wholesale_cost),
    SUM(ss_ext_wholesale_cost)
FROM store_sales
JOIN store ON s_store_sk = ss_store_sk
JOIN filtered_date ON ss_sold_date_sk = filtered_date.d_date_sk
JOIN filtered_cd_hd ON 
    ss_cdemo_sk = filtered_cd_hd.cd_demo_sk 
    AND ss_hdemo_sk = filtered_cd_hd.hd_demo_sk
JOIN filtered_ca ON ss_addr_sk = filtered_ca.ca_address_sk
WHERE
    (
        (
            filtered_cd_hd.cd_marital_status = 'U'
            AND filtered_cd_hd.cd_education_status = 'College'
            AND ss_sales_price BETWEEN 100.00 AND 150.00
            AND filtered_cd_hd.hd_dep_count = 3
        )
        OR (
            filtered_cd_hd.cd_marital_status = 'W'
            AND filtered_cd_hd.cd_education_status = 'Secondary'
            AND ss_sales_price BETWEEN 50.00 AND 100.00
            AND filtered_cd_hd.hd_dep_count = 1
        )
        OR (
            filtered_cd_hd.cd_marital_status = 'D'
            AND filtered_cd_hd.cd_education_status = 'Secondary'
            AND ss_sales_price BETWEEN 150.00 AND 200.00
            AND filtered_cd_hd.hd_dep_count = 1
        )
    )
    AND (
        (
            filtered_ca.ca_state IN ('IA', 'MO', 'TX')
            AND ss_net_profit BETWEEN 100 AND 200
        )
        OR (
            filtered_ca.ca_state IN ('GA', 'LA', 'SD')
            AND ss_net_profit BETWEEN 150 AND 300
        )
        OR (
            filtered_ca.ca_state IN ('TN', 'TX', 'VA')
            AND ss_net_profit BETWEEN 50 AND 250
        )
    );
```

Changes: Created three CTEs to pre-filter dimension tables: `filtered_date` for year 2001, `filtered_cd_hd` for the three demographic combinations, and `filtered_ca` for US states. Converted implicit joins to explicit JOIN syntax and moved dimension filters into CTE WHERE clauses.

Expected speedup: 1.8-2.5x due to reduced dimension table sizes before joining with the large fact table, enabling better hash join planning and reduced intermediate result sizes.
