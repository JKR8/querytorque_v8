```sql
WITH filtered_ws AS (
    SELECT 
        ws1.ws_order_number,
        ws1.ws_ext_ship_cost,
        ws1.ws_net_profit,
        ws1.ws_warehouse_sk,
        COUNT(DISTINCT ws2.ws_warehouse_sk) 
            OVER (PARTITION BY ws1.ws_order_number) AS warehouse_count
    FROM web_sales ws1
    JOIN date_dim ON ws1.ws_ship_date_sk = d_date_sk
    JOIN customer_address ON ws1.ws_ship_addr_sk = ca_address_sk  
    JOIN web_site ON ws1.ws_web_site_sk = web_site_sk
    LEFT JOIN web_sales ws2 ON ws1.ws_order_number = ws2.ws_order_number 
        AND ws1.ws_warehouse_sk <> ws2.ws_warehouse_sk
    WHERE d_date BETWEEN '2002-10-01' AND CAST('2002-10-01' AS DATE) + INTERVAL '60 DAY'
      AND ca_state IN ('IL', 'KS', 'LA', 'MN', 'MT', 'SC')
      AND web_gmt_offset >= -5
      AND ws1.ws_list_price BETWEEN 184 AND 213
),
returned_orders AS (
    SELECT DISTINCT wr_order_number
    FROM web_returns
    WHERE wr_reason_sk IN (17, 48, 50, 56, 68)
)
SELECT 
    COUNT(DISTINCT ws_order_number) AS "order count",
    SUM(ws_ext_ship_cost) AS "total shipping cost", 
    SUM(ws_net_profit) AS "total net profit"
FROM filtered_ws
WHERE warehouse_count >= 2
  AND ws_order_number NOT IN (SELECT wr_order_number FROM returned_orders)
ORDER BY COUNT(DISTINCT ws_order_number)
LIMIT 100
```

**Changes**: 
1. Created a single filtered base CTE with all dimension joins and filters applied once, using explicit JOIN syntax.
2. Replaced correlated EXISTS with window function COUNT(DISTINCT) OVER() by LEFT JOINing web_sales on itself, computing warehouse_count per order.
3. Materialized returned orders as a separate CTE and converted NOT EXISTS to NOT IN.
4. Combined all logic in a single pass aggregation with WHERE clauses based on pre-computed counts.

**Expected speedup**: ~2-3x by eliminating correlated subqueries, reducing dimension table scans, and enabling better join planning with explicit syntax.
