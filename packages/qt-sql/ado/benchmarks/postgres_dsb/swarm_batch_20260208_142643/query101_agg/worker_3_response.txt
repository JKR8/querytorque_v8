```sql
WITH filtered_item AS (
    SELECT i_item_sk
    FROM item
    WHERE i_category IN ('Children', 'Electronics', 'Women')
),
filtered_d1 AS (
    SELECT d_date_sk, d_date
    FROM date_dim
    WHERE d_year = 2000
),
filtered_ca AS (
    SELECT ca_address_sk
    FROM customer_address
    WHERE ca_state IN ('GA', 'IA', 'LA', 'MO', 'SD')
),
filtered_hd AS (
    SELECT hd_demo_sk
    FROM household_demographics
    WHERE hd_income_band_sk BETWEEN 12 AND 18
      AND hd_buy_potential = '0-500'
),
filtered_ss AS (
    SELECT ss_ticket_number, ss_customer_sk, ss_item_sk
    FROM store_sales
    INNER JOIN filtered_item ON ss_item_sk = i_item_sk
    WHERE ss_sales_price / ss_list_price BETWEEN 0.34 AND 0.54
),
joined_facts AS (
    SELECT 
        ss.ss_customer_sk,
        ss.ss_item_sk,
        sr.sr_returned_date_sk,
        ws.ws_sold_date_sk
    FROM filtered_ss ss
    INNER JOIN store_returns sr ON ss.ss_ticket_number = sr.sr_ticket_number
        AND ss.ss_item_sk = sr.sr_item_sk
    INNER JOIN web_sales ws ON ss.ss_customer_sk = ws.ws_bill_customer_sk
        AND ss.ss_item_sk = ws.ws_item_sk
)
SELECT
    c.c_customer_sk,
    c.c_first_name,
    c.c_last_name,
    COUNT(*) AS cnt
FROM joined_facts jf
INNER JOIN filtered_d1 d1 ON jf.sr_returned_date_sk = d1.d_date_sk
INNER JOIN date_dim d2 ON jf.ws_sold_date_sk = d2.d_date_sk
INNER JOIN customer c ON jf.ss_customer_sk = c.c_customer_sk
INNER JOIN filtered_ca ca ON c.c_current_addr_sk = ca.ca_address_sk
INNER JOIN filtered_hd hd ON c.c_current_hdemo_sk = hd.hd_demo_sk
WHERE d2.d_date BETWEEN d1.d_date AND (d1.d_date + INTERVAL '90 DAY')
GROUP BY c.c_customer_sk, c.c_first_name, c.c_last_name
ORDER BY cnt
```

Changes: Materialized all selective dimension filters into separate CTEs (item, date_dim d1, customer_address, household_demographics) and pre-filtered store_sales by item category and price ratio. Created a joined_facts CTE to link the three fact tables once with precise join conditions, eliminating redundant scans. This creates small hash tables for dimensions and reduces the fact join to a single pass.

Expected speedup: 2.5-3.5x from reduced table scans, better cardinality estimates, and elimination of Cartesian products in the original implicit join.
