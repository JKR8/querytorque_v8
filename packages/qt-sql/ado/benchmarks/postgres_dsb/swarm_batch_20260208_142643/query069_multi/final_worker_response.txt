```sql
WITH date_filter AS (
    SELECT d_date_sk
    FROM date_dim
    WHERE d_year = 2002
      AND d_moy BETWEEN 3 AND 3 + 2
),
store_sales_customers AS (
    SELECT DISTINCT ss_customer_sk
    FROM store_sales
    JOIN date_filter ON ss_sold_date_sk = d_date_sk
    WHERE ss_list_price BETWEEN 100 AND 189
),
web_sales_customers AS (
    SELECT DISTINCT ws_bill_customer_sk
    FROM web_sales
    JOIN date_filter ON ws_sold_date_sk = d_date_sk
    WHERE ws_list_price BETWEEN 100 AND 189
),
catalog_sales_customers AS (
    SELECT DISTINCT cs_ship_customer_sk
    FROM catalog_sales
    JOIN date_filter ON cs_sold_date_sk = d_date_sk
    WHERE cs_list_price BETWEEN 100 AND 189
),
filtered_customers AS (
    SELECT 
        c.c_customer_sk,
        cd.cd_gender,
        cd.cd_marital_status,
        cd.cd_education_status,
        cd.cd_purchase_estimate,
        cd.cd_credit_rating
    FROM customer c
    JOIN customer_address ca ON c.c_current_addr_sk = ca.ca_address_sk
    JOIN customer_demographics cd ON cd.cd_demo_sk = c.c_current_cdemo_sk
    WHERE ca.ca_state IN ('IA', 'MO', 'TX')
      AND cd.cd_marital_status = 'S'
      AND cd.cd_education_status IN ('Primary', 'Secondary')
      AND EXISTS (SELECT 1 FROM store_sales_customers ssc WHERE ssc.ss_customer_sk = c.c_customer_sk)
      AND NOT EXISTS (SELECT 1 FROM web_sales_customers wsc WHERE wsc.ws_bill_customer_sk = c.c_customer_sk)
      AND NOT EXISTS (SELECT 1 FROM catalog_sales_customers csc WHERE csc.cs_ship_customer_sk = c.c_customer_sk)
)
SELECT
    cd_gender,
    cd_marital_status,
    cd_education_status,
    COUNT(*) AS cnt1,
    cd_purchase_estimate,
    COUNT(*) AS cnt2,
    cd_credit_rating,
    COUNT(*) AS cnt3
FROM filtered_customers
GROUP BY
    cd_gender,
    cd_marital_status,
    cd_education_status,
    cd_purchase_estimate,
    cd_credit_rating
ORDER BY
    cd_gender,
    cd_marital_status,
    cd_education_status,
    cd_purchase_estimate,
    cd_credit_rating
LIMIT 100
```

Changes: Created lightweight CTEs to pre-aggregate each sales channel's qualifying customers, then performed a single pass join with customer/demographics/address using EXISTS/NOT EXISTS logic with the precomputed CTEs. This reduces multiple correlated subqueries to single scans per fact table.

Expected speedup: 1.5-2x due to eliminating repeated date_dim joins and enabling better join ordering.
