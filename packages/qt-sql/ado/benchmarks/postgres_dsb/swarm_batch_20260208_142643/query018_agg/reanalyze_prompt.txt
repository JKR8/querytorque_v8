        ca_country,

        avg( cast(cs_coupon_amt as decimal(12,2))) agg3,

      cs_item_sk = i_item_sk and

      c_birth_month = 1 and

      and cs_wholesale_cost BETWEEN 52 AND 57
      AND i_category = 'Jewelry'
 group by rollup (i_item_id, ca_country, ca_state, ca_county)

        ca_county,
