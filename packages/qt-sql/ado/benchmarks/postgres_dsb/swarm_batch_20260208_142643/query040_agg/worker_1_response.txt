```sql
WITH filtered_date AS (
    SELECT d_date_sk, d_date
    FROM date_dim
    WHERE d_date BETWEEN (CAST('1999-04-19' AS DATE) - INTERVAL '30 DAY')
                     AND (CAST('1999-04-19' AS DATE) + INTERVAL '30 DAY')
),
filtered_item AS (
    SELECT i_item_sk, i_item_id
    FROM item
    WHERE i_category = 'Home'
      AND i_manager_id BETWEEN 25 AND 64
),
filtered_catalog_returns AS (
    SELECT cr_order_number, cr_item_sk, cr_refunded_cash
    FROM catalog_returns
    WHERE cr_reason_sk = 16
)
SELECT
    w.w_state,
    i.i_item_id,
    SUM(
        CASE
            WHEN (CAST(d.d_date AS DATE) < CAST('1999-04-19' AS DATE))
            THEN cs.cs_sales_price - COALESCE(cr.cr_refunded_cash, 0)
            ELSE 0
        END
    ) AS sales_before,
    SUM(
        CASE
            WHEN (CAST(d.d_date AS DATE) >= CAST('1999-04-19' AS DATE))
            THEN cs.cs_sales_price - COALESCE(cr.cr_refunded_cash, 0)
            ELSE 0
        END
    ) AS sales_after
FROM catalog_sales cs
INNER JOIN filtered_date d ON cs.cs_sold_date_sk = d.d_date_sk
INNER JOIN filtered_item i ON cs.cs_item_sk = i.i_item_sk
INNER JOIN warehouse w ON cs.cs_warehouse_sk = w.w_warehouse_sk
INNER JOIN filtered_catalog_returns cr ON (
    cs.cs_order_number = cr.cr_order_number
    AND cs.cs_item_sk = cr.cr_item_sk
)
WHERE cs.cs_wholesale_cost BETWEEN 17 AND 36
GROUP BY
    w.w_state,
    i.i_item_id
ORDER BY
    w.w_state,
    i.i_item_id
LIMIT 100
```

**Changes**: Created separate CTEs for filtered dimension tables (date_dim, item, catalog_returns) with their respective filters pushed down early, and converted implicit joins to explicit INNER JOINs (since the cr_reason_sk=16 condition makes the LEFT JOIN behave like an inner join). This allows PostgreSQL to process smaller hash tables and choose optimal join order.

**Expected speedup**: ~2-3x due to early reduction of large fact tables via selective dimension filters and explicit join control.
