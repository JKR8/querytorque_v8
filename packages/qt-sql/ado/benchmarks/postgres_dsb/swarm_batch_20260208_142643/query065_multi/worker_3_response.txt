```sql
WITH filtered_date AS (
  SELECT d_date_sk
  FROM date_dim
  WHERE d_month_seq BETWEEN 1213 AND 1213 + 11
),
store_sales_revenue AS (
  SELECT
    ss_store_sk,
    ss_item_sk,
    SUM(ss_sales_price) AS revenue
  FROM store_sales
  JOIN filtered_date ON store_sales.ss_sold_date_sk = filtered_date.d_date_sk
  WHERE ss_sales_price / ss_list_price BETWEEN 38 * 0.01 AND 48 * 0.01
  GROUP BY ss_store_sk, ss_item_sk
),
store_avg_revenue AS (
  SELECT
    ss_store_sk,
    AVG(revenue) AS ave
  FROM store_sales_revenue
  GROUP BY ss_store_sk
),
filtered_store AS (
  SELECT
    s_store_sk,
    s_store_name
  FROM store
  WHERE s_state IN ('TN', 'TX', 'VA')
),
filtered_item AS (
  SELECT
    i_item_sk,
    i_item_desc,
    i_current_price,
    i_wholesale_cost,
    i_brand
  FROM item
  WHERE i_manager_id BETWEEN 32 AND 36
)
SELECT
  filtered_store.s_store_name,
  filtered_item.i_item_desc,
  store_sales_revenue.revenue,
  filtered_item.i_current_price,
  filtered_item.i_wholesale_cost,
  filtered_item.i_brand
FROM store_avg_revenue
JOIN store_sales_revenue ON store_avg_revenue.ss_store_sk = store_sales_revenue.ss_store_sk
JOIN filtered_store ON store_sales_revenue.ss_store_sk = filtered_store.s_store_sk
JOIN filtered_item ON store_sales_revenue.ss_item_sk = filtered_item.i_item_sk
WHERE store_sales_revenue.revenue <= 0.1 * store_avg_revenue.ave
ORDER BY
  filtered_store.s_store_name,
  filtered_item.i_item_desc
LIMIT 100
```

**Changes**: Materialized filtered date dimension, store_sales revenue at store-item level, store averages, and filtered store/item dimensions into separate CTEs. Eliminated redundant store_sales+date_dim scans by deriving both aggregates from a single materialized result.

**Expected speedup**: 3-4x from eliminating duplicate fact table scans and creating small dimension hash tables.
