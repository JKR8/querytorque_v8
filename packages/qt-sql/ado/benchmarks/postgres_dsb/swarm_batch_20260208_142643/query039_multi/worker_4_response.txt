```sql
WITH filtered_dates AS (
    SELECT d_date_sk, d_moy
    FROM date_dim
    WHERE d_year = 2002
      AND d_moy IN (10, 11)
),
inventory_agg AS (
    SELECT
        w_warehouse_sk,
        i_item_sk,
        d_moy,
        AVG(inv_quantity_on_hand) AS mean,
        STDDEV_SAMP(inv_quantity_on_hand) AS stdev
    FROM inventory
    JOIN item ON inv_item_sk = i_item_sk
    JOIN warehouse ON inv_warehouse_sk = w_warehouse_sk
    JOIN filtered_dates ON inv_date_sk = d_date_sk
    WHERE i_category IN ('Home', 'Men')
      AND i_manager_id BETWEEN 25 AND 44
      AND inv_quantity_on_hand BETWEEN 140 AND 340
    GROUP BY w_warehouse_sk, i_item_sk, d_moy
),
filtered_stats AS (
    SELECT
        w_warehouse_sk,
        i_item_sk,
        d_moy,
        mean,
        CASE WHEN mean = 0 THEN NULL ELSE stdev / mean END AS cov
    FROM inventory_agg
    WHERE CASE WHEN mean = 0 THEN 0 ELSE stdev / mean END > 1
),
month_10 AS (
    SELECT *
    FROM filtered_stats
    WHERE d_moy = 10
),
month_11 AS (
    SELECT *
    FROM filtered_stats
    WHERE d_moy = 11
)
SELECT
    m10.w_warehouse_sk,
    m10.i_item_sk,
    m10.d_moy,
    m10.mean,
    m10.cov,
    m11.w_warehouse_sk,
    m11.i_item_sk,
    m11.d_moy,
    m11.mean,
    m11.cov
FROM month_10 m10
JOIN month_11 m11
    ON m10.w_warehouse_sk = m11.w_warehouse_sk
    AND m10.i_item_sk = m11.i_item_sk
ORDER BY
    m10.w_warehouse_sk,
    m10.i_item_sk,
    m10.d_moy,
    m10.mean,
    m10.cov,
    m11.d_moy,
    m11.mean,
    m11.cov
```

**Changes**: Created separate CTEs for date filtering, aggregation, and monthly splits. The date filter is pushed early to reduce join volume, and the self-join is replaced with explicit monthly CTEs joined on warehouse/item keys.

**Expected speedup**: 2-3x. Early date filtering reduces the fact table join size, and separate monthly CTEs avoid scanning the aggregated data twice.
