```sql
WITH filtered_d2 AS (
    SELECT d_date_sk, d_date
    FROM date_dim
    WHERE d_year = 2002
      AND d_moy = 8
),
filtered_d1 AS (
    SELECT d_date_sk, d_date
    FROM date_dim
    WHERE EXISTS (
        SELECT 1
        FROM filtered_d2
        WHERE date_dim.d_date BETWEEN (filtered_d2.d_date - INTERVAL '120 DAY') AND filtered_d2.d_date
    )
)
SELECT
    s_store_name,
    s_company_id,
    s_street_number,
    s_street_name,
    s_street_type,
    s_suite_number,
    s_city,
    s_county,
    s_state,
    s_zip,
    SUM(CASE WHEN (sr_returned_date_sk - ss_sold_date_sk <= 30) THEN 1 ELSE 0 END) AS "30 days",
    SUM(
        CASE
            WHEN (sr_returned_date_sk - ss_sold_date_sk > 30)
                 AND (sr_returned_date_sk - ss_sold_date_sk <= 60)
            THEN 1
            ELSE 0
        END
    ) AS "31-60 days",
    SUM(
        CASE
            WHEN (sr_returned_date_sk - ss_sold_date_sk > 60)
                 AND (sr_returned_date_sk - ss_sold_date_sk <= 90)
            THEN 1
            ELSE 0
        END
    ) AS "61-90 days",
    SUM(
        CASE
            WHEN (sr_returned_date_sk - ss_sold_date_sk > 90)
                 AND (sr_returned_date_sk - ss_sold_date_sk <= 120)
            THEN 1
            ELSE 0
        END
    ) AS "91-120 days",
    SUM(CASE WHEN (sr_returned_date_sk - ss_sold_date_sk > 120) THEN 1 ELSE 0 END) AS ">120 days"
FROM store_sales
JOIN store_returns
    ON ss_ticket_number = sr_ticket_number
   AND ss_item_sk = sr_item_sk
   AND ss_customer_sk = sr_customer_sk
JOIN filtered_d1 AS d1
    ON ss_sold_date_sk = d1.d_date_sk
JOIN filtered_d2 AS d2
    ON sr_returned_date_sk = d2.d_date_sk
   AND d1.d_date BETWEEN (d2.d_date - INTERVAL '120 DAY') AND d2.d_date
JOIN store
    ON ss_store_sk = s_store_sk
GROUP BY
    s_store_name,
    s_company_id,
    s_street_number,
    s_street_name,
    s_street_type,
    s_suite_number,
    s_city,
    s_county,
    s_state,
    s_zip
ORDER BY
    s_store_name,
    s_company_id,
    s_street_number,
    s_street_name,
    s_street_type,
    s_suite_number,
    s_city,
    s_county,
    s_state,
    s_zip
LIMIT 100;
```

**Changes**: Isolated selective date_dim filters into CTEs (`filtered_d2` for returned dates in August 2002, `filtered_d1` for sold dates within 120 days of those returns), converted implicit joins to explicit JOIN syntax, and preserved the original range condition between d1 and d2 in the main join.

**Expected speedup**: ~2-3x by creating small hash tables from selective date filters and enabling better join optimization with explicit syntax.
