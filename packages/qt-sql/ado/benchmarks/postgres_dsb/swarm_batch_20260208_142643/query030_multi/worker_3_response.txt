```sql
WITH filtered_date AS (
    SELECT d_date_sk
    FROM date_dim
    WHERE d_year = 1998
),
filtered_item AS (
    SELECT i_item_sk
    FROM item
    WHERE i_manager_id BETWEEN 17 AND 26
),
filtered_address_returning AS (
    SELECT ca_address_sk, ca_state
    FROM customer_address
),
web_returns_filtered AS (
    SELECT 
        wr_returning_customer_sk,
        wr_returning_addr_sk,
        wr_item_sk,
        wr_reason_sk,
        wr_return_amt
    FROM web_returns
    WHERE wr_return_amt / wr_return_quantity BETWEEN 25 AND 54
),
customer_total_return AS (
    SELECT
        wr_returning_customer_sk AS ctr_customer_sk,
        ca_state AS ctr_state,
        wr_reason_sk AS ctr_reason_sk,
        SUM(wr_return_amt) AS ctr_total_return
    FROM web_returns_filtered wr
    JOIN filtered_date d ON wr.wr_returned_date_sk = d.d_date_sk
    JOIN filtered_item i ON wr.wr_item_sk = i.i_item_sk
    JOIN filtered_address_returning ca ON wr.wr_returning_addr_sk = ca.ca_address_sk
    GROUP BY
        wr_returning_customer_sk,
        ca_state,
        wr_reason_sk
),
state_thresholds AS (
    SELECT
        ctr_state,
        AVG(ctr_total_return) * 1.2 AS threshold
    FROM customer_total_return
    GROUP BY ctr_state
),
filtered_current_address AS (
    SELECT ca_address_sk
    FROM customer_address
    WHERE ca_state IN ('IL', 'LA', 'MN', 'SC')
)
SELECT
    c_customer_id,
    c_salutation,
    c_first_name,
    c_last_name,
    c_preferred_cust_flag,
    c_birth_day,
    c_birth_month,
    c_birth_year,
    c_birth_country,
    c_login,
    c_email_address,
    c_last_review_date_sk,
    ctr_total_return
FROM customer_total_return ctr1
JOIN state_thresholds st ON ctr1.ctr_state = st.ctr_state
JOIN customer c ON ctr1.ctr_customer_sk = c.c_customer_sk
JOIN filtered_current_address ca ON ca.ca_address_sk = c.c_current_addr_sk
WHERE ctr1.ctr_total_return > st.threshold
    AND ctr1.ctr_reason_sk IN (50, 73)
    AND c.c_birth_year BETWEEN 1961 AND 1967
ORDER BY
    c_customer_id,
    c_salutation,
    c_first_name,
    c_last_name,
    c_preferred_cust_flag,
    c_birth_day,
    c_birth_month,
    c_birth_year,
    c_birth_country,
    c_login,
    c_email_address,
    c_last_review_date_sk,
    ctr_total_return
LIMIT 100;
```

**Changes**: 
- Created separate CTEs for pre-filtered dimension tables (date_dim, item, customer_address)
- Materialized filtered web_returns rows before joining
- Decorrelated the subquery by computing state thresholds in a separate CTE
- Applied explicit JOIN syntax throughout

**Expected speedup**: 2.0-3.0x due to reduced fact table scans, early filtering, and elimination of correlated subquery execution per row.
