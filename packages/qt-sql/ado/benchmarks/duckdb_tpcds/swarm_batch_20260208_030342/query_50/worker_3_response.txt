```sql
WITH filtered_return_dates AS (
    SELECT d_date_sk
    FROM date_dim
    WHERE d_year = 2001
      AND d_moy = 8
),
prefiltered_store_returns AS (
    SELECT sr_returned_date_sk, sr_ticket_number, sr_item_sk, sr_customer_sk
    FROM store_returns
    JOIN filtered_return_dates ON sr_returned_date_sk = d_date_sk
),
joined_facts AS (
    SELECT 
        ss_sold_date_sk,
        sr_returned_date_sk,
        ss_store_sk
    FROM store_sales
    JOIN prefiltered_store_returns 
      ON ss_ticket_number = sr_ticket_number
     AND ss_item_sk = sr_item_sk
     AND ss_customer_sk = sr_customer_sk
),
prefiltered_stores AS (
    SELECT 
        s_store_sk,
        s_store_name,
        s_company_id,
        s_street_number,
        s_street_name,
        s_street_type,
        s_suite_number,
        s_city,
        s_county,
        s_state,
        s_zip
    FROM store
    WHERE EXISTS (
        SELECT 1
        FROM joined_facts
        WHERE joined_facts.ss_store_sk = store.s_store_sk
    )
),
base_data AS (
    SELECT 
        s_store_name,
        s_company_id,
        s_street_number,
        s_street_name,
        s_street_type,
        s_suite_number,
        s_city,
        s_county,
        s_state,
        s_zip,
        sr_returned_date_sk - ss_sold_date_sk AS day_diff
    FROM joined_facts
    JOIN prefiltered_stores ON ss_store_sk = s_store_sk
)
SELECT
    s_store_name,
    s_company_id,
    s_street_number,
    s_street_name,
    s_street_type,
    s_suite_number,
    s_city,
    s_county,
    s_state,
    s_zip,
    SUM(CASE WHEN day_diff <= 30 THEN 1 ELSE 0 END) AS "30 days",
    SUM(CASE WHEN day_diff > 30 AND day_diff <= 60 THEN 1 ELSE 0 END) AS "31-60 days",
    SUM(CASE WHEN day_diff > 60 AND day_diff <= 90 THEN 1 ELSE 0 END) AS "61-90 days",
    SUM(CASE WHEN day_diff > 90 AND day_diff <= 120 THEN 1 ELSE 0 END) AS "91-120 days",
    SUM(CASE WHEN day_diff > 120 THEN 1 ELSE 0 END) AS ">120 days"
FROM base_data
GROUP BY
    s_store_name,
    s_company_id,
    s_street_number,
    s_street_name,
    s_street_type,
    s_suite_number,
    s_city,
    s_county,
    s_state,
    s_zip
ORDER BY
    s_store_name,
    s_company_id,
    s_street_number,
    s_street_name,
    s_street_type,
    s_suite_number,
    s_city,
    s_county,
    s_state,
    s_zip
LIMIT 100
```

Changes: Restructured using multiple CTEs to pre-filter date_dim (return dates), pre-join facts (store_sales with store_returns), and pre-filter store dimension using EXISTS to only include stores with matching sales. This reduces data early in the pipeline before the expensive GROUP BY.

Expected speedup: 2-3x by applying aggressive multi-CTE prefetch pattern, filtering dimensions early and reducing fact table joins before aggregation.
