```sql
WITH filtered_dates AS (
    SELECT d_date_sk
    FROM date_dim
    WHERE d_month_seq BETWEEN 1224 AND 1224 + 11
),
filtered_sales AS (
    SELECT 
        cs_sold_date_sk,
        cs_ship_date_sk,
        cs_warehouse_sk,
        cs_ship_mode_sk,
        cs_call_center_sk
    FROM catalog_sales
    JOIN filtered_dates ON cs_ship_date_sk = filtered_dates.d_date_sk
)
SELECT
    SUBSTRING(w_warehouse_name, 1, 20),
    sm_type,
    cc_name,
    SUM(CASE WHEN (cs_ship_date_sk - cs_sold_date_sk <= 30) THEN 1 ELSE 0 END) AS "30 days",
    SUM(CASE WHEN (cs_ship_date_sk - cs_sold_date_sk > 30 AND cs_ship_date_sk - cs_sold_date_sk <= 60) THEN 1 ELSE 0 END) AS "31-60 days",
    SUM(CASE WHEN (cs_ship_date_sk - cs_sold_date_sk > 60 AND cs_ship_date_sk - cs_sold_date_sk <= 90) THEN 1 ELSE 0 END) AS "61-90 days",
    SUM(CASE WHEN (cs_ship_date_sk - cs_sold_date_sk > 90 AND cs_ship_date_sk - cs_sold_date_sk <= 120) THEN 1 ELSE 0 END) AS "91-120 days",
    SUM(CASE WHEN (cs_ship_date_sk - cs_sold_date_sk > 120) THEN 1 ELSE 0 END) AS ">120 days"
FROM filtered_sales cs
JOIN warehouse w ON cs.cs_warehouse_sk = w.w_warehouse_sk
JOIN ship_mode sm ON cs.cs_ship_mode_sk = sm.sm_ship_mode_sk
JOIN call_center cc ON cs.cs_call_center_sk = cc.cc_call_center_sk
GROUP BY
    SUBSTRING(w_warehouse_name, 1, 20),
    sm_type,
    cc_name
ORDER BY
    SUBSTRING(w_warehouse_name, 1, 20),
    sm_type,
    cc_name
LIMIT 100
```

**Changes**: Isolated the filtered date dimension in a CTE first, then joined it with the fact table to create a filtered sales CTE containing only relevant date keys. This allows DuckDB to push the date filter down and reduce the fact table scan early.

**Expected speedup**: 1.5-2x by reducing the catalog_sales scan to only rows with ship dates in the specified month range, creating smaller hash tables for subsequent joins.
