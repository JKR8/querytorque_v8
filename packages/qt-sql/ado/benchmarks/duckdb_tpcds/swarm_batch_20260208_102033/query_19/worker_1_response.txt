```sql
WITH filtered_date AS (
    SELECT d_date_sk
    FROM date_dim
    WHERE d_moy = 12
      AND d_year = 1999
), filtered_item AS (
    SELECT i_item_sk, i_brand_id, i_brand, i_manufact_id, i_manufact
    FROM item
    WHERE i_manager_id = 2
)
SELECT
    i_brand_id AS brand_id,
    i_brand AS brand,
    i_manufact_id,
    i_manufact,
    SUM(ss_ext_sales_price) AS ext_price
FROM store_sales
JOIN filtered_date ON d_date_sk = ss_sold_date_sk
JOIN filtered_item ON ss_item_sk = i_item_sk
JOIN customer ON ss_customer_sk = c_customer_sk
JOIN customer_address ON c_current_addr_sk = ca_address_sk
JOIN store ON ss_store_sk = s_store_sk
WHERE SUBSTRING(ca_zip, 1, 5) <> SUBSTRING(s_zip, 1, 5)
GROUP BY
    i_brand,
    i_brand_id,
    i_manufact_id,
    i_manufact
ORDER BY
    ext_price DESC,
    i_brand,
    i_brand_id,
    i_manufact_id,
    i_manufact
LIMIT 100
```

Changes: Created CTEs to filter date_dim and item tables before joining to store_sales, reducing fact table rows early. Converted implicit joins to explicit JOIN syntax.

Expected speedup: 1.5-2x by pushing filters into CTEs and reducing fact table rows before the main join.
