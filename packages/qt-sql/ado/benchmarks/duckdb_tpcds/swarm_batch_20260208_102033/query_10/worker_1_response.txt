```sql
WITH filtered_date AS (
    SELECT d_date_sk
    FROM date_dim
    WHERE d_year = 2001
      AND d_moy BETWEEN 1 AND 4
),
filtered_address AS (
    SELECT ca_address_sk
    FROM customer_address
    WHERE ca_county IN (
        'Storey County',
        'Marquette County',
        'Warren County',
        'Cochran County',
        'Kandiyohi County'
    )
),
joined_customer AS (
    SELECT 
        c.c_customer_sk,
        cd.cd_gender,
        cd.cd_marital_status,
        cd.cd_education_status,
        cd.cd_purchase_estimate,
        cd.cd_credit_rating,
        cd.cd_dep_count,
        cd.cd_dep_employed_count,
        cd.cd_dep_college_count
    FROM customer c
    JOIN filtered_address fa ON c.c_current_addr_sk = fa.ca_address_sk
    JOIN customer_demographics cd ON cd.cd_demo_sk = c.c_current_cdemo_sk
    WHERE EXISTS (
        SELECT 1
        FROM store_sales, filtered_date fd
        WHERE c.c_customer_sk = ss_customer_sk
          AND ss_sold_date_sk = fd.d_date_sk
    )
      AND (
          EXISTS (
              SELECT 1
              FROM web_sales, filtered_date fd
              WHERE c.c_customer_sk = ws_bill_customer_sk
                AND ws_sold_date_sk = fd.d_date_sk
          )
          OR EXISTS (
              SELECT 1
              FROM catalog_sales, filtered_date fd
              WHERE c.c_customer_sk = cs_ship_customer_sk
                AND cs_sold_date_sk = fd.d_date_sk
          )
      )
)
SELECT
    cd_gender,
    cd_marital_status,
    cd_education_status,
    COUNT(*) AS cnt1,
    cd_purchase_estimate,
    COUNT(*) AS cnt2,
    cd_credit_rating,
    COUNT(*) AS cnt3,
    cd_dep_count,
    COUNT(*) AS cnt4,
    cd_dep_employed_count,
    COUNT(*) AS cnt5,
    cd_dep_college_count,
    COUNT(*) AS cnt6
FROM joined_customer
GROUP BY
    cd_gender,
    cd_marital_status,
    cd_education_status,
    cd_purchase_estimate,
    cd_credit_rating,
    cd_dep_count,
    cd_dep_employed_count,
    cd_dep_college_count
ORDER BY
    cd_gender,
    cd_marital_status,
    cd_education_status,
    cd_purchase_estimate,
    cd_credit_rating,
    cd_dep_count,
    cd_dep_employed_count,
    cd_dep_college_count
LIMIT 100
```

**Changes**: 
1. Created filtered_date CTE to isolate date dimension filtering (d_year=2001, d_moy 1-4)
2. Created filtered_address CTE to push county filter early
3. Created joined_customer CTE that combines customer filtering with address and demographics, and moves EXISTS subqueries inside to filter before aggregation
4. Maintained EXISTS semantics exactly as original (no conversion to JOINs)

**Expected speedup**: 1.5-2x through early reduction of customer rows before main aggregation, while preserving exact semantics.
