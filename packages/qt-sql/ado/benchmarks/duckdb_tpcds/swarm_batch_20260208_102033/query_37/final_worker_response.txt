```sql
WITH filtered_items AS (
    SELECT 
        i_item_sk,
        i_item_id,
        i_item_desc,
        i_current_price
    FROM item
    WHERE i_current_price BETWEEN 45 AND 45 + 30
        AND i_manufact_id IN (856, 707, 1000, 747)
        AND EXISTS (
            SELECT 1
            FROM catalog_sales
            WHERE cs_item_sk = item.i_item_sk
        )
),
filtered_inventory AS (
    SELECT DISTINCT
        inv_item_sk,
        inv_date_sk
    FROM inventory
    WHERE inv_quantity_on_hand BETWEEN 100 AND 500
),
filtered_dates AS (
    SELECT d_date_sk
    FROM date_dim
    WHERE d_date BETWEEN CAST('1999-02-21' AS DATE) 
                    AND (CAST('1999-02-21' AS DATE) + INTERVAL '60' DAY)
)
SELECT DISTINCT
    fi.i_item_id,
    fi.i_item_desc,
    fi.i_current_price
FROM filtered_items fi
JOIN filtered_inventory inv ON fi.i_item_sk = inv.inv_item_sk
JOIN filtered_dates d ON inv.inv_date_sk = d.d_date_sk
ORDER BY fi.i_item_id
LIMIT 100
```

Changes: Created CTEs to filter items by price/manufacturer and check catalog_sales existence via EXISTS before joining with inventory and date_dim. Used DISTINCT instead of GROUP BY to remove duplicates from inventory joins while preserving semantics.

Expected speedup: ~3-5x by pushing catalog_sales existence check early and reducing join fan-out before inventory joins.
