```sql
WITH filtered_date AS (
    SELECT d_date_sk
    FROM date_dim
    WHERE d_year = 2000
),
filtered_ca AS (
    SELECT ca_address_sk
    FROM customer_address
    WHERE ca_country = 'United States'
      AND ca_state IN ('FL', 'TX', 'DE', 'IN', 'ND', 'ID', 'MT', 'IL', 'OH')
),
filtered_cd AS (
    SELECT cd_demo_sk, cd_marital_status, cd_education_status
    FROM customer_demographics
    WHERE (cd_marital_status = 'M' AND cd_education_status = '4 yr Degree')
       OR (cd_marital_status = 'S' AND cd_education_status = 'Secondary')
       OR (cd_marital_status = 'W' AND cd_education_status = 'Advanced Degree')
),
prejoined_fact AS (
    SELECT 
        ws_web_page_sk,
        ws_item_sk,
        ws_order_number,
        ws_quantity,
        ws_sales_price,
        ws_net_profit,
        wr_refunded_cdemo_sk,
        wr_returning_cdemo_sk,
        wr_refunded_addr_sk,
        wr_reason_sk,
        wr_refunded_cash,
        wr_fee,
        d_date_sk
    FROM web_sales
    JOIN filtered_date ON ws_sold_date_sk = d_date_sk
    JOIN web_returns ON ws_item_sk = wr_item_sk 
                     AND ws_order_number = wr_order_number
),
filtered_join AS (
    SELECT 
        p.*,
        r_reason_desc,
        cd1.cd_marital_status AS cd1_ms,
        cd1.cd_education_status AS cd1_es,
        cd2.cd_marital_status AS cd2_ms,
        cd2.cd_education_status AS cd2_es,
        ca.ca_address_sk
    FROM prejoined_fact p
    JOIN web_page wp ON p.ws_web_page_sk = wp_web_page_sk
    JOIN filtered_cd cd1 ON p.wr_refunded_cdemo_sk = cd1.cd_demo_sk
    JOIN filtered_cd cd2 ON p.wr_returning_cdemo_sk = cd2.cd_demo_sk
    JOIN filtered_ca ca ON p.wr_refunded_addr_sk = ca.ca_address_sk
    JOIN reason r ON p.wr_reason_sk = r_reason_sk
    WHERE cd1.cd_marital_status = cd2.cd_marital_status
      AND cd1.cd_education_status = cd2.cd_education_status
      AND (
          (cd1.cd_marital_status = 'M' AND cd1.cd_education_status = '4 yr Degree' AND ws_sales_price BETWEEN 100.00 AND 150.00)
          OR (cd1.cd_marital_status = 'S' AND cd1.cd_education_status = 'Secondary' AND ws_sales_price BETWEEN 50.00 AND 100.00)
          OR (cd1.cd_marital_status = 'W' AND cd1.cd_education_status = 'Advanced Degree' AND ws_sales_price BETWEEN 150.00 AND 200.00)
      )
      AND (
          (ca_state IN ('FL', 'TX', 'DE') AND ws_net_profit BETWEEN 100 AND 200)
          OR (ca_state IN ('IN', 'ND', 'ID') AND ws_net_profit BETWEEN 150 AND 300)
          OR (ca_state IN ('MT', 'IL', 'OH') AND ws_net_profit BETWEEN 50 AND 250)
      )
)
SELECT
    SUBSTRING(r_reason_desc, 1, 20),
    AVG(ws_quantity),
    AVG(wr_refunded_cash),
    AVG(wr_fee)
FROM filtered_join
GROUP BY r_reason_desc
ORDER BY
    SUBSTRING(r_reason_desc, 1, 20),
    AVG(ws_quantity),
    AVG(wr_refunded_cash),
    AVG(wr_fee)
LIMIT 100;
```

**Changes**: Created early CTEs to filter dimension tables (date_dim, customer_address, customer_demographics), then pre-joined the filtered dimensions with web_sales and web_returns in a single pass, consolidating all OR conditions into the final filtered_join CTE to minimize repeated table scans.

**Expected speedup**: ~2-3x by reducing dimension table scans and enabling better join order optimization through staged pre-joins.
