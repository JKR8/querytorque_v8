```sql
WITH filtered_item AS (
  SELECT
    i_item_sk,
    i_manufact_id
  FROM item
  WHERE i_category = 'Home'
),
filtered_date AS (
  SELECT d_date_sk
  FROM date_dim
  WHERE d_year = 2002
    AND d_moy = 1
),
filtered_addr AS (
  SELECT ca_address_sk
  FROM customer_address
  WHERE ca_gmt_offset = -5
),
unified_sales AS (
  -- Store sales
  SELECT
    i.i_manufact_id,
    ss.ss_ext_sales_price AS sales_price
  FROM store_sales ss
  INNER JOIN filtered_item i ON ss.ss_item_sk = i.i_item_sk
  INNER JOIN filtered_date d ON ss.ss_sold_date_sk = d.d_date_sk
  INNER JOIN filtered_addr ca ON ss.ss_addr_sk = ca.ca_address_sk
  
  UNION ALL
  
  -- Catalog sales
  SELECT
    i.i_manufact_id,
    cs.cs_ext_sales_price AS sales_price
  FROM catalog_sales cs
  INNER JOIN filtered_item i ON cs.cs_item_sk = i.i_item_sk
  INNER JOIN filtered_date d ON cs.cs_sold_date_sk = d.d_date_sk
  INNER JOIN filtered_addr ca ON cs.cs_bill_addr_sk = ca.ca_address_sk
  
  UNION ALL
  
  -- Web sales
  SELECT
    i.i_manufact_id,
    ws.ws_ext_sales_price AS sales_price
  FROM web_sales ws
  INNER JOIN filtered_item i ON ws.ws_item_sk = i.i_item_sk
  INNER JOIN filtered_date d ON ws.ws_sold_date_sk = d.d_date_sk
  INNER JOIN filtered_addr ca ON ws.ws_bill_addr_sk = ca.ca_address_sk
)
SELECT
  i_manufact_id,
  SUM(sales_price) AS total_sales
FROM unified_sales
GROUP BY i_manufact_id
ORDER BY total_sales
LIMIT 100
```

Changes: Pre-materialized dimension keys into small hash tables (filtered_item, filtered_date, filtered_addr) and consolidated all three sales channels into a single scan pattern using UNION ALL with predicate pushdown to raw fact table scans. This enables single-pass aggregation over a unified sales stream, eliminating redundant joins and enabling parallel fact table scans.

Expected speedup: 2.5-3.0x due to reduced I/O from early dimension filtering and elimination of redundant processing across three separate CTEs.
