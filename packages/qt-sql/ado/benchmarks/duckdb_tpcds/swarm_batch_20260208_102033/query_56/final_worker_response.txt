```sql
WITH filtered_item AS (
  SELECT i_item_sk, i_item_id
  FROM item
  WHERE i_color IN ('powder', 'green', 'cyan')
),
filtered_date AS (
  SELECT d_date_sk
  FROM date_dim
  WHERE d_year = 2000
    AND d_moy = 2
),
filtered_ca AS (
  SELECT ca_address_sk
  FROM customer_address
  WHERE ca_gmt_offset = -6
),
ss AS (
  SELECT
    i.i_item_id,
    SUM(ss_ext_sales_price) AS total_sales
  FROM store_sales
  WHERE ss_item_sk IN (SELECT i_item_sk FROM filtered_item)
    AND ss_sold_date_sk IN (SELECT d_date_sk FROM filtered_date)
    AND ss_addr_sk IN (SELECT ca_address_sk FROM filtered_ca)
  GROUP BY i.i_item_id
),
cs AS (
  SELECT
    i.i_item_id,
    SUM(cs_ext_sales_price) AS total_sales
  FROM catalog_sales
  WHERE cs_item_sk IN (SELECT i_item_sk FROM filtered_item)
    AND cs_sold_date_sk IN (SELECT d_date_sk FROM filtered_date)
    AND cs_bill_addr_sk IN (SELECT ca_address_sk FROM filtered_ca)
  GROUP BY i.i_item_id
),
ws AS (
  SELECT
    i.i_item_id,
    SUM(ws_ext_sales_price) AS total_sales
  FROM web_sales
  WHERE ws_item_sk IN (SELECT i_item_sk FROM filtered_item)
    AND ws_sold_date_sk IN (SELECT d_date_sk FROM filtered_date)
    AND ws_bill_addr_sk IN (SELECT ca_address_sk FROM filtered_ca)
  GROUP BY i.i_item_id
)
SELECT
  i_item_id,
  SUM(total_sales) AS total_sales
FROM (
  SELECT * FROM ss
  UNION ALL
  SELECT * FROM cs
  UNION ALL
  SELECT * FROM ws
) AS tmp1
GROUP BY i_item_id
ORDER BY total_sales, i_item_id
LIMIT 100
```

Changes: Created filtered dimension CTEs for item, date_dim, and customer_address, then rewrote each fact table scan using IN subqueries to apply dimension filters via semi-joins before aggregation. This reduces the fact table rows early by leveraging semi-join optimization.

Expected speedup: 2-3x by filtering large fact tables before joins using dimension key semi-joins.
