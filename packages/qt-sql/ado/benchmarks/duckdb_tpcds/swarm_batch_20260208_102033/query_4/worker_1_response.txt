```sql
WITH filtered_date AS (
  SELECT d_date_sk, d_year
  FROM date_dim
  WHERE d_year IN (1999, 2000)
),
customer_prefetch AS (
  SELECT
    c_customer_sk,
    c_customer_id,
    c_first_name,
    c_last_name,
    c_birth_country
  FROM customer
),
store_sales_totals AS (
  SELECT
    c.c_customer_id AS customer_id,
    c.c_first_name AS customer_first_name,
    c.c_last_name AS customer_last_name,
    c.c_birth_country AS customer_birth_country,
    fd.d_year AS dyear,
    SUM(
      (
        (
          ss_ext_list_price - ss_ext_wholesale_cost - ss_ext_discount_amt
        ) + ss_ext_sales_price
      ) / 2
    ) AS year_total,
    's' AS sale_type
  FROM customer_prefetch c
  JOIN store_sales ss ON c.c_customer_sk = ss_customer_sk
  JOIN filtered_date fd ON ss_sold_date_sk = fd.d_date_sk
  GROUP BY
    c.c_customer_id,
    c.c_first_name,
    c.c_last_name,
    c.c_birth_country,
    fd.d_year
),
catalog_sales_totals AS (
  SELECT
    c.c_customer_id AS customer_id,
    c.c_first_name AS customer_first_name,
    c.c_last_name AS customer_last_name,
    c.c_birth_country AS customer_birth_country,
    fd.d_year AS dyear,
    SUM(
      (
        (
          cs_ext_list_price - cs_ext_wholesale_cost - cs_ext_discount_amt
        ) + cs_ext_sales_price
      ) / 2
    ) AS year_total,
    'c' AS sale_type
  FROM customer_prefetch c
  JOIN catalog_sales cs ON c.c_customer_sk = cs_bill_customer_sk
  JOIN filtered_date fd ON cs_sold_date_sk = fd.d_date_sk
  GROUP BY
    c.c_customer_id,
    c.c_first_name,
    c.c_last_name,
    c.c_birth_country,
    fd.d_year
),
web_sales_totals AS (
  SELECT
    c.c_customer_id AS customer_id,
    c.c_first_name AS customer_first_name,
    c.c_last_name AS customer_last_name,
    c.c_birth_country AS customer_birth_country,
    fd.d_year AS dyear,
    SUM(
      (
        (
          ws_ext_list_price - ws_ext_wholesale_cost - ws_ext_discount_amt
        ) + ws_ext_sales_price
      ) / 2
    ) AS year_total,
    'w' AS sale_type
  FROM customer_prefetch c
  JOIN web_sales ws ON c.c_customer_sk = ws_bill_customer_sk
  JOIN filtered_date fd ON ws_sold_date_sk = fd.d_date_sk
  GROUP BY
    c.c_customer_id,
    c.c_first_name,
    c.c_last_name,
    c.c_birth_country,
    fd.d_year
)
SELECT
  t_s_secyear.customer_id,
  t_s_secyear.customer_first_name,
  t_s_secyear.customer_last_name,
  t_s_secyear.customer_birth_country
FROM store_sales_totals t_s_firstyear
JOIN store_sales_totals t_s_secyear 
  ON t_s_firstyear.customer_id = t_s_secyear.customer_id
JOIN catalog_sales_totals t_c_firstyear 
  ON t_s_firstyear.customer_id = t_c_firstyear.customer_id
JOIN catalog_sales_totals t_c_secyear 
  ON t_s_firstyear.customer_id = t_c_secyear.customer_id
JOIN web_sales_totals t_w_firstyear 
  ON t_s_firstyear.customer_id = t_w_firstyear.customer_id
JOIN web_sales_totals t_w_secyear 
  ON t_s_firstyear.customer_id = t_w_secyear.customer_id
WHERE t_s_firstyear.dyear = 1999
  AND t_s_secyear.dyear = 2000
  AND t_c_firstyear.dyear = 1999
  AND t_c_secyear.dyear = 2000
  AND t_w_firstyear.dyear = 1999
  AND t_w_secyear.dyear = 2000
  AND t_s_firstyear.year_total > 0
  AND t_c_firstyear.year_total > 0
  AND t_w_firstyear.year_total > 0
  AND CASE
    WHEN t_c_firstyear.year_total > 0
    THEN t_c_secyear.year_total / t_c_firstyear.year_total
    ELSE NULL
  END > CASE
    WHEN t_s_firstyear.year_total > 0
    THEN t_s_secyear.year_total / t_s_firstyear.year_total
    ELSE NULL
  END
  AND CASE
    WHEN t_c_firstyear.year_total > 0
    THEN t_c_secyear.year_total / t_c_firstyear.year_total
    ELSE NULL
  END > CASE
    WHEN t_w_firstyear.year_total > 0
    THEN t_w_secyear.year_total / t_w_firstyear.year_total
    ELSE NULL
  END
ORDER BY
  t_s_secyear.customer_id,
  t_s_secyear.customer_first_name,
  t_s_secyear.customer_last_name,
  t_s_secyear.customer_birth_country
LIMIT 100
```

**Changes**: 
1. Created `filtered_date` CTE to isolate date_dim filtering (only years 1999-2000)
2. Created `customer_prefetch` CTE to isolate customer dimension with only needed columns
3. Split the original UNION into three separate CTEs, each with early filters on date and dimension tables
4. Converted implicit joins to explicit JOIN syntax for better optimizer planning
5. Removed unnecessary columns (customer_preferred_cust_flag, customer_login, customer_email_address) from aggregation CTEs since they're not used in final output

**Expected speedup**: ~2-3x due to reduced row counts in early stages, better join selectivity, and elimination of union overhead.
