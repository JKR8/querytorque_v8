WITH filtered_item AS (SELECT i_item_sk, i_item_id FROM item WHERE i_current_price BETWEEN 0.99 AND 1.49), joined_inventory AS (SELECT w.w_warehouse_name, fi.i_item_id, d.d_date, inv.inv_quantity_on_hand FROM warehouse AS w INNER JOIN inventory AS inv ON w.w_warehouse_sk = inv.inv_warehouse_sk INNER JOIN filtered_item AS fi ON inv.inv_item_sk = fi.i_item_sk INNER JOIN date_dim AS d ON inv.inv_date_sk = d.d_date_sk WHERE d.d_date BETWEEN CAST('2002-01-28' AS DATE) AND CAST('2002-03-29' AS DATE)) SELECT w_warehouse_name, i_item_id, SUM(CASE WHEN d_date < CAST('2002-02-27' AS DATE) THEN inv_quantity_on_hand ELSE 0 END) AS inv_before, SUM(CASE WHEN d_date >= CAST('2002-02-27' AS DATE) THEN inv_quantity_on_hand ELSE 0 END) AS inv_after FROM joined_inventory GROUP BY w_warehouse_name, i_item_id HAVING (CASE WHEN inv_before > 0 THEN inv_after / inv_before ELSE NULL END) BETWEEN 2.0 / 3.0 AND 3.0 / 2.0 ORDER BY w_warehouse_name, i_item_id LIMIT 100;
