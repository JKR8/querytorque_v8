```sql
WITH may_1998_date AS (
    SELECT
        d_date_sk,
        d_month_seq
    FROM date_dim
    WHERE d_year = 1998
      AND d_moy = 5
),
filtered_items AS (
    SELECT
        i_item_sk
    FROM item
    WHERE i_category = 'Women'
      AND i_class = 'maternity'
),
catalog_customers AS (
    SELECT DISTINCT
        cs_bill_customer_sk AS customer_sk
    FROM catalog_sales
    INNER JOIN filtered_items ON cs_item_sk = i_item_sk
    INNER JOIN may_1998_date ON cs_sold_date_sk = d_date_sk
),
web_customers AS (
    SELECT DISTINCT
        ws_bill_customer_sk AS customer_sk
    FROM web_sales
    INNER JOIN filtered_items ON ws_item_sk = i_item_sk
    INNER JOIN may_1998_date ON ws_sold_date_sk = d_date_sk
),
my_customers AS (
    SELECT
        c_customer_sk,
        c_current_addr_sk
    FROM customer
    WHERE c_customer_sk IN (
        SELECT customer_sk FROM catalog_customers
        UNION ALL
        SELECT customer_sk FROM web_customers
    )
),
month_range AS (
    SELECT
        d_month_seq + 1 AS start_seq,
        d_month_seq + 3 AS end_seq
    FROM may_1998_date
),
my_revenue AS (
    SELECT
        c_customer_sk,
        SUM(ss_ext_sales_price) AS revenue
    FROM my_customers
    INNER JOIN store_sales ON c_customer_sk = ss_customer_sk
    INNER JOIN customer_address ON c_current_addr_sk = ca_address_sk
    INNER JOIN store ON ca_county = s_county AND ca_state = s_state
    INNER JOIN date_dim ON ss_sold_date_sk = d_date_sk
    CROSS JOIN month_range
    WHERE d_month_seq BETWEEN start_seq AND end_seq
    GROUP BY c_customer_sk
),
segments AS (
    SELECT
        CAST((revenue / 50) AS INT) AS segment
    FROM my_revenue
)
SELECT
    segment,
    COUNT(*) AS num_customers,
    segment * 50 AS segment_base
FROM segments
GROUP BY segment
ORDER BY segment, num_customers
LIMIT 100;
```

**Changes:** 
1. Extracted shared dimension filters (`may_1998_date` for date_dim and `filtered_items` for item) to avoid redundant scans
2. Split the UNION ALL into separate CTEs (`catalog_customers`, `web_customers`) to push dimension filters deeper
3. Materialized the month range calculation (`month_range`) to replace correlated subqueries
4. Converted implicit joins to explicit JOIN syntax for better optimizer control
