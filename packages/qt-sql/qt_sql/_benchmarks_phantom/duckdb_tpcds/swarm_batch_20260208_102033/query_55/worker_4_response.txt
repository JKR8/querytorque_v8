```sql
WITH filtered_dates AS (
    SELECT d_date_sk
    FROM date_dim
    WHERE d_moy = 12
      AND d_year = 2000
), filtered_items AS (
    SELECT i_item_sk, i_brand_id, i_brand
    FROM item
    WHERE i_manager_id = 100
)
SELECT
    i.i_brand_id AS brand_id,
    i.i_brand AS brand,
    SUM(ss.ss_ext_sales_price) AS ext_price
FROM store_sales ss
JOIN filtered_dates d ON ss.ss_sold_date_sk = d.d_date_sk
JOIN filtered_items i ON ss.ss_item_sk = i.i_item_sk
GROUP BY i.i_brand, i.i_brand_id
ORDER BY ext_price DESC, i.i_brand_id
LIMIT 100
```

```
Changes: Isolated date and item filtering into separate CTEs to enable better join order planning and early row reduction.
Expected speedup: 1.3-1.7x through reduced intermediate result sizes and improved filter pushdown.
```
