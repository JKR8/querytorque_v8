```sql
WITH filtered_customers AS (
    SELECT
        c_customer_sk,
        cd_marital_status,
        cd_education_status
    FROM customer
    JOIN customer_demographics ON cd_demo_sk = c_current_cdemo_sk
    JOIN household_demographics ON hd_demo_sk = c_current_hdemo_sk
    JOIN customer_address ON ca_address_sk = c_current_addr_sk
    WHERE (
        (cd_marital_status = 'M' AND cd_education_status = 'Unknown')
        OR (cd_marital_status = 'W' AND cd_education_status = 'Advanced Degree')
    )
    AND hd_buy_potential LIKE '1001-5000%'
    AND ca_gmt_offset = -6
),
filtered_returns AS (
    SELECT
        cr_call_center_sk,
        cr_net_loss,
        cd_marital_status,
        cd_education_status
    FROM catalog_returns
    JOIN date_dim ON cr_returned_date_sk = d_date_sk
    JOIN filtered_customers ON cr_returning_customer_sk = c_customer_sk
    WHERE d_year = 2001
    AND d_moy = 11
)
SELECT
    cc_call_center_id AS Call_Center,
    cc_name AS Call_Center_Name,
    cc_manager AS Manager,
    SUM(cr_net_loss) AS Returns_Loss
FROM call_center
JOIN filtered_returns ON cr_call_center_sk = cc_call_center_sk
GROUP BY
    cc_call_center_id,
    cc_name,
    cc_manager,
    cd_marital_status,
    cd_education_status
ORDER BY
    SUM(cr_net_loss) DESC
```

Changes: Created two CTEs - first to pre-join and filter all customer-related dimensions into a small set of customer keys, second to filter catalog_returns by date and join with filtered customers. This ensures the largest table (catalog_returns) is reduced early using selective filters before joining to call_center.

Expected speedup: 3-4x due to early reduction of fact table via selective dimension filters and date restriction.
