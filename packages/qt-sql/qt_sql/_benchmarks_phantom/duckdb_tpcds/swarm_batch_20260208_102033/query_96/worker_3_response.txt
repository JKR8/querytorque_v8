```sql
WITH filtered_time AS (
    SELECT t_time_sk
    FROM time_dim
    WHERE t_hour = 8
      AND t_minute >= 30
),
filtered_household AS (
    SELECT hd_demo_sk
    FROM household_demographics
    WHERE hd_dep_count = 3
),
filtered_store AS (
    SELECT s_store_sk
    FROM store
    WHERE s_store_name = 'ese'
),
filtered_sales AS (
    SELECT ss_sold_time_sk, ss_hdemo_sk, ss_store_sk
    FROM store_sales
    WHERE ss_sold_time_sk IN (SELECT t_time_sk FROM filtered_time)
      AND ss_hdemo_sk IN (SELECT hd_demo_sk FROM filtered_household)
      AND ss_store_sk IN (SELECT s_store_sk FROM filtered_store)
)
SELECT
  COUNT(*)
FROM filtered_sales
ORDER BY
  COUNT(*)
LIMIT 100
```

**Changes**: Isolated dimension filters into separate CTEs, then applied them as semi-joins against the fact table in a staged pipeline. This reduces the fact table scan early by pushing down dimension keys.

**Expected speedup**: 2-3x from reduced join cardinality and early fact table filtering.
