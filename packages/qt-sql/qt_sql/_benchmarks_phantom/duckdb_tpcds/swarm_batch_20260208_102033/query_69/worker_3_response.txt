```sql
WITH filtered_dates AS (
  SELECT d_date_sk
  FROM date_dim
  WHERE d_year = 2000
    AND d_moy BETWEEN 1 AND 3
),
store_customers AS (
  SELECT DISTINCT ss_customer_sk
  FROM store_sales
  JOIN filtered_dates ON ss_sold_date_sk = d_date_sk
),
web_customers AS (
  SELECT DISTINCT ws_bill_customer_sk
  FROM web_sales
  JOIN filtered_dates ON ws_sold_date_sk = d_date_sk
),
catalog_customers AS (
  SELECT DISTINCT cs_ship_customer_sk
  FROM catalog_sales
  JOIN filtered_dates ON cs_sold_date_sk = d_date_sk
)
SELECT
  cd_gender,
  cd_marital_status,
  cd_education_status,
  COUNT(*) AS cnt1,
  cd_purchase_estimate,
  COUNT(*) AS cnt2,
  cd_credit_rating,
  COUNT(*) AS cnt3
FROM customer AS c
JOIN customer_address AS ca ON c.c_current_addr_sk = ca.ca_address_sk
JOIN customer_demographics ON cd_demo_sk = c.c_current_cdemo_sk
JOIN store_customers ON c.c_customer_sk = store_customers.ss_customer_sk
LEFT JOIN web_customers ON c.c_customer_sk = web_customers.ws_bill_customer_sk
LEFT JOIN catalog_customers ON c.c_customer_sk = catalog_customers.cs_ship_customer_sk
WHERE ca_state IN ('TX', 'VA', 'MI')
  AND web_customers.ws_bill_customer_sk IS NULL
  AND catalog_customers.cs_ship_customer_sk IS NULL
GROUP BY
  cd_gender,
  cd_marital_status,
  cd_education_status,
  cd_purchase_estimate,
  cd_credit_rating
ORDER BY
  cd_gender,
  cd_marital_status,
  cd_education_status,
  cd_purchase_estimate,
  cd_credit_rating
LIMIT 100
```

Changes: Decorrelated the three EXISTS/NOT EXISTS subqueries by creating separate CTEs for store, web, and catalog customers who transacted in Jan-Mar 2000, then converted to explicit joins and anti-joins. Created a shared date CTE to avoid redundant date_dim scans.

Expected speedup: ~2.5x by eliminating correlated subquery overhead and enabling better join planning.
