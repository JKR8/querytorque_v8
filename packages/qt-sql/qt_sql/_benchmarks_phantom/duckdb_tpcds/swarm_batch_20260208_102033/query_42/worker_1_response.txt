```sql
WITH filtered_dates AS (
    SELECT d_date_sk, d_year
    FROM date_dim
    WHERE d_moy = 11
      AND d_year = 2002
), filtered_items AS (
    SELECT i_item_sk, i_category_id, i_category
    FROM item
    WHERE i_manager_id = 1
)
SELECT
    fd.d_year,
    fi.i_category_id,
    fi.i_category,
    SUM(ss_ext_sales_price) AS "SUM(ss_ext_sales_price)"
FROM store_sales ss
JOIN filtered_dates fd ON fd.d_date_sk = ss.ss_sold_date_sk
JOIN filtered_items fi ON fi.i_item_sk = ss.ss_item_sk
GROUP BY
    fd.d_year,
    fi.i_category_id,
    fi.i_category
ORDER BY
    SUM(ss_ext_sales_price) DESC,
    fd.d_year,
    fi.i_category_id,
    fi.i_category
LIMIT 100
```

```
Changes: Isolated filtered dimension tables (date_dim and item) into CTEs with early predicate pushdown before joining with the large fact table store_sales.
Expected speedup: 2-4x by reducing fact table scanning and enabling hash join with pre-filtered dimension tables.
```
