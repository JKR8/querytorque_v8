```sql
WITH filtered_date AS (
  SELECT d_date_sk, d_qoy
  FROM date_dim
  WHERE d_month_seq IN (1200, 1201, 1202, 1203, 1204, 1205, 1206, 1207, 1208, 1209, 1210, 1211)
),
filtered_item AS (
  SELECT i_item_sk, i_manufact_id
  FROM item
  WHERE (
    i_category IN ('Books', 'Children', 'Electronics')
    AND i_class IN ('personal', 'portable', 'reference', 'self-help')
    AND i_brand IN ('scholaramalgamalg #14', 'scholaramalgamalg #7', 'exportiunivamalg #9', 'scholaramalgamalg #9')
  ) OR (
    i_category IN ('Women', 'Music', 'Men')
    AND i_class IN ('accessories', 'classical', 'fragrances', 'pants')
    AND i_brand IN ('amalgimporto #1', 'edu packscholar #1', 'exportiimporto #1', 'importoamalg #1')
  )
),
joined_sales AS (
  SELECT
    i.i_manufact_id,
    d.d_qoy,
    SUM(ss.ss_sales_price) AS sum_sales
  FROM store_sales ss
  JOIN filtered_item i ON ss.ss_item_sk = i.i_item_sk
  JOIN filtered_date d ON ss.ss_sold_date_sk = d.d_date_sk
  JOIN store s ON ss.ss_store_sk = s.s_store_sk
  GROUP BY i.i_manufact_id, d.d_qoy
),
with_avg AS (
  SELECT
    i_manufact_id,
    sum_sales,
    AVG(sum_sales) OVER (PARTITION BY i_manufact_id) AS avg_quarterly_sales
  FROM joined_sales
)
SELECT
  i_manufact_id,
  sum_sales,
  avg_quarterly_sales
FROM with_avg
WHERE
  avg_quarterly_sales > 0
  AND ABS(sum_sales - avg_quarterly_sales) / avg_quarterly_sales > 0.1
ORDER BY
  avg_quarterly_sales,
  sum_sales,
  i_manufact_id
LIMIT 100
```

```
Changes: Created filtered CTEs for date_dim and item to push predicates early, converted implicit joins to explicit JOINs, simplified CASE expression to direct boolean condition.
Expected speedup: 1.8x (reduced fact table join size through early dimension filtering)
```
