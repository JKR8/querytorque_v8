```sql
WITH filtered_dates AS (
    SELECT d_date_sk, d_date
    FROM date_dim
    WHERE d_date BETWEEN (
        CAST('2001-04-02' AS DATE) - INTERVAL '30' DAY
    ) AND (
        CAST('2001-04-02' AS DATE) + INTERVAL '30' DAY
    )
),
filtered_items AS (
    SELECT i_item_sk, i_item_id, i_current_price
    FROM item
    WHERE i_current_price BETWEEN 0.99 AND 1.49
),
sales_data AS (
    SELECT
        w.w_state,
        i.i_item_id,
        d.d_date,
        cs.cs_sales_price - COALESCE(cr.cr_refunded_cash, 0) AS net_amount
    FROM catalog_sales cs
    LEFT JOIN catalog_returns cr
        ON cs.cs_order_number = cr.cr_order_number
        AND cs.cs_item_sk = cr.cr_item_sk
    JOIN warehouse w ON cs.cs_warehouse_sk = w.w_warehouse_sk
    JOIN filtered_items i ON cs.cs_item_sk = i.i_item_sk
    JOIN filtered_dates d ON cs.cs_sold_date_sk = d.d_date_sk
)
SELECT
    w_state,
    i_item_id,
    SUM(CASE WHEN d_date < CAST('2001-04-02' AS DATE) THEN net_amount ELSE 0 END) AS sales_before,
    SUM(CASE WHEN d_date >= CAST('2001-04-02' AS DATE) THEN net_amount ELSE 0 END) AS sales_after
FROM sales_data
GROUP BY w_state, i_item_id
ORDER BY w_state, i_item_id
LIMIT 100
```

Changes: Converted implicit joins to explicit JOIN syntax, extracted filtered dimension tables into CTEs to reduce join sizes early, and preserved the original conditional aggregation logic while improving readability.

Expected speedup: 1.3-1.5x from early filtering and better join optimization
