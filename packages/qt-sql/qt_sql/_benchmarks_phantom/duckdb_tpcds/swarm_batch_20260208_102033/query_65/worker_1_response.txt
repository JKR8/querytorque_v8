```sql
WITH date_filter AS (
    SELECT d_date_sk
    FROM date_dim
    WHERE d_month_seq BETWEEN 1221 AND 1232
),
store_item_revenue AS (
    SELECT
        ss_store_sk,
        ss_item_sk,
        SUM(ss_sales_price) AS revenue
    FROM store_sales
    JOIN date_filter ON ss_sold_date_sk = d_date_sk
    GROUP BY ss_store_sk, ss_item_sk
),
store_average AS (
    SELECT
        ss_store_sk,
        AVG(revenue) AS ave
    FROM store_item_revenue
    GROUP BY ss_store_sk
)
SELECT
    s_store_name,
    i_item_desc,
    sc.revenue,
    i_current_price,
    i_wholesale_cost,
    i_brand
FROM store_item_revenue sc
JOIN store_average sb ON sc.ss_store_sk = sb.ss_store_sk
JOIN store ON s_store_sk = sc.ss_store_sk
JOIN item ON i_item_sk = sc.ss_item_sk
WHERE sc.revenue <= 0.1 * sb.ave
ORDER BY s_store_name, i_item_desc
LIMIT 100;
```

**Changes**: Consolidated duplicate fact table scans into a single CTE with early date filtering, then derived store averages from the materialized per-store-item revenues.

**Expected speedup**: ~1.8-2.0x from eliminating redundant fact table scans and reducing intermediate data volume through early filtering.
