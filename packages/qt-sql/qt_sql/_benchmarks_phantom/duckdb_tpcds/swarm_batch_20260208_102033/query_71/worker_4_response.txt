```sql
WITH filtered_time AS (
    SELECT t_time_sk, t_hour, t_minute
    FROM time_dim
    WHERE t_meal_time = 'breakfast' OR t_meal_time = 'dinner'
),
web_sales_filtered AS (
    SELECT
        ws_ext_sales_price AS ext_price,
        ws_item_sk AS sold_item_sk,
        t.t_hour,
        t.t_minute
    FROM web_sales
    JOIN date_dim d ON d.d_date_sk = ws_sold_date_sk
    JOIN filtered_time t ON ws_sold_time_sk = t.t_time_sk
    WHERE d.d_moy = 12 AND d.d_year = 1998
),
catalog_sales_filtered AS (
    SELECT
        cs_ext_sales_price AS ext_price,
        cs_item_sk AS sold_item_sk,
        t.t_hour,
        t.t_minute
    FROM catalog_sales
    JOIN date_dim d ON d.d_date_sk = cs_sold_date_sk
    JOIN filtered_time t ON cs_sold_time_sk = t.t_time_sk
    WHERE d.d_moy = 12 AND d.d_year = 1998
),
store_sales_filtered AS (
    SELECT
        ss_ext_sales_price AS ext_price,
        ss_item_sk AS sold_item_sk,
        t.t_hour,
        t.t_minute
    FROM store_sales
    JOIN date_dim d ON d.d_date_sk = ss_sold_date_sk
    JOIN filtered_time t ON ss_sold_time_sk = t.t_time_sk
    WHERE d.d_moy = 12 AND d.d_year = 1998
),
union_sales AS (
    SELECT * FROM web_sales_filtered
    UNION ALL
    SELECT * FROM catalog_sales_filtered
    UNION ALL
    SELECT * FROM store_sales_filtered
)
SELECT
    i.i_brand_id AS brand_id,
    i.i_brand AS brand,
    u.t_hour,
    u.t_minute,
    SUM(u.ext_price) AS ext_price
FROM item i
JOIN union_sales u ON i.i_item_sk = u.sold_item_sk
WHERE i.i_manager_id = 1
GROUP BY
    i.i_brand,
    i.i_brand_id,
    u.t_hour,
    u.t_minute
ORDER BY
    ext_price DESC,
    i_brand_id
```

**Changes**: Restructured the query by (1) creating a filtered time CTE to isolate meal_time conditions, (2) splitting the UNION ALL into three specialized CTEs that join with date_dim and filtered_time early, and (3) converting implicit joins to explicit JOIN syntax. This allows each channel to apply filters and joins independently before unioning.

**Expected speedup**: ~1.5-2x due to early reduction of sales rows via time_dim join and better join-order flexibility.
