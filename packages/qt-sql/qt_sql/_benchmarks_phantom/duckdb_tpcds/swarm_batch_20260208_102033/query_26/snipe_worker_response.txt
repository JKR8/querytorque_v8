```sql
WITH filtered_dates AS (
    SELECT d_date_sk
    FROM date_dim
    WHERE d_year = 2001
),
filtered_customer_demographics AS (
    SELECT cd_demo_sk
    FROM customer_demographics
    WHERE cd_gender = 'M'
        AND cd_marital_status = 'S'
        AND cd_education_status = 'Unknown'
),
filtered_promotions AS (
    SELECT DISTINCT p_promo_sk
    FROM promotion
    WHERE p_channel_email = 'N' OR p_channel_event = 'N'
),
filtered_catalog_sales AS (
    SELECT 
        cs_item_sk,
        cs_quantity,
        cs_list_price,
        cs_coupon_amt,
        cs_sales_price
    FROM catalog_sales cs
    INNER JOIN filtered_dates fd ON cs.cs_sold_date_sk = fd.d_date_sk
    INNER JOIN filtered_customer_demographics fcd ON cs.cs_bill_cdemo_sk = fcd.cd_demo_sk
    INNER JOIN filtered_promotions fp ON cs.cs_promo_sk = fp.p_promo_sk
)
SELECT
    i.i_item_id,
    AVG(fcs.cs_quantity) AS agg1,
    AVG(fcs.cs_list_price) AS agg2,
    AVG(fcs.cs_coupon_amt) AS agg3,
    AVG(fcs.cs_sales_price) AS agg4
FROM filtered_catalog_sales fcs
INNER JOIN item i ON fcs.cs_item_sk = i.i_item_sk
GROUP BY i.i_item_id
ORDER BY i.i_item_id
LIMIT 100
```

```
Changes: Applied dimension_cte_isolate pattern with prefetch_fact_join optimization. Pre-filtered dimension tables into separate CTEs (date, customer_demographics, promotion) and pre-joined them with the fact table in a dedicated CTE before final aggregation with the item table. This creates small hash tables from filtered dimensions for faster fact table probing and reduces data volume early in the pipeline.
Expected speedup: 2.2x - 2.8x
```
