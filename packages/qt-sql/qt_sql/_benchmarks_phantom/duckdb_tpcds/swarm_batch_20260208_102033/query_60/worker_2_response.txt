```sql
WITH filtered_items AS (
  SELECT
    i_item_id,
    i_item_sk
  FROM item
  WHERE i_category IN ('Children')
),
filtered_dates AS (
  SELECT d_date_sk
  FROM date_dim
  WHERE d_year = 2000
    AND d_moy = 8
),
filtered_addresses AS (
  SELECT ca_address_sk
  FROM customer_address
  WHERE ca_gmt_offset = -7
),
ss AS (
  SELECT
    i.i_item_id,
    SUM(ss_ext_sales_price) AS total_sales
  FROM store_sales
  JOIN filtered_dates d ON ss_sold_date_sk = d.d_date_sk
  JOIN filtered_addresses ca ON ss_addr_sk = ca.ca_address_sk
  JOIN filtered_items i ON ss_item_sk = i.i_item_sk
  GROUP BY i.i_item_id
),
cs AS (
  SELECT
    i.i_item_id,
    SUM(cs_ext_sales_price) AS total_sales
  FROM catalog_sales
  JOIN filtered_dates d ON cs_sold_date_sk = d.d_date_sk
  JOIN filtered_addresses ca ON cs_bill_addr_sk = ca.ca_address_sk
  JOIN filtered_items i ON cs_item_sk = i.i_item_sk
  GROUP BY i.i_item_id
),
ws AS (
  SELECT
    i.i_item_id,
    SUM(ws_ext_sales_price) AS total_sales
  FROM web_sales
  JOIN filtered_dates d ON ws_sold_date_sk = d.d_date_sk
  JOIN filtered_addresses ca ON ws_bill_addr_sk = ca.ca_address_sk
  JOIN filtered_items i ON ws_item_sk = i.i_item_sk
  GROUP BY i.i_item_id
)
SELECT
  i_item_id,
  SUM(total_sales) AS total_sales
FROM (
  SELECT * FROM ss
  UNION ALL
  SELECT * FROM cs
  UNION ALL
  SELECT * FROM ws
) tmp1
GROUP BY i_item_id
ORDER BY i_item_id, total_sales
LIMIT 100
```

Changes: Isolated filtered dimension tables (date, address, item) into separate CTEs before joining with fact tables, converting implicit joins to explicit joins and removing redundant subqueries.
Expected speedup: 1.5-2x due to reduced dimension table scans and smaller hash tables for joins.
