```sql
WITH target_month AS (
    SELECT DISTINCT d_month_seq
    FROM date_dim
    WHERE d_year = 2002 AND d_moy = 3
),
category_avg_price AS (
    SELECT 
        i_category,
        AVG(i_current_price) AS avg_category_price
    FROM item
    GROUP BY i_category
),
filtered_items AS (
    SELECT 
        i_item_sk,
        i_category,
        i_current_price
    FROM item
    INNER JOIN category_avg_price cap 
        ON item.i_category = cap.i_category
    WHERE i_current_price > 1.2 * cap.avg_category_price
),
filtered_dates AS (
    SELECT 
        d_date_sk
    FROM date_dim
    INNER JOIN target_month ON date_dim.d_month_seq = target_month.d_month_seq
),
filtered_sales AS (
    SELECT 
        ss_customer_sk,
        ss_item_sk
    FROM store_sales
    INNER JOIN filtered_dates ON store_sales.ss_sold_date_sk = filtered_dates.d_date_sk
    INNER JOIN filtered_items ON store_sales.ss_item_sk = filtered_items.i_item_sk
)
SELECT
    a.ca_state AS state,
    COUNT(*) AS cnt
FROM filtered_sales s
INNER JOIN customer c ON s.ss_customer_sk = c.c_customer_sk
INNER JOIN customer_address a ON c.c_current_addr_sk = a.ca_address_sk
GROUP BY a.ca_state
HAVING COUNT(*) >= 10
ORDER BY cnt, a.ca_state
LIMIT 100
```

**Changes**: Created a staged CTE pipeline that first pre-filters date_dim to the target month sequence, pre-computes item category averages in a single pass, filters items based on the pre-computed averages, then joins the reduced fact table with remaining dimensions.

**Expected speedup**: 2.5-3.5x due to early reduction of the large fact table join, elimination of correlated subquery, and efficient single-pass category average computation.
