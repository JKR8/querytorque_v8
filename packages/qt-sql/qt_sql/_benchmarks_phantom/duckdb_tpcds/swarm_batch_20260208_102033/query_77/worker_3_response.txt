```sql
WITH filtered_date AS (
  SELECT d_date_sk
  FROM date_dim
  WHERE d_date BETWEEN CAST('1998-08-05' AS DATE) 
    AND (CAST('1998-08-05' AS DATE) + INTERVAL '30' DAY)
),
ss AS (
  SELECT
    s_store_sk,
    SUM(ss_ext_sales_price) AS sales,
    SUM(ss_net_profit) AS profit
  FROM store_sales
  INNER JOIN filtered_date ON ss_sold_date_sk = d_date_sk
  INNER JOIN store ON ss_store_sk = s_store_sk
  GROUP BY s_store_sk
),
sr AS (
  SELECT
    s_store_sk,
    SUM(sr_return_amt) AS "returns",
    SUM(sr_net_loss) AS profit_loss
  FROM store_returns
  INNER JOIN filtered_date ON sr_returned_date_sk = d_date_sk
  INNER JOIN store ON sr_store_sk = s_store_sk
  GROUP BY s_store_sk
),
cs AS (
  SELECT
    cs_call_center_sk,
    SUM(cs_ext_sales_price) AS sales,
    SUM(cs_net_profit) AS profit
  FROM catalog_sales
  INNER JOIN filtered_date ON cs_sold_date_sk = d_date_sk
  GROUP BY cs_call_center_sk
),
cr AS (
  SELECT
    cr_call_center_sk,
    SUM(cr_return_amount) AS "returns",
    SUM(cr_net_loss) AS profit_loss
  FROM catalog_returns
  INNER JOIN filtered_date ON cr_returned_date_sk = d_date_sk
  GROUP BY cr_call_center_sk
),
ws AS (
  SELECT
    wp_web_page_sk,
    SUM(ws_ext_sales_price) AS sales,
    SUM(ws_net_profit) AS profit
  FROM web_sales
  INNER JOIN filtered_date ON ws_sold_date_sk = d_date_sk
  INNER JOIN web_page ON ws_web_page_sk = wp_web_page_sk
  GROUP BY wp_web_page_sk
),
wr AS (
  SELECT
    wp_web_page_sk,
    SUM(wr_return_amt) AS "returns",
    SUM(wr_net_loss) AS profit_loss
  FROM web_returns
  INNER JOIN filtered_date ON wr_returned_date_sk = d_date_sk
  INNER JOIN web_page ON wr_web_page_sk = wp_web_page_sk
  GROUP BY wp_web_page_sk
)
SELECT
  channel,
  id,
  SUM(sales) AS sales,
  SUM("returns") AS "returns",
  SUM(profit) AS profit
FROM (
  SELECT
    'store channel' AS channel,
    ss.s_store_sk AS id,
    sales,
    COALESCE("returns", 0) AS "returns",
    (profit - COALESCE(profit_loss, 0)) AS profit
  FROM ss
  LEFT JOIN sr ON ss.s_store_sk = sr.s_store_sk
  UNION ALL
  SELECT
    'catalog channel' AS channel,
    cs_call_center_sk AS id,
    sales,
    "returns",
    (profit - profit_loss) AS profit
  FROM cs, cr
  UNION ALL
  SELECT
    'web channel' AS channel,
    ws.wp_web_page_sk AS id,
    sales,
    COALESCE("returns", 0) AS "returns",
    (profit - COALESCE(profit_loss, 0)) AS profit
  FROM ws
  LEFT JOIN wr ON ws.wp_web_page_sk = wr.wp_web_page_sk
) AS x
GROUP BY ROLLUP(channel, id)
ORDER BY channel, id
LIMIT 100
```

**Changes**: Created a shared `filtered_date` CTE to prefetch date keys once, then used it in all fact table joins. This consolidates the repeated date filter across six CTEs into a single scan.

**Expected speedup**: 1.5-2x from eliminating redundant date_dim scans and enabling better join optimization.
