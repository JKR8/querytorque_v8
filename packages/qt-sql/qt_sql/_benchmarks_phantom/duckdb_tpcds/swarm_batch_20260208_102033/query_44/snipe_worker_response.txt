```sql
WITH store_avg AS (
    SELECT AVG(ss_net_profit) AS avg_profit
    FROM store_sales
    WHERE ss_store_sk = 146 AND ss_addr_sk IS NULL
    GROUP BY ss_store_sk
),
item_stats AS (
    SELECT ss_item_sk AS item_sk,
           AVG(ss_net_profit) AS avg_net_profit
    FROM store_sales
    WHERE ss_store_sk = 146
    GROUP BY ss_item_sk
    HAVING AVG(ss_net_profit) > 0.9 * (SELECT avg_profit FROM store_avg)
),
ranked_items AS (
    SELECT item_sk,
           avg_net_profit,
           RANK() OVER (ORDER BY avg_net_profit ASC) AS rnk_asc,
           RANK() OVER (ORDER BY avg_net_profit DESC) AS rnk_desc
    FROM item_stats
)
SELECT ri1.rnk_asc AS rnk,
       i1.i_product_name AS best_performing,
       i2.i_product_name AS worst_performing
FROM ranked_items ri1
JOIN ranked_items ri2 ON ri1.rnk_asc = ri2.rnk_desc
JOIN item i1 ON ri1.item_sk = i1.i_item_sk
JOIN item i2 ON ri2.item_sk = i2.i_item_sk
WHERE ri1.rnk_asc < 11
  AND ri2.rnk_desc < 11
ORDER BY ri1.rnk_asc
LIMIT 100
```

Changes: Consolidated duplicate aggregations and subqueries into single-pass CTEs, computing both ascending and descending ranks in one pass over filtered item averages. Extracted the repeated scalar subquery into a materialized CTE.
