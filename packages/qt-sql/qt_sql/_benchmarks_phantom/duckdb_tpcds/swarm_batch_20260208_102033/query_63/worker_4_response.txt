```sql
WITH 
-- Pre-filter date dimension for the 12 month sequences
filtered_dates AS (
    SELECT d_date_sk, d_moy
    FROM date_dim
    WHERE d_month_seq IN (1181, 1182, 1183, 1184, 1185, 1186, 1187, 1188, 1189, 1190, 1191, 1192)
),
-- First OR branch: Books/Children/Electronics with specific classes/brands
branch1_items AS (
    SELECT i_item_sk, i_manager_id
    FROM item
    WHERE i_category IN ('Books', 'Children', 'Electronics')
      AND i_class IN ('personal', 'portable', 'reference', 'self-help')
      AND i_brand IN ('scholaramalgamalg #14', 'scholaramalgamalg #7', 'exportiunivamalg #9', 'scholaramalgamalg #9')
),
-- Second OR branch: Women/Music/Men with specific classes/brands  
branch2_items AS (
    SELECT i_item_sk, i_manager_id
    FROM item
    WHERE i_category IN ('Women', 'Music', 'Men')
      AND i_class IN ('accessories', 'classical', 'fragrances', 'pants')
      AND i_brand IN ('amalgimporto #1', 'edu packscholar #1', 'exportiimporto #1', 'importoamalg #1')
),
-- Union of both branches with explicit JOINs
unioned_sales AS (
    -- Branch 1 sales
    SELECT 
        b1.i_manager_id,
        fd.d_moy,
        ss.ss_sales_price
    FROM store_sales ss
    JOIN branch1_items b1 ON ss.ss_item_sk = b1.i_item_sk
    JOIN filtered_dates fd ON ss.ss_sold_date_sk = fd.d_date_sk
    JOIN store s ON ss.ss_store_sk = s.s_store_sk
    
    UNION ALL
    
    -- Branch 2 sales  
    SELECT 
        b2.i_manager_id,
        fd.d_moy,
        ss.ss_sales_price
    FROM store_sales ss
    JOIN branch2_items b2 ON ss.ss_item_sk = b2.i_item_sk
    JOIN filtered_dates fd ON ss.ss_sold_date_sk = fd.d_date_sk
    JOIN store s ON ss.ss_store_sk = s.s_store_sk
),
-- Aggregate by manager and month
monthly_aggregates AS (
    SELECT
        i_manager_id,
        d_moy,
        SUM(ss_sales_price) AS sum_sales
    FROM unioned_sales
    GROUP BY i_manager_id, d_moy
),
-- Compute window function
windowed_results AS (
    SELECT
        i_manager_id,
        sum_sales,
        AVG(sum_sales) OVER (PARTITION BY i_manager_id) AS avg_monthly_sales
    FROM monthly_aggregates
)
-- Final filtering and ordering
SELECT *
FROM windowed_results
WHERE
    CASE
        WHEN avg_monthly_sales > 0
        THEN ABS(sum_sales - avg_monthly_sales) / avg_monthly_sales
        ELSE NULL
    END > 0.1
ORDER BY
    i_manager_id,
    avg_monthly_sales,
    sum_sales
LIMIT 100
```

**Changes**: Split the complex OR condition into separate CTEs for each branch with UNION ALL, converted implicit joins to explicit JOIN syntax, pre-filtered date dimension, and separated aggregation from window computation.

**Expected speedup**: 2.5-3x due to better join planning, parallel branch execution, and reduced predicate complexity in the main scan.
