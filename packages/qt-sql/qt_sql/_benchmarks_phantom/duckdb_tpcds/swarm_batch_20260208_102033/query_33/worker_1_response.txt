```sql
WITH filtered_items AS (
  SELECT
    i_item_sk,
    i_manufact_id
  FROM item
  WHERE i_category IN ('Home')
),
filtered_date AS (
  SELECT d_date_sk
  FROM date_dim
  WHERE d_year = 2002
    AND d_moy = 1
),
filtered_address AS (
  SELECT ca_address_sk
  FROM customer_address
  WHERE ca_gmt_offset = -5
),
ss AS (
  SELECT
    i.i_manufact_id,
    SUM(ss_ext_sales_price) AS total_sales
  FROM store_sales
  JOIN filtered_items i ON ss_item_sk = i.i_item_sk
  JOIN filtered_date d ON ss_sold_date_sk = d.d_date_sk
  JOIN filtered_address a ON ss_addr_sk = a.ca_address_sk
  GROUP BY i.i_manufact_id
),
cs AS (
  SELECT
    i.i_manufact_id,
    SUM(cs_ext_sales_price) AS total_sales
  FROM catalog_sales
  JOIN filtered_items i ON cs_item_sk = i.i_item_sk
  JOIN filtered_date d ON cs_sold_date_sk = d.d_date_sk
  JOIN filtered_address a ON cs_bill_addr_sk = a.ca_address_sk
  GROUP BY i.i_manufact_id
),
ws AS (
  SELECT
    i.i_manufact_id,
    SUM(ws_ext_sales_price) AS total_sales
  FROM web_sales
  JOIN filtered_items i ON ws_item_sk = i.i_item_sk
  JOIN filtered_date d ON ws_sold_date_sk = d.d_date_sk
  JOIN filtered_address a ON ws_bill_addr_sk = a.ca_address_sk
  GROUP BY i.i_manufact_id
)
SELECT
  i_manufact_id,
  SUM(total_sales) AS total_sales
FROM (
  SELECT * FROM ss
  UNION ALL
  SELECT * FROM cs
  UNION ALL
  SELECT * FROM ws
) AS tmp1
GROUP BY i_manufact_id
ORDER BY total_sales
LIMIT 100
```

**Changes**: Materialized the repeated item subquery and pre-filtered dimension tables (date_dim, customer_address) into shared CTEs, then joined these filtered CTEs with each sales fact table. This pushes filters early and avoids redundant computation across all three sales channels.

**Expected speedup**: 1.8-2.2x due to reduced scanning of large fact tables by applying dimension filters first and computing the item filter once.
