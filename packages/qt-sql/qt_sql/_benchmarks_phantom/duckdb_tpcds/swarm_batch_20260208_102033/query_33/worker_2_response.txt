WITH filtered_date AS (
    SELECT d_date_sk
    FROM date_dim
    WHERE d_year = 2002
      AND d_moy = 1
),
filtered_item AS (
    SELECT i_item_sk, i_manufact_id
    FROM item
    WHERE i_category IN ('Home')
),
filtered_address AS (
    SELECT ca_address_sk
    FROM customer_address
    WHERE ca_gmt_offset = -5
),
ss AS (
    SELECT
        fi.i_manufact_id,
        SUM(ss_ext_sales_price) AS total_sales
    FROM store_sales
    JOIN filtered_item fi ON ss_item_sk = fi.i_item_sk
    JOIN filtered_date fd ON ss_sold_date_sk = fd.d_date_sk
    JOIN filtered_address fa ON ss_addr_sk = fa.ca_address_sk
    GROUP BY fi.i_manufact_id
),
cs AS (
    SELECT
        fi.i_manufact_id,
        SUM(cs_ext_sales_price) AS total_sales
    FROM catalog_sales
    JOIN filtered_item fi ON cs_item_sk = fi.i_item_sk
    JOIN filtered_date fd ON cs_sold_date_sk = fd.d_date_sk
    JOIN filtered_address fa ON cs_bill_addr_sk = fa.ca_address_sk
    GROUP BY fi.i_manufact_id
),
ws AS (
    SELECT
        fi.i_manufact_id,
        SUM(ws_ext_sales_price) AS total_sales
    FROM web_sales
    JOIN filtered_item fi ON ws_item_sk = fi.i_item_sk
    JOIN filtered_date fd ON ws_sold_date_sk = fd.d_date_sk
    JOIN filtered_address fa ON ws_bill_addr_sk = fa.ca_address_sk
    GROUP BY fi.i_manufact_id
)
SELECT
    i_manufact_id,
    SUM(total_sales) AS total_sales
FROM (
    SELECT * FROM ss
    UNION ALL
    SELECT * FROM cs
    UNION ALL
    SELECT * FROM ws
) AS tmp1
GROUP BY i_manufact_id
ORDER BY total_sales
LIMIT 100;

Changes: Created three filtered dimension CTEs (date, item, address) to materialize small filtered subsets once, then reused them across all three sales channel CTEs. This eliminates repeated scanning and filtering of the same dimension tables and allows efficient hash joins with the large fact tables.
Expected speedup: 1.3x-2x via dimension isolation and shared reuse across channels.
