```sql
WITH filtered_dates AS (
    SELECT d_date_sk
    FROM date_dim
    WHERE d_year = 2001
      AND d_qoy < 4
)
SELECT
    ca_state,
    cd_gender,
    cd_marital_status,
    cd_dep_count,
    COUNT(*) AS cnt1,
    MAX(cd_dep_count),
    SUM(cd_dep_count),
    MAX(cd_dep_count),
    cd_dep_employed_count,
    COUNT(*) AS cnt2,
    MAX(cd_dep_employed_count),
    SUM(cd_dep_employed_count),
    MAX(cd_dep_employed_count),
    cd_dep_college_count,
    COUNT(*) AS cnt3,
    MAX(cd_dep_college_count),
    SUM(cd_dep_college_count),
    MAX(cd_dep_college_count)
FROM customer AS c
JOIN customer_address AS ca ON c.c_current_addr_sk = ca.ca_address_sk
JOIN customer_demographics ON cd_demo_sk = c.c_current_cdemo_sk
WHERE EXISTS(
    SELECT 1
    FROM store_sales
    JOIN filtered_dates ON store_sales.ss_sold_date_sk = filtered_dates.d_date_sk
    WHERE c.c_customer_sk = store_sales.ss_customer_sk
)
AND (
    EXISTS(
        SELECT 1
        FROM web_sales
        JOIN filtered_dates ON web_sales.ws_sold_date_sk = filtered_dates.d_date_sk
        WHERE c.c_customer_sk = web_sales.ws_bill_customer_sk
    )
    OR EXISTS(
        SELECT 1
        FROM catalog_sales
        JOIN filtered_dates ON catalog_sales.cs_sold_date_sk = filtered_dates.d_date_sk
        WHERE c.c_customer_sk = catalog_sales.cs_ship_customer_sk
    )
)
GROUP BY
    ca_state,
    cd_gender,
    cd_marital_status,
    cd_dep_count,
    cd_dep_employed_count,
    cd_dep_college_count
ORDER BY
    ca_state,
    cd_gender,
    cd_marital_status,
    cd_dep_count,
    cd_dep_employed_count,
    cd_dep_college_count
LIMIT 100;
```

Changes: Created a CTE `filtered_dates` that isolates the date dimension filtering (year=2001, qoy<4) once, then reused this CTE in all three EXISTS subqueries instead of repeatedly scanning and filtering the date_dim table. Also converted implicit joins to explicit JOIN syntax for clarity and optimizer benefits.

Expected speedup: 2-3x due to eliminating redundant date dimension scans and allowing the small filtered dates set to build efficient hash tables for probing.
