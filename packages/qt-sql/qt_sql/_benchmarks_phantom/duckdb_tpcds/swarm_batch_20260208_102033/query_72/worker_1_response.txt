```sql
WITH filtered_household_demo AS (
    SELECT hd_demo_sk
    FROM household_demographics
    WHERE hd_buy_potential = '501-1000'
), filtered_customer_demo AS (
    SELECT cd_demo_sk
    FROM customer_demographics
    WHERE cd_marital_status = 'W'
), filtered_date_d1 AS (
    SELECT d_date_sk, d_date, d_week_seq
    FROM date_dim
    WHERE d_year = 2002
)
SELECT
    i_item_desc,
    w_warehouse_name,
    d1.d_week_seq,
    SUM(CASE WHEN p_promo_sk IS NULL THEN 1 ELSE 0 END) AS no_promo,
    SUM(CASE WHEN NOT p_promo_sk IS NULL THEN 1 ELSE 0 END) AS promo,
    COUNT(*) AS total_cnt
FROM catalog_sales cs
INNER JOIN filtered_date_d1 d1 ON cs.cs_sold_date_sk = d1.d_date_sk
INNER JOIN filtered_household_demo hd ON cs.cs_bill_hdemo_sk = hd.hd_demo_sk
INNER JOIN filtered_customer_demo cd ON cs.cs_bill_cdemo_sk = cd.cd_demo_sk
INNER JOIN inventory inv ON (
    cs.cs_item_sk = inv.inv_item_sk
    AND inv.inv_quantity_on_hand < cs.cs_quantity
)
INNER JOIN date_dim d2 ON (
    inv.inv_date_sk = d2.d_date_sk
    AND d1.d_week_seq = d2.d_week_seq
)
INNER JOIN date_dim d3 ON (
    cs.cs_ship_date_sk = d3.d_date_sk
    AND d3.d_date > d1.d_date + 5
)
INNER JOIN warehouse w ON w.w_warehouse_sk = inv.inv_warehouse_sk
INNER JOIN item i ON i.i_item_sk = cs.cs_item_sk
LEFT OUTER JOIN promotion p ON cs.cs_promo_sk = p.p_promo_sk
LEFT OUTER JOIN catalog_returns cr ON (
    cr.cr_item_sk = cs.cs_item_sk
    AND cr.cr_order_number = cs.cs_order_number
)
GROUP BY
    i_item_desc,
    w_warehouse_name,
    d1.d_week_seq
ORDER BY
    total_cnt DESC,
    i_item_desc,
    w_warehouse_name,
    d1.d_week_seq
LIMIT 100
```

**Changes**: Created filtered CTEs for small dimension tables (household_demographics, customer_demographics, date_dim d1) to push down filters before joining with large fact tables. Also moved filter conditions into explicit JOIN ON clauses where possible to help DuckDB optimize join order.

**Expected speedup**: 1.3-1.7x by reducing intermediate data volume through early filtering of dimension tables and enabling better join order optimization.
