```sql
WITH
-- Early filter on dimension tables to reduce join size
filtered_date AS (
    SELECT d_date_sk
    FROM date_dim
    WHERE d_dom BETWEEN 1 AND 2
      AND d_year IN (1998, 1999, 2000)
),
filtered_store AS (
    SELECT s_store_sk, s_city
    FROM store
    WHERE s_city IN ('Pleasant Hill', 'Five Points')
),
filtered_hd AS (
    SELECT hd_demo_sk
    FROM household_demographics
    WHERE hd_dep_count = 8 OR hd_vehicle_count = -1
),
filtered_ca AS (
    SELECT ca_address_sk, ca_city
    FROM customer_address
),
-- Pre-aggregated fact data with early filtering
dn AS (
    SELECT
        ss_ticket_number,
        ss_customer_sk,
        filtered_ca.ca_city AS bought_city,
        SUM(ss_ext_sales_price) AS extended_price,
        SUM(ss_ext_list_price) AS list_price,
        SUM(ss_ext_tax) AS extended_tax
    FROM store_sales
    INNER JOIN filtered_date ON store_sales.ss_sold_date_sk = filtered_date.d_date_sk
    INNER JOIN filtered_store ON store_sales.ss_store_sk = filtered_store.s_store_sk
    INNER JOIN filtered_hd ON store_sales.ss_hdemo_sk = filtered_hd.hd_demo_sk
    INNER JOIN filtered_ca ON store_sales.ss_addr_sk = filtered_ca.ca_address_sk
    GROUP BY ss_ticket_number, ss_customer_sk, store_sales.ss_addr_sk, filtered_ca.ca_city
)
SELECT
    c_last_name,
    c_first_name,
    current_addr.ca_city,
    dn.bought_city,
    dn.ss_ticket_number,
    dn.extended_price,
    dn.extended_tax,
    dn.list_price
FROM dn
JOIN customer ON dn.ss_customer_sk = customer.c_customer_sk
JOIN customer_address AS current_addr ON customer.c_current_addr_sk = current_addr.ca_address_sk
WHERE current_addr.ca_city <> dn.bought_city
ORDER BY c_last_name, dn.ss_ticket_number
LIMIT 100;
```

**Changes**: Created early-filtering CTEs for dimension tables (date_dim, store, household_demographics, customer_address) to reduce fact table joins, converted implicit joins to explicit JOIN syntax for better optimizer control, and maintained all original grouping and filtering logic.

**Expected speedup**: 1.5-2x by reducing the fact table join size through early dimension filtering and enabling better join order optimization.
