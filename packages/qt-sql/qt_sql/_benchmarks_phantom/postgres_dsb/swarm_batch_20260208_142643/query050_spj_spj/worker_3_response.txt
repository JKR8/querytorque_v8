```sql
WITH filtered_store AS (
    SELECT s_store_sk, s_store_name, s_company_id, s_street_number,
           s_street_name, s_suite_number, s_city, s_zip
    FROM store
    WHERE s_state IN ('LA', 'TX', 'VA')
),
filtered_d2 AS (
    SELECT d_date_sk, d_date
    FROM date_dim
    WHERE d_moy = 12
),
filtered_d1 AS (
    SELECT d_date_sk, d_date
    FROM date_dim
    WHERE d_dow = 1
),
joined_dates AS (
    SELECT d1.d_date_sk AS d1_date_sk,
           d2.d_date_sk AS d2_date_sk,
           d2.d_date AS d2_date
    FROM filtered_d1 d1
    INNER JOIN filtered_d2 d2 
        ON d1.d_date BETWEEN (d2.d_date - INTERVAL '120 DAY') AND d2.d_date
),
filtered_sales AS (
    SELECT ss_ticket_number, ss_item_sk, ss_customer_sk,
           ss_sold_date_sk, ss_store_sk
    FROM store_sales
    WHERE ss_sold_date_sk IN (SELECT d1_date_sk FROM joined_dates)
),
filtered_returns AS (
    SELECT sr_ticket_number, sr_item_sk, sr_customer_sk,
           sr_returned_date_sk, sr_store_sk
    FROM store_returns
    WHERE sr_returned_date_sk IN (SELECT d2_date_sk FROM joined_dates)
)
SELECT MIN(s_store_name),
       MIN(s_company_id),
       MIN(s_street_number),
       MIN(s_street_name),
       MIN(s_suite_number),
       MIN(s_city),
       MIN(s_zip),
       MIN(ss_ticket_number),
       MIN(ss_sold_date_sk),
       MIN(sr_returned_date_sk),
       MIN(ss_item_sk),
       MIN(d1.d_date_sk)
FROM filtered_sales
INNER JOIN filtered_returns 
    ON ss_ticket_number = sr_ticket_number
   AND ss_item_sk = sr_item_sk
   AND ss_customer_sk = sr_customer_sk
INNER JOIN filtered_store s 
    ON ss_store_sk = s.s_store_sk 
   AND sr_store_sk = s.s_store_sk
INNER JOIN joined_dates 
    ON ss_sold_date_sk = d1_date_sk
   AND sr_returned_date_sk = d2_date_sk
INNER JOIN filtered_d1 d1 
    ON ss_sold_date_sk = d1.d_date_sk;
```

**Changes**: Materialized all dimension filters (store by state, date_dim by month/day-of-week) into separate CTEs, created a CTE for the date range join, and pushed date filters into fact table CTEs using semi-joins before the main join. This reduces the working set before the complex multi-table join.

**Expected speedup**: 2-3x from eliminating redundant full table scans and enabling hash joins on pre-filtered datasets.
