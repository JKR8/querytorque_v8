```sql
WITH filtered_dates AS (
    SELECT d_date_sk
    FROM date_dim
    WHERE d_year = 2002
      AND d_moy BETWEEN 3 AND 5
),
filtered_customer_address AS (
    SELECT ca_address_sk
    FROM customer_address
    WHERE ca_state IN ('IA', 'MO', 'TX')
),
filtered_customer_demographics AS (
    SELECT cd_demo_sk, cd_gender, cd_marital_status, 
           cd_education_status, cd_purchase_estimate, cd_credit_rating
    FROM customer_demographics
    WHERE cd_marital_status IN ('S', 'S', 'S')
      AND cd_education_status IN ('Primary', 'Secondary')
)
SELECT
    cd_gender,
    cd_marital_status,
    cd_education_status,
    COUNT(*) AS cnt1,
    cd_purchase_estimate,
    COUNT(*) AS cnt2,
    cd_credit_rating,
    COUNT(*) AS cnt3
FROM customer AS c
JOIN filtered_customer_address AS ca ON c.c_current_addr_sk = ca.ca_address_sk
JOIN filtered_customer_demographics AS cd ON cd.cd_demo_sk = c.c_current_cdemo_sk
WHERE EXISTS (
    SELECT 1
    FROM store_sales
    JOIN filtered_dates ON ss_sold_date_sk = filtered_dates.d_date_sk
    WHERE c.c_customer_sk = ss_customer_sk
      AND ss_list_price BETWEEN 100 AND 189
)
AND NOT EXISTS (
    SELECT 1
    FROM web_sales
    JOIN filtered_dates ON ws_sold_date_sk = filtered_dates.d_date_sk
    WHERE c.c_customer_sk = ws_bill_customer_sk
      AND ws_list_price BETWEEN 100 AND 189
)
AND NOT EXISTS (
    SELECT 1
    FROM catalog_sales
    JOIN filtered_dates ON cs_sold_date_sk = filtered_dates.d_date_sk
    WHERE c.c_customer_sk = cs_ship_customer_sk
      AND cs_list_price BETWEEN 100 AND 189
)
GROUP BY
    cd_gender,
    cd_marital_status,
    cd_education_status,
    cd_purchase_estimate,
    cd_credit_rating
ORDER BY
    cd_gender,
    cd_marital_status,
    cd_education_status,
    cd_purchase_estimate,
    cd_credit_rating
LIMIT 100
```

```
Changes: Converted implicit joins to explicit JOIN syntax; created CTEs for filtered date_dim, customer_address, and customer_demographics to push down selective filters early; kept EXISTS/NOT EXISTS subqueries as-is but now join with pre-filtered CTEs.
Expected speedup: 1.5-2x from reduced row volumes in joins and better join planning with explicit syntax.
```
