```sql
WITH filtered_items AS (
  SELECT i_item_sk
  FROM item
  WHERE i_color IN ('blanched', 'rosy')
    AND i_manager_id BETWEEN 16 AND 35
),
filtered_addresses AS (
  SELECT ca_address_sk, ca_county
  FROM customer_address
  WHERE ca_state IN ('TX', 'VA')
),
ss_base AS (
  SELECT
    ca.ca_county,
    d.d_qoy,
    d.d_year,
    SUM(ss.ss_ext_sales_price) AS store_sales
  FROM store_sales ss
  JOIN date_dim d ON ss.ss_sold_date_sk = d.d_date_sk
  JOIN filtered_addresses ca ON ss.ss_addr_sk = ca.ca_address_sk
  JOIN filtered_items i ON ss.ss_item_sk = i.i_item_sk
  WHERE ss.ss_list_price BETWEEN 286 AND 300
    AND d.d_year = 1998
    AND d.d_qoy IN (1, 2, 3)
  GROUP BY ca.ca_county, d.d_qoy, d.d_year
),
ws_base AS (
  SELECT
    ca.ca_county,
    d.d_qoy,
    d.d_year,
    SUM(ws.ws_ext_sales_price) AS web_sales
  FROM web_sales ws
  JOIN date_dim d ON ws.ws_sold_date_sk = d.d_date_sk
  JOIN filtered_addresses ca ON ws.ws_bill_addr_sk = ca.ca_address_sk
  JOIN filtered_items i ON ws.ws_item_sk = i.i_item_sk
  WHERE ws.ws_list_price BETWEEN 286 AND 300
    AND d.d_year = 1998
    AND d.d_qoy IN (1, 2, 3)
  GROUP BY ca.ca_county, d.d_qoy, d.d_year
),
ss_pivot AS (
  SELECT
    ca_county,
    d_year,
    MAX(CASE WHEN d_qoy = 1 THEN store_sales END) AS store_q1,
    MAX(CASE WHEN d_qoy = 2 THEN store_sales END) AS store_q2,
    MAX(CASE WHEN d_qoy = 3 THEN store_sales END) AS store_q3
  FROM ss_base
  GROUP BY ca_county, d_year
),
ws_pivot AS (
  SELECT
    ca_county,
    d_year,
    MAX(CASE WHEN d_qoy = 1 THEN web_sales END) AS web_q1,
    MAX(CASE WHEN d_qoy = 2 THEN web_sales END) AS web_q2,
    MAX(CASE WHEN d_qoy = 3 THEN web_sales END) AS web_q3
  FROM ws_base
  GROUP BY ca_county, d_year
)
SELECT
  sp.ca_county,
  sp.d_year,
  CASE WHEN w.web_q1 > 0 THEN w.web_q2 / w.web_q1 ELSE NULL END AS web_q1_q2_increase,
  CASE WHEN sp.store_q1 > 0 THEN sp.store_q2 / sp.store_q1 ELSE NULL END AS store_q1_q2_increase,
  CASE WHEN w.web_q2 > 0 THEN w.web_q3 / w.web_q2 ELSE NULL END AS web_q2_q3_increase,
  CASE WHEN sp.store_q2 > 0 THEN sp.store_q3 / sp.store_q2 ELSE NULL END AS store_q2_q3_increase
FROM ss_pivot sp
JOIN ws_pivot w ON sp.ca_county = w.ca_county AND sp.d_year = w.d_year
WHERE
  CASE WHEN w.web_q1 > 0 THEN w.web_q2 / w.web_q1 ELSE NULL END > 
  CASE WHEN sp.store_q1 > 0 THEN sp.store_q2 / sp.store_q1 ELSE NULL END
  AND CASE WHEN w.web_q2 > 0 THEN w.web_q3 / w.web_q2 ELSE NULL END > 
  CASE WHEN sp.store_q2 > 0 THEN sp.store_q3 / sp.store_q2 ELSE NULL END
ORDER BY web_q1_q2_increase
```

**Changes**: Replaced the 6-way self-join with pivot CTEs that aggregate quarterly sales into columns, allowing a single join between store and web results. Added pre-filtering CTEs for dimensions and pushed year/quarter filters early.
