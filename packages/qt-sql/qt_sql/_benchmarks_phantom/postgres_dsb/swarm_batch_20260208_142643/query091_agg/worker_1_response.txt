```sql
WITH filtered_date AS (
    SELECT d_date_sk
    FROM date_dim
    WHERE d_year = 1999
      AND d_moy = 1
),
filtered_customer_demographics AS (
    SELECT cd_demo_sk, cd_marital_status, cd_education_status
    FROM customer_demographics
    WHERE (cd_marital_status = 'M' AND cd_education_status = 'Unknown')
       OR (cd_marital_status = 'W' AND cd_education_status = 'Advanced Degree')
),
filtered_household_demographics AS (
    SELECT hd_demo_sk
    FROM household_demographics
    WHERE hd_buy_potential LIKE '1001-5000%'
),
filtered_customer_address AS (
    SELECT ca_address_sk
    FROM customer_address
    WHERE ca_gmt_offset = -7
)
SELECT
    cc_call_center_id AS Call_Center,
    cc_name AS Call_Center_Name,
    cc_manager AS Manager,
    SUM(cr_net_loss) AS Returns_Loss
FROM catalog_returns
JOIN filtered_date ON cr_returned_date_sk = filtered_date.d_date_sk
JOIN call_center ON cr_call_center_sk = cc_call_center_sk
JOIN customer ON cr_returning_customer_sk = c_customer_sk
JOIN filtered_customer_demographics ON cd_demo_sk = c_current_cdemo_sk
JOIN filtered_household_demographics ON hd_demo_sk = c_current_hdemo_sk
JOIN filtered_customer_address ON ca_address_sk = c_current_addr_sk
GROUP BY
    cc_call_center_id,
    cc_name,
    cc_manager,
    cd_marital_status,
    cd_education_status
ORDER BY
    SUM(cr_net_loss) DESC
```

**Changes:** Created separate CTEs for each filtered dimension table (date_dim, customer_demographics, household_demographics, customer_address) to push selective filters early, then joined these pre-filtered results using explicit JOIN syntax. This reduces intermediate row counts before the expensive catalog_returns join and aggregation.

**Expected speedup:** 2-3x, similar to the pg_dimension_prefetch_star example, by creating tiny hash tables from selective dimension filters and enabling better join planning.
