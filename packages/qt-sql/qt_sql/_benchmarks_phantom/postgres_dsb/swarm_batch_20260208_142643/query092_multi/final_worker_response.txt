```sql
WITH filtered_dates AS (
    SELECT d_date_sk
    FROM date_dim
    WHERE d_date BETWEEN '1998-03-13' AND CAST('1998-03-13' AS DATE) + INTERVAL '90 DAY'
),
filtered_web_sales AS (
    SELECT ws_item_sk, ws_ext_discount_amt, ws_sales_price, ws_list_price, ws_sold_date_sk, ws_wholesale_cost
    FROM web_sales
    JOIN filtered_dates ON ws_sold_date_sk = d_date_sk
    WHERE ws_wholesale_cost BETWEEN 26 AND 46
),
item_avg_discount AS (
    SELECT 
        ws_item_sk,
        1.3 * AVG(ws_ext_discount_amt) AS avg_threshold
    FROM filtered_web_sales
    WHERE ws_sales_price / ws_list_price BETWEEN 34 * 0.01 AND 49 * 0.01
    GROUP BY ws_item_sk
)
SELECT
    SUM(ws_ext_discount_amt) AS "Excess Discount Amount"
FROM filtered_web_sales ws
JOIN item ON ws.ws_item_sk = i_item_sk
LEFT JOIN item_avg_discount iad ON ws.ws_item_sk = iad.ws_item_sk
WHERE (
    i_manufact_id BETWEEN 341 AND 540 
    OR i_category IN ('Home', 'Men', 'Music')
)
AND ws.ws_ext_discount_amt > COALESCE(iad.avg_threshold, 0)
ORDER BY
    SUM(ws_ext_discount_amt)
LIMIT 100;
```

Changes: Decorrelated the subquery by precomputing per-item average discount thresholds in a CTE, created filtered CTEs for date_dim and web_sales with explicit joins, and converted implicit joins to explicit JOIN syntax. Used LEFT JOIN with COALESCE to handle items with no subquery matches (treated as threshold=0).

Expected speedup: 1.5-2x by eliminating correlated subquery execution and enabling better join ordering.
