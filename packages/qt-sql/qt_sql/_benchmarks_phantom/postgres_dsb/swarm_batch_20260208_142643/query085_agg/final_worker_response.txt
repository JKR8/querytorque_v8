```sql
WITH 
-- Stage 1: Filter most selective fact table (web_sales) by combined price/profit ranges
filtered_web_sales AS (
    SELECT 
        ws_item_sk,
        ws_order_number,
        ws_web_page_sk,
        ws_sold_date_sk,
        ws_quantity,
        ws_sales_price,
        ws_net_profit
    FROM web_sales
    WHERE ws_sales_price BETWEEN 50.00 AND 200.00
      AND ws_net_profit BETWEEN 50.00 AND 300.00
),
-- Stage 2: Join to web_returns early to reduce fanout
sales_returns AS (
    SELECT 
        fws.ws_item_sk,
        fws.ws_order_number,
        fws.ws_web_page_sk,
        fws.ws_sold_date_sk,
        fws.ws_quantity,
        fws.ws_sales_price,
        fws.ws_net_profit,
        wr.wr_refunded_cash,
        wr.wr_fee,
        wr.wr_refunded_cdemo_sk,
        wr.wr_returning_cdemo_sk,
        wr.wr_refunded_addr_sk,
        wr.wr_reason_sk
    FROM filtered_web_sales fws
    JOIN web_returns wr ON fws.ws_item_sk = wr.wr_item_sk 
                       AND fws.ws_order_number = wr.wr_order_number
),
-- Stage 3: Preserved-column dimension CTEs (maintain all needed columns)
filtered_date AS (
    SELECT d_date_sk, d_year
    FROM date_dim
    WHERE d_year = 1998
),
filtered_customer_address AS (
    SELECT 
        ca_address_sk,
        ca_country,
        ca_state
    FROM customer_address
    WHERE ca_country = 'United States'
      AND ca_state IN ('IA', 'PA', 'TX', 'GA', 'MO', 'SD', 'LA', 'TX', 'VA')
)
SELECT
    SUBSTRING(r_reason_desc FROM 1 FOR 20),
    AVG(sr.ws_quantity),
    AVG(sr.wr_refunded_cash),
    AVG(sr.wr_fee)
FROM sales_returns sr
JOIN filtered_date fd ON sr.ws_sold_date_sk = fd.d_date_sk
JOIN web_page wp ON sr.ws_web_page_sk = wp.wp_web_page_sk
JOIN customer_demographics cd1 ON sr.wr_refunded_cdemo_sk = cd1.cd_demo_sk
JOIN customer_demographics cd2 ON sr.wr_returning_cdemo_sk = cd2.cd_demo_sk
JOIN filtered_customer_address ca ON sr.wr_refunded_addr_sk = ca.ca_address_sk
JOIN reason r ON sr.wr_reason_sk = r.r_reason_sk
WHERE (
    (
        cd1.cd_marital_status = 'D'
        AND cd1.cd_marital_status = cd2.cd_marital_status
        AND cd1.cd_education_status = 'Secondary'
        AND cd1.cd_education_status = cd2.cd_education_status
        AND sr.ws_sales_price BETWEEN 100.00 AND 150.00
    )
    OR (
        cd1.cd_marital_status = 'M'
        AND cd1.cd_marital_status = cd2.cd_marital_status
        AND cd1.cd_education_status = '4 yr Degree'
        AND cd1.cd_education_status = cd2.cd_education_status
        AND sr.ws_sales_price BETWEEN 50.00 AND 100.00
    )
    OR (
        cd1.cd_marital_status = 'U'
        AND cd1.cd_marital_status = cd2.cd_marital_status
        AND cd1.cd_education_status = 'Unknown'
        AND cd1.cd_education_status = cd2.cd_education_status
        AND sr.ws_sales_price BETWEEN 150.00 AND 200.00
    )
)
AND (
    (
        ca.ca_country = 'United States'
        AND ca.ca_state IN ('IA', 'PA', 'TX')
        AND sr.ws_net_profit BETWEEN 100 AND 200
    )
    OR (
        ca.ca_country = 'United States'
        AND ca.ca_state IN ('GA', 'MO', 'SD')
        AND sr.ws_net_profit BETWEEN 150 AND 300
    )
    OR (
        ca.ca_country = 'United States'
        AND ca.ca_state IN ('LA', 'TX', 'VA')
        AND sr.ws_net_profit BETWEEN 50 AND 250
    )
)
GROUP BY
    r_reason_desc
ORDER BY
    SUBSTRING(r_reason_desc FROM 1 FOR 20),
    AVG(sr.ws_quantity),
    AVG(sr.wr_refunded_cash),
    AVG(sr.wr_fee)
LIMIT 100;
```

Changes: Applied multi-stage filtering starting with web_sales using combined price/profit ranges, joined to web_returns early, created preserved-column CTEs for dimensions with selective filters, and applied complex OR conditions in a single WHERE clause after all joins.

Expected speedup: 2.0-3.0x from reduced fact table scanning and early fanout reduction.
