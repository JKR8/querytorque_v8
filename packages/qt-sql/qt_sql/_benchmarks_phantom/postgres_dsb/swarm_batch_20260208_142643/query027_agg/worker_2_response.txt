```sql
WITH filtered_dates AS (
    SELECT d_date_sk
    FROM date_dim
    WHERE d_year = 2002
),
filtered_customer_demographics AS (
    SELECT cd_demo_sk
    FROM customer_demographics
    WHERE cd_gender = 'M'
      AND cd_marital_status = 'S'
      AND cd_education_status = 'Advanced Degree'
),
filtered_store AS (
    SELECT s_store_sk, s_state
    FROM store
    WHERE s_state = 'VA'
),
filtered_item AS (
    SELECT i_item_sk, i_item_id, i_category
    FROM item
    WHERE i_category = 'Jewelry'
)
SELECT
    i_item_id,
    s_state,
    GROUPING(s_state) AS g_state,
    AVG(ss_quantity) AS agg1,
    AVG(ss_list_price) AS agg2,
    AVG(ss_coupon_amt) AS agg3,
    AVG(ss_sales_price) AS agg4
FROM store_sales
JOIN filtered_dates ON ss_sold_date_sk = d_date_sk
JOIN filtered_item ON ss_item_sk = i_item_sk
JOIN filtered_store ON ss_store_sk = s_store_sk
JOIN filtered_customer_demographics ON ss_cdemo_sk = cd_demo_sk
GROUP BY ROLLUP(i_item_id, s_state)
ORDER BY i_item_id, s_state
LIMIT 100;
```

Changes: Isolated all four selective dimension filters (date_dim, customer_demographics, store, item) into separate CTEs, then used explicit JOINs to combine them with the fact table. This enables PostgreSQL to create small hash tables from the filtered dimensions before joining to the large fact table.

Expected speedup: 2-3x due to better join order control and reduced early join cardinality.
