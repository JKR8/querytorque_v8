WITH filtered_item AS (SELECT i_item_sk, i_item_id FROM item WHERE i_category = 'Jewelry' AND i_manager_id BETWEEN 61 AND 100), preagg_sales AS (SELECT cs_item_sk, cs_warehouse_sk, cs_sold_date_sk, cs_sales_price, cr_refunded_cash FROM catalog_sales LEFT OUTER JOIN catalog_returns ON cs_order_number = cr_order_number AND cs_item_sk = cr_item_sk WHERE cs_wholesale_cost BETWEEN 42 AND 61 AND (cr_reason_sk = 33 OR cr_reason_sk IS NULL)), filtered_date AS (SELECT d_date_sk, d_date FROM date_dim WHERE d_date BETWEEN (CAST('2002-02-19' AS DATE) - INTERVAL '30 DAY') AND (CAST('2002-02-19' AS DATE) + INTERVAL '30 DAY')) SELECT w_state, i_item_id, SUM(CASE WHEN (CAST(d.d_date AS DATE) < CAST('2002-02-19' AS DATE)) THEN ps.cs_sales_price - COALESCE(ps.cr_refunded_cash, 0) ELSE 0 END) AS sales_before, SUM(CASE WHEN (CAST(d.d_date AS DATE) >= CAST('2002-02-19' AS DATE)) THEN ps.cs_sales_price - COALESCE(ps.cr_refunded_cash, 0) ELSE 0 END) AS sales_after FROM preagg_sales AS ps JOIN filtered_item AS i ON ps.cs_item_sk = i.i_item_sk JOIN warehouse AS w ON ps.cs_warehouse_sk = w.w_warehouse_sk JOIN filtered_date AS d ON ps.cs_sold_date_sk = d.d_date_sk GROUP BY w_state, i_item_id ORDER BY w_state, i_item_id LIMIT 100;
