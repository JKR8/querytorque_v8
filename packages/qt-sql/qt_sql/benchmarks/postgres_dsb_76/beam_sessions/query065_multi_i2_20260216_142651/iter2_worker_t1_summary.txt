WITH filtered_dates AS (SELECT d_date_sk FROM date_dim WHERE d_month_seq BETWEEN 1215 AND 1215 + 11), filtered_stores AS (SELECT s_store_sk FROM store WHERE s_state IN ('KS', 'OH', 'SD')), filtered_items AS (SELECT i_item_sk FROM item WHERE i_manager_id BETWEEN 10 AND 14), base_sales AS (SELECT ss_store_sk, ss_item_sk, SUM(ss_sales_price) AS revenue FROM store_sales JOIN filtered_dates ON ss_sold_date_sk = d_date_sk JOIN filtered_stores ON ss_store_sk = s_store_sk JOIN filtered_items ON ss_item_sk = i_item_sk WHERE ss_sales_price / ss_list_price BETWEEN 0.79 AND 0.89 GROUP BY ss_store_sk, ss_item_sk) SELECT s_store_name, i_item_desc, sc.revenue, i_current_price, i_wholesale_cost, i_brand FROM store, item, (SELECT ss_store_sk, AVG(revenue) AS ave FROM base_sales GROUP BY ss_store_sk) AS sb, base_sales AS sc WHERE sb.ss_store_sk = sc.ss_store_sk AND sc.revenue <= 0.1 * sb.ave AND s_store_sk = sc.ss_store_sk AND i_item_sk = sc.ss_item_sk ORDER BY s_store_name, i_item_desc LIMIT 100;
