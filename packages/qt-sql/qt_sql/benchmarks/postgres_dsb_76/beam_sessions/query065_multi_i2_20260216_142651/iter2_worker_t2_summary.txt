WITH filtered_dates AS (SELECT d_date_sk FROM date_dim WHERE d_month_seq BETWEEN 1215 AND 1215 + 11), base_revenue AS (SELECT ss_store_sk, ss_item_sk, SUM(ss_sales_price) AS revenue FROM store_sales JOIN filtered_dates ON ss_sold_date_sk = d_date_sk WHERE ss_sales_price / ss_list_price BETWEEN 0.79 AND 0.89 GROUP BY ss_store_sk, ss_item_sk), avg_revenue_by_store AS (SELECT ss_store_sk, AVG(revenue) AS ave FROM base_revenue GROUP BY ss_store_sk) SELECT s_store_name, i_item_desc, sc.revenue, i_current_price, i_wholesale_cost, i_brand FROM store JOIN item ON TRUE JOIN base_revenue AS sc ON s_store_sk = sc.ss_store_sk AND i_item_sk = sc.ss_item_sk JOIN avg_revenue_by_store AS sb ON sb.ss_store_sk = sc.ss_store_sk WHERE sc.revenue <= 0.1 * sb.ave AND i_manager_id BETWEEN 10 AND 14 AND s_state IN ('KS', 'OH', 'SD') ORDER BY s_store_name, i_item_desc LIMIT 100;
