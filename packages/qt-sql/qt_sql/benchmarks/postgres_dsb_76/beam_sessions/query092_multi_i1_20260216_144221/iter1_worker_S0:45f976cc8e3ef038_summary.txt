WITH filtered_sales AS (SELECT ws_item_sk, ws_ext_discount_amt, ws_sales_price, ws_list_price FROM web_sales INNER JOIN date_dim ON d_date_sk = ws_sold_date_sk WHERE d_date BETWEEN '2002-03-06' AND CAST('2002-03-06' AS DATE) + INTERVAL '90 DAY' AND ws_wholesale_cost BETWEEN 16 AND 36), item_thresholds AS (SELECT ws_item_sk, 1.3 * AVG(ws_ext_discount_amt) AS threshold FROM filtered_sales WHERE ws_sales_price / ws_list_price BETWEEN 35 * 0.01 AND 50 * 0.01 GROUP BY ws_item_sk), qualified_items AS (SELECT fs.ws_ext_discount_amt FROM filtered_sales AS fs INNER JOIN item ON i_item_sk = fs.ws_item_sk LEFT JOIN item_thresholds AS it ON fs.ws_item_sk = it.ws_item_sk WHERE (i_manufact_id BETWEEN 566 AND 765 OR i_category IN ('Children', 'Jewelry', 'Men')) AND fs.ws_ext_discount_amt > COALESCE(it.threshold, 0)) SELECT SUM(ws_ext_discount_amt) AS "Excess Discount Amount" FROM qualified_items WHERE TRUE ORDER BY SUM(ws_ext_discount_amt) LIMIT 100;
