WITH filtered_date_dim AS (SELECT * FROM date_dim WHERE d_year BETWEEN 1998 AND 1999), filtered_item1 AS (SELECT * FROM item WHERE i_category IN ('Electronics', 'Jewelry')), filtered_item2 AS (SELECT * FROM item WHERE i_manager_id BETWEEN 81 AND 100) SELECT item1.i_item_sk, item2.i_item_sk, COUNT(*) AS cnt FROM filtered_item1 AS item1 JOIN store_sales AS s1 ON item1.i_item_sk = s1.ss_item_sk JOIN filtered_date_dim AS date_dim ON s1.ss_sold_date_sk = date_dim.d_date_sk JOIN customer ON s1.ss_customer_sk = customer.c_customer_sk JOIN customer_address ON customer.c_current_addr_sk = customer_address.ca_address_sk JOIN customer_demographics ON customer.c_current_cdemo_sk = customer_demographics.cd_demo_sk JOIN store_sales AS s2 ON s1.ss_ticket_number = s2.ss_ticket_number JOIN filtered_item2 AS item2 ON s2.ss_item_sk = item2.i_item_sk WHERE item1.i_item_sk < item2.i_item_sk AND cd_marital_status = 'S' AND cd_education_status = 'College' AND s1.ss_list_price BETWEEN 108 AND 122 AND s2.ss_list_price BETWEEN 108 AND 122 GROUP BY item1.i_item_sk, item2.i_item_sk ORDER BY cnt;
