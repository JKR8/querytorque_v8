WITH ss_sr_item_agg AS (SELECT ss_customer_sk, ss_item_sk, COUNT(*) AS ss_sr_count FROM store_sales JOIN store_returns ON ss_ticket_number = sr_ticket_number AND ss_item_sk = sr_item_sk JOIN item ON ss_item_sk = i_item_sk WHERE i_category IN ('Children', 'Home', 'Women') AND ss_sales_price / ss_list_price BETWEEN 0.76 AND 0.96 GROUP BY ss_customer_sk, ss_item_sk), filtered_customer AS (SELECT c_customer_sk, c_first_name, c_last_name, c_current_addr_sk, c_current_hdemo_sk FROM customer JOIN customer_address ON c_current_addr_sk = ca_address_sk JOIN household_demographics ON c_current_hdemo_sk = hd_demo_sk WHERE ca_state IN ('AR', 'GA', 'IN', 'KY', 'VA') AND hd_income_band_sk BETWEEN 8 AND 14 AND hd_buy_potential = '501-1000') SELECT c.c_customer_sk, c.c_first_name, c.c_last_name, COUNT(*) AS cnt FROM ss_sr_item_agg AS agg JOIN web_sales ON agg.ss_customer_sk = ws_bill_customer_sk AND agg.ss_item_sk = ws_item_sk JOIN filtered_customer AS c ON agg.ss_customer_sk = c.c_customer_sk JOIN date_dim AS d1 ON sr_returned_date_sk = d1.d_date_sk JOIN date_dim AS d2 ON ws_sold_date_sk = d2.d_date_sk WHERE d2.d_date BETWEEN d1.d_date AND (d1.d_date + INTERVAL '90 DAY') AND d1.d_year = 2000 GROUP BY c.c_customer_sk, c.c_first_name, c.c_last_name ORDER BY cnt;
