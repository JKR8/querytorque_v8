WITH date_range AS (SELECT d_date_sk FROM date_dim WHERE d_year = 1999 AND d_moy BETWEEN 3 AND 5), valid_store_customers AS (SELECT DISTINCT ss_customer_sk FROM store_sales JOIN date_range ON ss_sold_date_sk = d_date_sk WHERE ss_list_price BETWEEN 52 AND 141), invalid_web_customers AS (SELECT DISTINCT ws_bill_customer_sk FROM web_sales JOIN date_range ON ws_sold_date_sk = d_date_sk WHERE ws_list_price BETWEEN 52 AND 141), invalid_catalog_customers AS (SELECT DISTINCT cs_ship_customer_sk FROM catalog_sales JOIN date_range ON cs_sold_date_sk = d_date_sk WHERE cs_list_price BETWEEN 52 AND 141) SELECT cd_gender, cd_marital_status, cd_education_status, COUNT(*) AS cnt1, cd_purchase_estimate, COUNT(*) AS cnt2, cd_credit_rating, COUNT(*) AS cnt3 FROM customer AS c JOIN customer_address AS ca ON c.c_current_addr_sk = ca.ca_address_sk JOIN customer_demographics AS cd ON cd_demo_sk = c.c_current_cdemo_sk WHERE ca_state IN ('GA', 'KY', 'SD') AND cd_marital_status IN ('D', 'W', 'U') AND cd_education_status IN ('Primary', 'College') AND c.c_customer_sk IN (SELECT ss_customer_sk FROM valid_store_customers) AND NOT c.c_customer_sk IN (SELECT ws_bill_customer_sk FROM invalid_web_customers) AND NOT c.c_customer_sk IN (SELECT cs_ship_customer_sk FROM invalid_catalog_customers) GROUP BY cd_gender, cd_marital_status, cd_education_status, cd_purchase_estimate, cd_credit_rating ORDER BY cd_gender, cd_marital_status, cd_education_status, cd_purchase_estimate, cd_credit_rating LIMIT 100;
