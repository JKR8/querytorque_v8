WITH date_filter AS (SELECT d_date_sk FROM date_dim WHERE d_year = 1999 AND d_moy BETWEEN 3 AND 5), filtered_customers AS (SELECT c.c_customer_sk, c.c_current_cdemo_sk, c.c_current_addr_sk FROM customer AS c INNER JOIN customer_address AS ca ON c.c_current_addr_sk = ca.ca_address_sk WHERE ca_state IN ('GA', 'KY', 'SD')) SELECT cd_gender, cd_marital_status, cd_education_status, COUNT(*) AS cnt1, cd_purchase_estimate, COUNT(*) AS cnt2, cd_credit_rating, COUNT(*) AS cnt3 FROM filtered_customers AS fc INNER JOIN customer_demographics AS cd ON fc.c_current_cdemo_sk = cd.cd_demo_sk LEFT JOIN store_sales AS ss ON fc.c_customer_sk = ss.ss_customer_sk AND ss.ss_sold_date_sk IN (SELECT d_date_sk FROM date_filter) AND ss.ss_list_price BETWEEN 52 AND 141 WHERE cd_marital_status IN ('D', 'W', 'U') AND cd_education_status IN ('Primary', 'College') AND NOT ss.ss_customer_sk IS NULL AND NOT EXISTS(SELECT 1 FROM web_sales AS ws INNER JOIN date_filter AS df ON ws.ws_sold_date_sk = df.d_date_sk WHERE ws.ws_bill_customer_sk = fc.c_customer_sk AND ws.ws_list_price BETWEEN 52 AND 141) AND NOT EXISTS(SELECT 1 FROM catalog_sales AS cs INNER JOIN date_filter AS df ON cs.cs_sold_date_sk = df.d_date_sk WHERE cs.cs_ship_customer_sk = fc.c_customer_sk AND cs.cs_list_price BETWEEN 52 AND 141) GROUP BY cd_gender, cd_marital_status, cd_education_status, cd_purchase_estimate, cd_credit_rating ORDER BY cd_gender, cd_marital_status, cd_education_status, cd_purchase_estimate, cd_credit_rating LIMIT 100;
