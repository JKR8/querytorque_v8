WITH date_filtered AS (SELECT d_date_sk FROM date_dim WHERE d_month_seq BETWEEN 1195 AND 1195 + 11), sa AS (SELECT ss_store_sk, ss_item_sk, SUM(ss_sales_price) AS revenue FROM store_sales INNER JOIN date_filtered ON ss_sold_date_sk = d_date_sk WHERE ss_sales_price / ss_list_price BETWEEN 38 * 0.01 AND 48 * 0.01 GROUP BY ss_store_sk, ss_item_sk), sb AS (SELECT ss_store_sk, AVG(revenue) AS ave FROM sa GROUP BY ss_store_sk), sc AS (SELECT ss_store_sk, ss_item_sk, SUM(ss_sales_price) AS revenue FROM store_sales INNER JOIN date_filtered ON ss_sold_date_sk = d_date_sk WHERE ss_sales_price / ss_list_price BETWEEN 38 * 0.01 AND 48 * 0.01 GROUP BY ss_store_sk, ss_item_sk) SELECT s_store_name, i_item_desc, sc.revenue, i_current_price, i_wholesale_cost, i_brand FROM sb INNER JOIN sc ON sb.ss_store_sk = sc.ss_store_sk INNER JOIN store ON sc.ss_store_sk = store.s_store_sk INNER JOIN item ON sc.ss_item_sk = item.i_item_sk WHERE sc.revenue <= 0.1 * sb.ave AND store.s_state IN ('IA', 'IL', 'NC') AND item.i_manager_id BETWEEN 80 AND 84 ORDER BY s_store_name, i_item_desc LIMIT 100;
