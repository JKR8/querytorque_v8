WITH filtered_catalog_sales AS (SELECT cs_item_sk, cs_warehouse_sk, cs_sold_date_sk, cs_order_number, cs_sales_price, cs_wholesale_cost FROM catalog_sales WHERE cs_wholesale_cost BETWEEN 68 AND 87), joined_data AS (SELECT cs.cs_item_sk, cs.cs_warehouse_sk, cs.cs_sold_date_sk, cs.cs_order_number, cs.cs_sales_price, cr.cr_refunded_cash, w.w_state, i.i_item_id, d.d_date FROM filtered_catalog_sales AS cs LEFT JOIN catalog_returns AS cr ON cs.cs_order_number = cr.cr_order_number AND cs.cs_item_sk = cr.cr_item_sk AND cr.cr_reason_sk = 40 INNER JOIN warehouse AS w ON cs.cs_warehouse_sk = w.w_warehouse_sk INNER JOIN item AS i ON cs.cs_item_sk = i.i_item_sk INNER JOIN date_dim AS d ON cs.cs_sold_date_sk = d.d_date_sk WHERE d.d_date BETWEEN (CAST('2002-02-20' AS DATE) - INTERVAL '30 DAY') AND (CAST('2002-02-20' AS DATE) + INTERVAL '30 DAY') AND i.i_category = 'Shoes' AND i.i_manager_id BETWEEN 42 AND 81) SELECT w_state, i_item_id, SUM(CASE WHEN (CAST(d_date AS DATE) < CAST('2002-02-20' AS DATE)) THEN cs_sales_price - COALESCE(cr_refunded_cash, 0) ELSE 0 END) AS sales_before, SUM(CASE WHEN (CAST(d_date AS DATE) >= CAST('2002-02-20' AS DATE)) THEN cs_sales_price - COALESCE(cr_refunded_cash, 0) ELSE 0 END) AS sales_after FROM joined_data WHERE TRUE GROUP BY w_state, i_item_id ORDER BY w_state, i_item_id LIMIT 100;
