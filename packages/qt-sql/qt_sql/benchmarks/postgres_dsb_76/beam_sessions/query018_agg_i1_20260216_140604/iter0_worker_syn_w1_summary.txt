WITH filtered_date AS (SELECT d_date_sk FROM date_dim WHERE d_year = 1998), filtered_item AS (SELECT i_item_sk, i_item_id FROM item WHERE i_category = 'Sports'), filtered_cd AS (SELECT cd_demo_sk FROM customer_demographics WHERE cd_gender = 'F' AND cd_education_status = 'College'), filtered_ca AS (SELECT ca_address_sk, ca_country, ca_state, ca_county FROM customer_address WHERE ca_state IN ('AR', 'NY', 'TX')) SELECT i_item_id, ca_country, ca_state, ca_county, AVG(CAST(cs_quantity AS DECIMAL(12, 2))) AS agg1, AVG(CAST(cs_list_price AS DECIMAL(12, 2))) AS agg2, AVG(CAST(cs_coupon_amt AS DECIMAL(12, 2))) AS agg3, AVG(CAST(cs_sales_price AS DECIMAL(12, 2))) AS agg4, AVG(CAST(cs_net_profit AS DECIMAL(12, 2))) AS agg5, AVG(CAST(c_birth_year AS DECIMAL(12, 2))) AS agg6 FROM catalog_sales AS cs INNER JOIN filtered_date AS fd ON cs.cs_sold_date_sk = fd.d_date_sk INNER JOIN filtered_item AS fi ON cs.cs_item_sk = fi.i_item_sk INNER JOIN filtered_cd AS fcd ON cs.cs_bill_cdemo_sk = fcd.cd_demo_sk INNER JOIN customer AS c ON cs.cs_bill_customer_sk = c.c_customer_sk INNER JOIN filtered_ca AS fca ON c.c_current_addr_sk = fca.ca_address_sk WHERE cs.cs_wholesale_cost BETWEEN 41 AND 46 AND c.c_birth_month = 2 GROUP BY ROLLUP (i_item_id, ca_country, ca_state, ca_county) ORDER BY ca_country, ca_state, ca_county, i_item_id LIMIT 100;
