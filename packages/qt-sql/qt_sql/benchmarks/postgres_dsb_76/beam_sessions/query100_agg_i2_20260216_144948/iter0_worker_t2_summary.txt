WITH prefetched_sales AS (SELECT ss_ticket_number, ss_item_sk, ss_customer_sk, ss_sold_date_sk FROM store_sales WHERE ss_list_price BETWEEN 115 AND 129) SELECT item1.i_item_sk, item2.i_item_sk, COUNT(*) AS cnt FROM prefetched_sales AS s1 INNER JOIN item AS item1 ON s1.ss_item_sk = item1.i_item_sk INNER JOIN prefetched_sales AS s2 ON s1.ss_ticket_number = s2.ss_ticket_number AND s1.ss_item_sk < s2.ss_item_sk INNER JOIN item AS item2 ON s2.ss_item_sk = item2.i_item_sk INNER JOIN date_dim ON s1.ss_sold_date_sk = d_date_sk INNER JOIN customer ON s1.ss_customer_sk = c_customer_sk INNER JOIN customer_address ON c_current_addr_sk = ca_address_sk INNER JOIN customer_demographics ON c_current_cdemo_sk = cd_demo_sk WHERE d_year BETWEEN 1999 AND 1999 + 1 AND item1.i_category IN ('Electronics', 'Men') AND item2.i_manager_id BETWEEN 44 AND 63 AND cd_marital_status = 'D' AND cd_education_status = 'College' GROUP BY item1.i_item_sk, item2.i_item_sk ORDER BY cnt;
