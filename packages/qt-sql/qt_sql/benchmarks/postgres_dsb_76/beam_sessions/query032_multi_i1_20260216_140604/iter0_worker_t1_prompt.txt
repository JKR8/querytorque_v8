select  sum(cs_ext_discount_amt)  as "excess discount amount"
from

(i_manufact_id in (47, 226, 612, 676, 818)
or i_manager_id BETWEEN 71 and 100)
and i_item_sk = cs_item_sk
and d_date between '1998-01-06' and
        cast('1998-01-06' as date) + interval '90 day'
and d_date_sk = cs_sold_date_sk
and cs_ext_discount_amt
     > (
         select
            1.3 * avg(cs_ext_discount_amt)
         from
            catalog_sales
           ,date_dim

              cs_item_sk = i_item_sk
          and d_date between '1998-01-06' and
                             cast('1998-01-06' as date) + interval '90 day'
          and d_date_sk = cs_sold_date_sk
          and cs_list_price between 108 and 137
          and cs_sales_price / cs_list_price BETWEEN 23 * 0.01 AND 43 * 0.01
      )
order by sum(cs_ext_discount_amt)
limit 100;
