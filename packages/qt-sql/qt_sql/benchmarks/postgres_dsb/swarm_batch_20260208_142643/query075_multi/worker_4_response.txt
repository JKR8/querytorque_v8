```sql
WITH filtered_date AS (
    SELECT d_date_sk
    FROM date_dim
    WHERE d_year IN (2001, 2002)
),
filtered_item AS (
    SELECT 
        i_item_sk,
        i_brand_id,
        i_class_id,
        i_category_id,
        i_manufact_id
    FROM item
    WHERE i_category = 'Music'
),
sales_detail AS (
    SELECT
        d_year,
        i_brand_id,
        i_class_id,
        i_category_id,
        i_manufact_id,
        cs_quantity - COALESCE(cr_return_quantity, 0) AS sales_cnt,
        cs_ext_sales_price - COALESCE(cr_return_amount, 0.0) AS sales_amt
    FROM catalog_sales
    JOIN filtered_item ON i_item_sk = cs_item_sk
    JOIN filtered_date ON d_date_sk = cs_sold_date_sk
    LEFT JOIN catalog_returns ON (
        cs_order_number = cr_order_number 
        AND cs_item_sk = cr_item_sk
        AND cr_reason_sk IN (17, 48, 50, 56, 68)
    )
    WHERE cs_sales_price / cs_list_price BETWEEN 33 * 0.01 AND 53 * 0.01
    
    UNION ALL
    
    SELECT
        d_year,
        i_brand_id,
        i_class_id,
        i_category_id,
        i_manufact_id,
        ss_quantity - COALESCE(sr_return_quantity, 0) AS sales_cnt,
        ss_ext_sales_price - COALESCE(sr_return_amt, 0.0) AS sales_amt
    FROM store_sales
    JOIN filtered_item ON i_item_sk = ss_item_sk
    JOIN filtered_date ON d_date_sk = ss_sold_date_sk
    LEFT JOIN store_returns ON (
        ss_ticket_number = sr_ticket_number 
        AND ss_item_sk = sr_item_sk
        AND sr_reason_sk IN (17, 48, 50, 56, 68)
    )
    WHERE ss_sales_price / ss_list_price BETWEEN 33 * 0.01 AND 53 * 0.01
    
    UNION ALL
    
    SELECT
        d_year,
        i_brand_id,
        i_class_id,
        i_category_id,
        i_manufact_id,
        ws_quantity - COALESCE(wr_return_quantity, 0) AS sales_cnt,
        ws_ext_sales_price - COALESCE(wr_return_amt, 0.0) AS sales_amt
    FROM web_sales
    JOIN filtered_item ON i_item_sk = ws_item_sk
    JOIN filtered_date ON d_date_sk = ws_sold_date_sk
    LEFT JOIN web_returns ON (
        ws_order_number = wr_order_number 
        AND ws_item_sk = wr_item_sk
        AND wr_reason_sk IN (17, 48, 50, 56, 68)
    )
    WHERE ws_sales_price / ws_list_price BETWEEN 33 * 0.01 AND 53 * 0.01
),
all_sales AS (
    SELECT
        d_year,
        i_brand_id,
        i_class_id,
        i_category_id,
        i_manufact_id,
        SUM(sales_cnt) AS sales_cnt,
        SUM(sales_amt) AS sales_amt
    FROM sales_detail
    GROUP BY
        d_year,
        i_brand_id,
        i_class_id,
        i_category_id,
        i_manufact_id
),
pivot_years AS (
    SELECT
        i_brand_id,
        i_class_id,
        i_category_id,
        i_manufact_id,
        MAX(CASE WHEN d_year = 2001 THEN sales_cnt END) AS prev_yr_cnt,
        MAX(CASE WHEN d_year = 2001 THEN sales_amt END) AS prev_yr_amt,
        MAX(CASE WHEN d_year = 2002 THEN sales_cnt END) AS curr_yr_cnt,
        MAX(CASE WHEN d_year = 2002 THEN sales_amt END) AS curr_yr_amt
    FROM all_sales
    WHERE d_year IN (2001, 2002)
    GROUP BY
        i_brand_id,
        i_class_id,
        i_category_id,
        i_manufact_id
)
SELECT
    2001 AS prev_year,
    2002 AS year,
    i_brand_id,
    i_class_id,
    i_category_id,
    i_manufact_id,
    prev_yr_cnt,
    curr_yr_cnt,
    curr_yr_cnt - prev_yr_cnt AS sales_cnt_diff,
    curr_yr_amt - prev_yr_amt AS sales_amt_diff
FROM pivot_years
WHERE prev_yr_cnt > 0
    AND CAST(curr_yr_cnt AS DECIMAL(17, 2)) / CAST(prev_yr_cnt AS DECIMAL(17, 2)) < 0.9
ORDER BY sales_cnt_diff, sales_amt_diff
LIMIT 100;
```

**Changes:** 
1. Decomposed the self-join by creating a single CTE (`pivot_years`) that uses conditional aggregation to compute both year's metrics in one pass over the aggregated data.
2. Added pre-filtering CTEs for date_dim (years 2001-2002) and item (category='Music') to reduce early join sizes.
3. Moved reason_sk filters into the LEFT JOIN conditions to apply them before the UNION.

**Expected speedup:** 2-3x from eliminating redundant scans and early dimension filtering.
