```sql
WITH filtered_date AS (
    SELECT d_date_sk
    FROM date_dim
    WHERE d_year BETWEEN 2000 AND 2000 + 1
),
filtered_demographics AS (
    SELECT cd_demo_sk
    FROM customer_demographics
    WHERE cd_marital_status = 'S'
      AND cd_education_status = 'Secondary'
),
filtered_sales AS (
    SELECT 
        ss_ticket_number,
        ss_item_sk,
        ss_customer_sk
    FROM store_sales
    INNER JOIN filtered_date ON ss_sold_date_sk = d_date_sk
    INNER JOIN filtered_demographics ON ss_cdemo_sk = cd_demo_sk
    WHERE ss_list_price BETWEEN 16 AND 30
),
sales_with_item1 AS (
    SELECT 
        f.ss_ticket_number,
        f.ss_item_sk,
        i.i_item_sk,
        i.i_category
    FROM filtered_sales f
    INNER JOIN item i ON f.ss_item_sk = i.i_item_sk
    WHERE i.i_category IN ('Electronics', 'Men')
),
sales_with_item2 AS (
    SELECT 
        f.ss_ticket_number,
        f.ss_item_sk,
        i.i_item_sk,
        i.i_manager_id
    FROM filtered_sales f
    INNER JOIN item i ON f.ss_item_sk = i.i_item_sk
    WHERE i.i_manager_id BETWEEN 81 AND 100
)
SELECT
    s1.i_item_sk,
    s2.i_item_sk,
    COUNT(*) AS cnt
FROM sales_with_item1 s1
INNER JOIN sales_with_item2 s2 
    ON s1.ss_ticket_number = s2.ss_ticket_number
    AND s1.i_item_sk < s2.i_item_sk
GROUP BY
    s1.i_item_sk,
    s2.i_item_sk
ORDER BY
    cnt;
```

**Changes**: Pre-filtered dimensions (date, demographics) and fact table (store_sales) into separate CTEs before joining to items. Created separate derived tables for item1 and item2 conditions, then self-joined on ticket number with inequality. Removed unnecessary customer_address join.

**Expected speedup**: 2-3x due to early filtering of large fact table, reduced join cardinality, and elimination of redundant customer_address joins.
