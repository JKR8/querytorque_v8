```sql
WITH filtered_date AS (
    SELECT d_date_sk
    FROM date_dim
    WHERE d_date BETWEEN '1999-03-07' AND CAST('1999-03-07' AS DATE) + INTERVAL '90 DAY'
),
filtered_item AS (
    SELECT i_item_sk
    FROM item
    WHERE i_manufact_id IN (1, 78, 97, 516, 521) 
       OR i_manager_id BETWEEN 25 AND 54
),
item_thresholds AS (
    SELECT 
        cs_item_sk,
        1.3 * AVG(cs_ext_discount_amt) AS threshold
    FROM catalog_sales
    JOIN filtered_date ON cs_sold_date_sk = d_date_sk
    WHERE cs_list_price BETWEEN 16 AND 45
      AND cs_sales_price / cs_list_price BETWEEN 63 * 0.01 AND 83 * 0.01
    GROUP BY cs_item_sk
)
SELECT
    SUM(cs_ext_discount_amt) AS "excess discount amount"
FROM catalog_sales
JOIN filtered_item ON cs_item_sk = i_item_sk
JOIN filtered_date ON cs_sold_date_sk = d_date_sk
LEFT JOIN item_thresholds ON cs_item_sk = item_thresholds.cs_item_sk
WHERE cs_ext_discount_amt > COALESCE(item_thresholds.threshold, -1)
ORDER BY
    SUM(cs_ext_discount_amt)
LIMIT 100;
```

**Changes**: Decorrelated the subquery by precomputing item-specific thresholds in a separate CTE, isolated selective date and item filters into independent CTEs, and converted implicit joins to explicit JOIN syntax. Used LEFT JOIN with COALESCE to handle items without matching thresholds.

**Expected speedup**: ~2-3x due to eliminating per-row correlated subquery execution, enabling early filtering, and better join planning with small dimension CTEs.
