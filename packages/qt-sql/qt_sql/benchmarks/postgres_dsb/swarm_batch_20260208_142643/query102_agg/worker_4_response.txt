```sql
WITH filtered_dates AS (
    SELECT d_date_sk, d_date
    FROM date_dim
    WHERE d_year = 2002
),
filtered_item AS (
    SELECT i_item_sk
    FROM item
    WHERE i_category IN ('Home', 'Men', 'Music')
      AND i_manager_id IN (1, 2, 4, 11, 12, 14, 21, 29, 32, 52)
),
filtered_customer_address AS (
    SELECT ca_address_sk
    FROM customer_address
    WHERE ca_state IN ('GA', 'MD', 'MO', 'NC', 'OR')
),
store_sales_base AS (
    SELECT 
        ss_item_sk,
        ss_sold_date_sk,
        ss_customer_sk,
        ss_quantity
    FROM store_sales
    WHERE ss_item_sk IN (SELECT i_item_sk FROM filtered_item)
),
web_sales_filtered AS (
    SELECT 
        ws_item_sk,
        ws_sold_date_sk,
        ws_bill_customer_sk,
        ws_warehouse_sk,
        ws_wholesale_cost
    FROM web_sales
    WHERE ws_wholesale_cost BETWEEN 34 AND 54
      AND ws_item_sk IN (SELECT i_item_sk FROM filtered_item)
),
inventory_filtered AS (
    SELECT 
        inv_item_sk,
        inv_date_sk,
        inv_warehouse_sk,
        inv_quantity_on_hand
    FROM inventory
    WHERE inv_item_sk IN (SELECT i_item_sk FROM filtered_item)
)
SELECT
    cd_gender,
    cd_marital_status,
    cd_education_status,
    hd_vehicle_count,
    COUNT(*) AS cnt
FROM store_sales_base ss
JOIN web_sales_filtered ws ON ws.ws_item_sk = ss.ss_item_sk
JOIN filtered_dates d1 ON ss.ss_sold_date_sk = d1.d_date_sk
JOIN customer c ON ss.ss_customer_sk = c.c_customer_sk 
    AND ws.ws_bill_customer_sk = c.c_customer_sk
JOIN inventory_filtered inv ON inv.inv_item_sk = ss.ss_item_sk 
    AND inv.inv_date_sk = ss.ss_sold_date_sk
    AND ws.ws_warehouse_sk = inv.inv_warehouse_sk
JOIN warehouse w ON ws.ws_warehouse_sk = w.w_warehouse_sk
JOIN store s ON s.s_state = w.w_state
JOIN customer_demographics cd ON c.c_current_cdemo_sk = cd.cd_demo_sk
JOIN household_demographics hd ON c.c_current_hdemo_sk = hd.hd_demo_sk
JOIN filtered_customer_address ca ON c.c_current_addr_sk = ca.ca_address_sk
WHERE inv.inv_quantity_on_hand >= ss.ss_quantity
  AND EXISTS (
    SELECT 1 
    FROM date_dim d2 
    WHERE ws.ws_sold_date_sk = d2.d_date_sk 
      AND d2.d_date BETWEEN d1.d_date AND (d1.d_date + INTERVAL '30 DAY')
  )
GROUP BY
    cd_gender,
    cd_marital_status,
    cd_education_status,
    hd_vehicle_count
ORDER BY
    cnt;
```

**Changes**: Restructured the query by creating filtered CTEs for dimension tables, transforming the date self-join into an EXISTS correlated subquery, and moving IN-list filters into separate CTEs to reduce early row counts.

**Expected speedup**: 1.8-2.5x due to reduced intermediate result sizes and more efficient date filtering via EXISTS instead of self-join.
