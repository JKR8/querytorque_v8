```sql
WITH filtered_item AS (
    SELECT i_item_sk
    FROM item
    WHERE i_category IN ('Children', 'Electronics', 'Women')
),
filtered_d1 AS (
    SELECT d_date_sk, d_date
    FROM date_dim
    WHERE d_year = 2000
),
filtered_ca AS (
    SELECT ca_address_sk
    FROM customer_address
    WHERE ca_state IN ('GA', 'IA', 'LA', 'MO', 'SD')
),
filtered_hd AS (
    SELECT hd_demo_sk
    FROM household_demographics
    WHERE hd_income_band_sk BETWEEN 12 AND 18
      AND hd_buy_potential = '0-500'
)
SELECT
    c.c_customer_sk,
    c.c_first_name,
    c.c_last_name,
    COUNT(*) AS cnt
FROM store_sales ss
INNER JOIN store_returns sr
    ON ss.ss_ticket_number = sr.sr_ticket_number
    AND ss.ss_item_sk = sr.sr_item_sk
INNER JOIN web_sales ws
    ON ss.ss_customer_sk = ws.ws_bill_customer_sk
    AND sr.sr_item_sk = ws.ws_item_sk
INNER JOIN filtered_item i
    ON ss.ss_item_sk = i.i_item_sk
INNER JOIN customer c
    ON ss.ss_customer_sk = c.c_customer_sk
INNER JOIN filtered_ca ca
    ON c.c_current_addr_sk = ca.ca_address_sk
INNER JOIN filtered_hd hd
    ON c.c_current_hdemo_sk = hd.hd_demo_sk
INNER JOIN filtered_d1 d1
    ON sr.sr_returned_date_sk = d1.d_date_sk
INNER JOIN date_dim d2
    ON ws.ws_sold_date_sk = d2.d_date_sk
    AND d2.d_date BETWEEN d1.d_date AND (d1.d_date + INTERVAL '90 DAY')
WHERE ss.ss_sales_price / ss.ss_list_price BETWEEN 34 * 0.01 AND 54 * 0.01
GROUP BY
    c.c_customer_sk,
    c.c_first_name,
    c.c_last_name
ORDER BY cnt
```

**Changes:** Isolated selective dimension filters (item, date_dim d1, customer_address, household_demographics) into separate CTEs, converted implicit joins to explicit JOIN syntax, and moved the date range condition into the JOIN ON clause for better predicate pushdown.

**Expected speedup:** ~2-3x from reduced intermediate result sizes and better hash join planning with small dimension CTEs.
