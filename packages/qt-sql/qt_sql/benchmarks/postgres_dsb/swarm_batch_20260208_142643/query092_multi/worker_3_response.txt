```sql
WITH filtered_date AS (
    SELECT d_date_sk
    FROM date_dim
    WHERE d_date BETWEEN '1998-03-13' AND CAST('1998-03-13' AS DATE) + INTERVAL '90 DAY'
),
filtered_item AS (
    SELECT i_item_sk
    FROM item
    WHERE i_manufact_id BETWEEN 341 AND 540 OR i_category IN ('Home', 'Men', 'Music')
),
ws_base AS (
    SELECT 
        ws_item_sk,
        ws_ext_discount_amt,
        ws_sales_price,
        ws_list_price,
        ws_wholesale_cost
    FROM web_sales
    INNER JOIN filtered_date ON ws_sold_date_sk = d_date_sk
    WHERE ws_wholesale_cost BETWEEN 26 AND 46
),
ws_avg_discount AS (
    SELECT 
        ws_item_sk,
        1.3 * AVG(ws_ext_discount_amt) AS threshold
    FROM ws_base
    WHERE ws_sales_price / ws_list_price BETWEEN 34 * 0.01 AND 49 * 0.01
    GROUP BY ws_item_sk
)
SELECT
    SUM(ws_ext_discount_amt) AS "Excess Discount Amount"
FROM ws_base
INNER JOIN filtered_item ON ws_base.ws_item_sk = filtered_item.i_item_sk
INNER JOIN ws_avg_discount ON ws_base.ws_item_sk = ws_avg_discount.ws_item_sk
WHERE ws_base.ws_ext_discount_amt > ws_avg_discount.threshold
ORDER BY SUM(ws_ext_discount_amt)
LIMIT 100;
```

Changes: Materialized all pre-filtered dimensions (date_dim, item) and fact table (web_sales) into separate CTEs, then computed the per-item average discount threshold once in a CTE before the main join, eliminating the correlated subquery.

Expected speedup: 3-4x by eliminating repeated scans and enabling better join optimization with pre-materialized small result sets.
