```sql
WITH date_filtered AS (
    SELECT d_date_sk
    FROM date_dim
    WHERE d_year = 2002
      AND d_moy = 12
)
SELECT
    MIN(i.i_brand_id),
    MIN(i.i_manufact_id),
    MIN(ss.ss_ext_sales_price)
FROM store_sales ss
JOIN date_filtered d ON ss.ss_sold_date_sk = d.d_date_sk
JOIN (
    SELECT i_item_sk, i_brand_id, i_manufact_id
    FROM item
    WHERE i_category = 'Home'
) i ON ss.ss_item_sk = i.i_item_sk
WHERE ss.ss_wholesale_cost BETWEEN 34 AND 54
  AND EXISTS (
    SELECT 1
    FROM customer c
    JOIN customer_address ca ON c.c_current_addr_sk = ca.ca_address_sk
    WHERE ss.ss_customer_sk = c.c_customer_sk
      AND c.c_birth_month = 1
      AND ca.ca_state = 'TX'
      AND EXISTS (
        SELECT 1
        FROM store s
        WHERE ss.ss_store_sk = s.s_store_sk
          AND SUBSTRING(ca.ca_zip FROM 1 FOR 5) <> SUBSTRING(s.s_zip FROM 1 FOR 5)
      )
  )
```

**Changes**: Converted date_dim and item to filtered inline subqueries, rewrote customer/address/store joins as EXISTS semi-joins to leverage foreign key indexes, kept wholesale_cost filter directly on store_sales, and preserved all original logic.

**Expected speedup**: ~2-3x by reducing dimension table sizes early and enabling hash joins with pre-filtered small dimension sets while maintaining efficient semi-join execution.
