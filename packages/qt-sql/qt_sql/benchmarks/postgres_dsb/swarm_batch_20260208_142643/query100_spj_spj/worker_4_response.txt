```sql
WITH filtered_dates AS (
    SELECT d_date_sk
    FROM date_dim
    WHERE d_year BETWEEN 2000 AND 2000 + 1
),
filtered_customer_demographics AS (
    SELECT cd_demo_sk
    FROM customer_demographics
    WHERE cd_marital_status = 'S'
      AND cd_education_status = 'Secondary'
),
filtered_items1 AS (
    SELECT i_item_sk
    FROM item
    WHERE i_category IN ('Electronics', 'Men')
),
filtered_items2 AS (
    SELECT i_item_sk
    FROM item
    WHERE i_manager_id BETWEEN 81 AND 100
),
ticket_sales AS (
    SELECT 
        ss_ticket_number,
        ss_item_sk,
        ss_customer_sk,
        ss_sold_date_sk
    FROM store_sales
    WHERE ss_list_price BETWEEN 16 AND 30
),
filtered_tickets AS (
    SELECT 
        ts.ss_ticket_number,
        ts.ss_item_sk,
        ts.ss_sold_date_sk
    FROM ticket_sales ts
    JOIN filtered_dates fd ON ts.ss_sold_date_sk = fd.d_date_sk
    JOIN customer c ON ts.ss_customer_sk = c.c_customer_sk
    JOIN customer_address ca ON c.c_current_addr_sk = ca.ca_address_sk
    JOIN filtered_customer_demographics fcd ON c.c_current_cdemo_sk = fcd.cd_demo_sk
)
SELECT
    MIN(item1.i_item_sk),
    MIN(item2.i_item_sk),
    MIN(t1.ss_ticket_number),
    MIN(t1.ss_item_sk)
FROM filtered_tickets t1
JOIN filtered_tickets t2 ON t1.ss_ticket_number = t2.ss_ticket_number
JOIN filtered_items1 item1 ON t1.ss_item_sk = item1.i_item_sk
JOIN filtered_items2 item2 ON t2.ss_item_sk = item2.i_item_sk
WHERE item1.i_item_sk < item2.i_item_sk;
```

**Changes**: Decomposed the self-join by first creating a filtered ticket CTE with all dimension joins applied once, then performing the self-join on tickets and joining to separately filtered item tables. This eliminates duplicate scanning of the fact table and dimensions.

**Expected speedup**: ~2-4x by reducing fact table scans from 2 to 1 and applying dimension filters early in materialized CTEs.
