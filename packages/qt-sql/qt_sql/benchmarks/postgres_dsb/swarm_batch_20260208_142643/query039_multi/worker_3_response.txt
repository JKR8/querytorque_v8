```sql
WITH 
filtered_date AS (
    SELECT d_date_sk, d_moy
    FROM date_dim
    WHERE d_year = 2002
),
filtered_item AS (
    SELECT i_item_sk
    FROM item
    WHERE i_category IN ('Home', 'Men')
      AND i_manager_id BETWEEN 25 AND 44
),
filtered_warehouse AS (
    SELECT w_warehouse_sk, w_warehouse_name
    FROM warehouse
),
base_inventory AS (
    SELECT
        w.w_warehouse_name,
        w.w_warehouse_sk,
        i.i_item_sk,
        fd.d_moy,
        STDDEV_SAMP(inv.inv_quantity_on_hand) AS stdev,
        AVG(inv.inv_quantity_on_hand) AS mean
    FROM inventory AS inv
    JOIN filtered_item AS i ON inv.inv_item_sk = i.i_item_sk
    JOIN filtered_warehouse AS w ON inv.inv_warehouse_sk = w.w_warehouse_sk
    JOIN filtered_date AS fd ON inv.inv_date_sk = fd.d_date_sk
    WHERE inv.inv_quantity_on_hand BETWEEN 140 AND 340
    GROUP BY
        w.w_warehouse_name,
        w.w_warehouse_sk,
        i.i_item_sk,
        fd.d_moy
),
inv_stats AS (
    SELECT
        w_warehouse_name,
        w_warehouse_sk,
        i_item_sk,
        d_moy,
        stdev,
        mean,
        CASE mean WHEN 0 THEN NULL ELSE stdev / mean END AS cov
    FROM base_inventory
    WHERE CASE mean WHEN 0 THEN 0 ELSE stdev / mean END > 1
)
SELECT
    inv1.w_warehouse_sk,
    inv1.i_item_sk,
    inv1.d_moy,
    inv1.mean,
    inv1.cov,
    inv2.w_warehouse_sk,
    inv2.i_item_sk,
    inv2.d_moy,
    inv2.mean,
    inv2.cov
FROM inv_stats AS inv1
JOIN inv_stats AS inv2 
    ON inv1.i_item_sk = inv2.i_item_sk
    AND inv1.w_warehouse_sk = inv2.w_warehouse_sk
    AND inv1.d_moy = 10
    AND inv2.d_moy = 11
ORDER BY
    inv1.w_warehouse_sk,
    inv1.i_item_sk,
    inv1.d_moy,
    inv1.mean,
    inv1.cov,
    inv2.d_moy,
    inv2.mean,
    inv2.cov;
```

**Changes:** Applied aggressive_prefetch_selfjoin strategy by materializing filtered dimension tables into CTEs, then computing base inventory aggregation once and deriving both sides of the self-join from the same result. Converted implicit joins to explicit JOIN syntax.

**Expected speedup:** 2.5-3.5x due to reduced dimension table scans, single aggregation computation, and better join cardinality estimates from pre-filtered CTEs.
