```sql
WITH filtered_date AS (
    SELECT d_date_sk, d_date
    FROM date_dim
    WHERE d_date BETWEEN (CAST('1999-04-19' AS DATE) - INTERVAL '30 DAY')
                     AND (CAST('1999-04-19' AS DATE) + INTERVAL '30 DAY')
),
filtered_item AS (
    SELECT i_item_sk, i_item_id
    FROM item
    WHERE i_category = 'Home'
      AND i_manager_id BETWEEN 25 AND 64
),
filtered_catalog_sales AS (
    SELECT 
        cs_item_sk,
        cs_order_number,
        cs_sales_price,
        cs_warehouse_sk,
        cs_sold_date_sk
    FROM catalog_sales
    WHERE cs_wholesale_cost BETWEEN 17 AND 36
),
filtered_catalog_returns AS (
    SELECT 
        cr_order_number,
        cr_item_sk,
        cr_refunded_cash
    FROM catalog_returns
    WHERE cr_reason_sk = 16
)
SELECT
    w.w_state,
    i.i_item_id,
    SUM(
        CASE
            WHEN CAST(d.d_date AS DATE) < CAST('1999-04-19' AS DATE)
            THEN cs.cs_sales_price - COALESCE(cr.cr_refunded_cash, 0)
            ELSE 0
        END
    ) AS sales_before,
    SUM(
        CASE
            WHEN CAST(d.d_date AS DATE) >= CAST('1999-04-19' AS DATE)
            THEN cs.cs_sales_price - COALESCE(cr.cr_refunded_cash, 0)
            ELSE 0
        END
    ) AS sales_after
FROM filtered_catalog_sales cs
INNER JOIN filtered_item i ON cs.cs_item_sk = i.i_item_sk
INNER JOIN filtered_date d ON cs.cs_sold_date_sk = d.d_date_sk
INNER JOIN warehouse w ON cs.cs_warehouse_sk = w.w_warehouse_sk
LEFT OUTER JOIN filtered_catalog_returns cr 
    ON cs.cs_order_number = cr.cr_order_number 
    AND cs.cs_item_sk = cr.cr_item_sk
GROUP BY
    w.w_state,
    i.i_item_id
ORDER BY
    w.w_state,
    i.i_item_id
LIMIT 100;
```

**Changes**: Materialized pre-filtered dimension tables (date_dim, item) and fact tables (catalog_sales, catalog_returns) into separate CTEs to reduce row counts before joins. Converted implicit joins to explicit JOIN syntax.

**Expected speedup**: 2-3x due to early reduction of large fact tables and dimension tables before the main join, reducing the combinatorial search space.
