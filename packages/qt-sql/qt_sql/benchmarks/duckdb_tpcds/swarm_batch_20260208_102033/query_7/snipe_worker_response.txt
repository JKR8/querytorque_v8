```sql
WITH filtered_dates AS (
    SELECT d_date_sk
    FROM date_dim
    WHERE d_year = 2001
), filtered_customer_demographics AS (
    SELECT cd_demo_sk
    FROM customer_demographics
    WHERE cd_gender = 'F'
      AND cd_marital_status = 'W'
      AND cd_education_status = 'College'
), filtered_promotions AS (
    SELECT p_promo_sk
    FROM promotion
    WHERE p_channel_email = 'N' OR p_channel_event = 'N'
), filtered_sales AS (
    SELECT
        ss_item_sk,
        ss_quantity,
        ss_list_price,
        ss_coupon_amt,
        ss_sales_price
    FROM store_sales
    INNER JOIN filtered_dates ON store_sales.ss_sold_date_sk = filtered_dates.d_date_sk
    INNER JOIN filtered_customer_demographics ON store_sales.ss_cdemo_sk = filtered_customer_demographics.cd_demo_sk
    INNER JOIN filtered_promotions ON store_sales.ss_promo_sk = filtered_promotions.p_promo_sk
)
SELECT
    i.i_item_id,
    AVG(fs.ss_quantity) AS agg1,
    AVG(fs.ss_list_price) AS agg2,
    AVG(fs.ss_coupon_amt) AS agg3,
    AVG(fs.ss_sales_price) AS agg4
FROM filtered_sales fs
INNER JOIN item i ON fs.ss_item_sk = i.i_item_sk
GROUP BY i.i_item_id
ORDER BY i.i_item_id
LIMIT 100
```

Changes: Applied dimension_cte_isolate pattern by creating three filtered CTEs for date_dim, customer_demographics, and promotion before joining with the fact table. This creates small hash tables for early filtering and reduces the join workload on the large store_sales table.

Expected speedup: 2.0x+ due to reduced hash table sizes and early fact table filtering through surrogate key joins.
