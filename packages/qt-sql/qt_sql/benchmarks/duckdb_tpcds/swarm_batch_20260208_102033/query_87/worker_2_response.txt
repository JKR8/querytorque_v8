```sql
WITH filtered_dates AS (
  SELECT d_date_sk, d_date
  FROM date_dim
  WHERE d_month_seq BETWEEN 1184 AND 1184 + 11
),
customer_data AS (
  SELECT c_customer_sk, c_last_name, c_first_name
  FROM customer
),
store_customers AS (
  SELECT DISTINCT
    customer_data.c_last_name,
    customer_data.c_first_name,
    filtered_dates.d_date
  FROM store_sales
  JOIN filtered_dates ON store_sales.ss_sold_date_sk = filtered_dates.d_date_sk
  JOIN customer_data ON store_sales.ss_customer_sk = customer_data.c_customer_sk
),
catalog_customers AS (
  SELECT DISTINCT
    customer_data.c_last_name,
    customer_data.c_first_name,
    filtered_dates.d_date
  FROM catalog_sales
  JOIN filtered_dates ON catalog_sales.cs_sold_date_sk = filtered_dates.d_date_sk
  JOIN customer_data ON catalog_sales.cs_bill_customer_sk = customer_data.c_customer_sk
),
web_customers AS (
  SELECT DISTINCT
    customer_data.c_last_name,
    customer_data.c_first_name,
    filtered_dates.d_date
  FROM web_sales
  JOIN filtered_dates ON web_sales.ws_sold_date_sk = filtered_dates.d_date_sk
  JOIN customer_data ON web_sales.ws_bill_customer_sk = customer_data.c_customer_sk
)
SELECT COUNT(*)
FROM (
  (SELECT * FROM store_customers)
  EXCEPT
  (SELECT * FROM catalog_customers)
  EXCEPT
  (SELECT * FROM web_customers)
) AS cool_cust;
```

```
Changes: Isolated date filtering and customer data into CTEs, then created separate CTEs for each sales channel that reuse these shared dimensions. This eliminates redundant scans of date_dim and customer tables across the three EXCEPT branches.

Expected speedup: 1.5-2x due to reduced dimension table scans and better join optimization.
```
