```sql
WITH filtered_date AS (
    SELECT d_date_sk, d_date
    FROM date_dim
    WHERE d_month_seq BETWEEN 1184 AND 1184 + 11
),
store_sales_distinct AS (
    SELECT DISTINCT
        customer.c_last_name,
        customer.c_first_name,
        filtered_date.d_date
    FROM store_sales
    JOIN filtered_date ON store_sales.ss_sold_date_sk = filtered_date.d_date_sk
    JOIN customer ON store_sales.ss_customer_sk = customer.c_customer_sk
),
catalog_sales_distinct AS (
    SELECT DISTINCT
        customer.c_last_name,
        customer.c_first_name,
        filtered_date.d_date
    FROM catalog_sales
    JOIN filtered_date ON catalog_sales.cs_sold_date_sk = filtered_date.d_date_sk
    JOIN customer ON catalog_sales.cs_bill_customer_sk = customer.c_customer_sk
),
web_sales_distinct AS (
    SELECT DISTINCT
        customer.c_last_name,
        customer.c_first_name,
        filtered_date.d_date
    FROM web_sales
    JOIN filtered_date ON web_sales.ws_sold_date_sk = filtered_date.d_date_sk
    JOIN customer ON web_sales.ws_bill_customer_sk = customer.c_customer_sk
),
all_sales AS (
    SELECT c_last_name, c_first_name, d_date, 'store' AS channel FROM store_sales_distinct
    UNION ALL
    SELECT c_last_name, c_first_name, d_date, 'catalog' AS channel FROM catalog_sales_distinct
    UNION ALL
    SELECT c_last_name, c_first_name, d_date, 'web' AS channel FROM web_sales_distinct
),
aggregated AS (
    SELECT
        c_last_name,
        c_first_name,
        d_date,
        MAX(CASE WHEN channel = 'store' THEN 1 ELSE 0 END) AS in_store,
        MAX(CASE WHEN channel = 'catalog' THEN 1 ELSE 0 END) AS in_catalog,
        MAX(CASE WHEN channel = 'web' THEN 1 ELSE 0 END) AS in_web
    FROM all_sales
    GROUP BY c_last_name, c_first_name, d_date
),
cool_cust AS (
    SELECT c_last_name, c_first_name, d_date
    FROM aggregated
    WHERE in_store = 1 AND in_catalog = 0 AND in_web = 0
)
SELECT COUNT(*)
FROM cool_cust
```

**Changes**: Pre-filtered date_dim into a CTE to scan it once, then built distinct sets for each sales channel, combined them via UNION ALL, used conditional aggregation to compute set membership in a single pass, and finally applied the EXCEPT logic via WHERE conditions.

**Expected speedup**: ~2x from scanning date_dim once and replacing multiple EXCEPT operations with a single aggregation pass.
