```sql
WITH filtered_dates AS (
    SELECT d_date_sk
    FROM date_dim
    WHERE (d_dom BETWEEN 1 AND 3 OR d_dom BETWEEN 25 AND 28)
      AND d_year IN (1998, 1999, 2000)
),
filtered_stores AS (
    SELECT s_store_sk
    FROM store
    WHERE s_county IN (
        'Ziebach County',
        'Daviess County',
        'Walker County',
        'Richland County',
        'Barrow County',
        'Franklin Parish',
        'Williamson County',
        'Luce County'
    )
),
filtered_hd AS (
    SELECT hd_demo_sk
    FROM household_demographics
    WHERE hd_vehicle_count > 0
      AND hd_dep_count / hd_vehicle_count > 1.2
      AND hd_buy_potential IN ('1001-5000', '0-500')
),
sales_agg AS (
    SELECT
        ss_ticket_number,
        ss_customer_sk,
        COUNT(*) AS cnt
    FROM store_sales
    JOIN filtered_dates ON store_sales.ss_sold_date_sk = filtered_dates.d_date_sk
    JOIN filtered_stores ON store_sales.ss_store_sk = filtered_stores.s_store_sk
    JOIN filtered_hd ON store_sales.ss_hdemo_sk = filtered_hd.hd_demo_sk
    GROUP BY ss_ticket_number, ss_customer_sk
    HAVING cnt BETWEEN 15 AND 20
)
SELECT
    c_last_name,
    c_first_name,
    c_salutation,
    c_preferred_cust_flag,
    sales_agg.ss_ticket_number,
    sales_agg.cnt
FROM sales_agg
JOIN customer ON sales_agg.ss_customer_sk = customer.c_customer_sk
ORDER BY
    c_last_name,
    c_first_name,
    c_salutation,
    c_preferred_cust_flag DESC,
    sales_agg.ss_ticket_number;
```

**Changes**: Created three focused CTEs to pre-filter dimension tables (date_dim, store, household_demographics) before joining with the large fact table. This reduces the hash table sizes for the initial joins and applies filters earlier. The CASE expression was simplified to a direct arithmetic condition since `hd_vehicle_count > 0` is already guaranteed.

**Expected speedup**: ~2.0x. The rewrite isolates dimension filtering into small CTEs, allowing the optimizer to build efficient hash tables for joining with the large store_sales table. This follows the "dimension_cte_isolate" pattern (1.93x in examples) and reduces the volume of data processed in the main aggregation.
