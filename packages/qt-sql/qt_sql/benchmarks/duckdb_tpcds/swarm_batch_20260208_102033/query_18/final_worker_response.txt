```sql
WITH filtered_date AS (
  SELECT d_date_sk
  FROM date_dim
  WHERE d_year = 1998
),
filtered_cd1 AS (
  SELECT cd_demo_sk, cd_dep_count
  FROM customer_demographics
  WHERE cd_gender = 'F'
    AND cd_education_status = 'Advanced Degree'
),
reduced_fact AS (
  SELECT
    cs_item_sk,
    cs_bill_customer_sk,
    cs_quantity,
    cs_list_price,
    cs_coupon_amt,
    cs_sales_price,
    cs_net_profit,
    fcd1.cd_dep_count
  FROM catalog_sales cs
  JOIN filtered_date fd ON cs.cs_sold_date_sk = fd.d_date_sk
  JOIN filtered_cd1 fcd1 ON cs.cs_bill_cdemo_sk = fcd1.cd_demo_sk
),
customer_complex AS (
  SELECT
    c_customer_sk,
    c_birth_year,
    ca_country,
    ca_state,
    ca_county
  FROM customer c
  JOIN customer_address ca ON c.c_current_addr_sk = ca.ca_address_sk
  JOIN customer_demographics cd2 ON c.c_current_cdemo_sk = cd2.cd_demo_sk
  WHERE c.c_birth_month IN (10, 7, 8, 4, 1, 2)
    AND ca.ca_state IN ('WA', 'GA', 'NC', 'ME', 'WY', 'OK', 'IN')
)
SELECT
  i.i_item_id,
  cc.ca_country,
  cc.ca_state,
  cc.ca_county,
  AVG(CAST(rf.cs_quantity AS DECIMAL(12, 2))) AS agg1,
  AVG(CAST(rf.cs_list_price AS DECIMAL(12, 2))) AS agg2,
  AVG(CAST(rf.cs_coupon_amt AS DECIMAL(12, 2))) AS agg3,
  AVG(CAST(rf.cs_sales_price AS DECIMAL(12, 2))) AS agg4,
  AVG(CAST(rf.cs_net_profit AS DECIMAL(12, 2))) AS agg5,
  AVG(CAST(cc.c_birth_year AS DECIMAL(12, 2))) AS agg6,
  AVG(CAST(rf.cd_dep_count AS DECIMAL(12, 2))) AS agg7
FROM reduced_fact rf
JOIN item i ON rf.cs_item_sk = i.i_item_sk
JOIN customer_complex cc ON rf.cs_bill_customer_sk = cc.c_customer_sk
GROUP BY ROLLUP (
  i.i_item_id,
  cc.ca_country,
  cc.ca_state,
  cc.ca_county
)
ORDER BY
  cc.ca_country,
  cc.ca_state,
  cc.ca_county,
  i.i_item_id
LIMIT 100;
```

**Changes**: Created a staged reduction pipeline: (1) `filtered_date` and `filtered_cd1` CTEs apply selective filters early; (2) `reduced_fact` joins catalog_sales with these filtered dimensions only; (3) `customer_complex` pre-joins customer, address, and cd2 with their filters; (4) final join combines reduced fact with item and customer complex. This minimizes the large fact table early and reduces join cardinality at each stage.

**Expected speedup**: ~2-3x by dramatically reducing the fact table early and creating efficient hash joins with pre-aggregated dimension CTEs.
