```sql
WITH filtered_date AS (
    SELECT d_date_sk
    FROM date_dim
    WHERE d_year = 1999
      AND d_moy = 12
),
all_sales_returns AS (
    -- Web channel
    SELECT 
        'web' AS channel,
        ws.ws_item_sk AS item_sk,
        ws.ws_quantity AS quantity,
        ws.ws_net_paid AS net_paid,
        wr.wr_return_quantity AS return_quantity,
        wr.wr_return_amt AS return_amt
    FROM web_sales ws
    INNER JOIN web_returns wr 
        ON ws.ws_order_number = wr.wr_order_number 
       AND ws.ws_item_sk = wr.wr_item_sk
    INNER JOIN filtered_date fd ON ws.ws_sold_date_sk = fd.d_date_sk
    WHERE wr.wr_return_amt > 10000
      AND ws.ws_net_profit > 1
      AND ws.ws_net_paid > 0
      AND ws.ws_quantity > 0

    UNION ALL

    -- Catalog channel
    SELECT 
        'catalog' AS channel,
        cs.cs_item_sk AS item_sk,
        cs.cs_quantity AS quantity,
        cs.cs_net_paid AS net_paid,
        cr.cr_return_quantity AS return_quantity,
        cr.cr_return_amount AS return_amt
    FROM catalog_sales cs
    INNER JOIN catalog_returns cr 
        ON cs.cs_order_number = cr.cr_order_number 
       AND cs.cs_item_sk = cr.cr_item_sk
    INNER JOIN filtered_date fd ON cs.cs_sold_date_sk = fd.d_date_sk
    WHERE cr.cr_return_amount > 10000
      AND cs.cs_net_profit > 1
      AND cs.cs_net_paid > 0
      AND cs.cs_quantity > 0

    UNION ALL

    -- Store channel
    SELECT 
        'store' AS channel,
        sts.ss_item_sk AS item_sk,
        sts.ss_quantity AS quantity,
        sts.ss_net_paid AS net_paid,
        sr.sr_return_quantity AS return_quantity,
        sr.sr_return_amt AS return_amt
    FROM store_sales sts
    INNER JOIN store_returns sr 
        ON sts.ss_ticket_number = sr.sr_ticket_number 
       AND sts.ss_item_sk = sr.sr_item_sk
    INNER JOIN filtered_date fd ON sts.ss_sold_date_sk = fd.d_date_sk
    WHERE sr.sr_return_amt > 10000
      AND sts.ss_net_profit > 1
      AND sts.ss_net_paid > 0
      AND sts.ss_quantity > 0
),
aggregated AS (
    SELECT 
        channel,
        item_sk AS item,
        CAST(SUM(return_quantity) AS DECIMAL(15,4)) / CAST(SUM(quantity) AS DECIMAL(15,4)) AS return_ratio,
        CAST(SUM(return_amt) AS DECIMAL(15,4)) / CAST(SUM(net_paid) AS DECIMAL(15,4)) AS currency_ratio
    FROM all_sales_returns
    GROUP BY channel, item_sk
),
ranked AS (
    SELECT 
        channel,
        item,
        return_ratio,
        currency_ratio,
        RANK() OVER (PARTITION BY channel ORDER BY return_ratio) AS return_rank,
        RANK() OVER (PARTITION BY channel ORDER BY currency_ratio) AS currency_rank
    FROM aggregated
)
SELECT 
    channel,
    item,
    return_ratio,
    return_rank,
    currency_rank
FROM ranked
WHERE return_rank <= 10 
   OR currency_rank <= 10
ORDER BY 1, 4, 5, 2
LIMIT 100;
```

**Changes**: Converted three separate channel subqueries into a single-pass UNION ALL with channel indicators, pushed date filters into a CTE, converted LEFT JOIN to INNER JOIN (semantically equivalent due to return_amt>10000 filter), and computed per-channel ranks using PARTITION BY in window functions. All filters are applied at the earliest possible point.

**Expected speedup**: 2-3x due to single scan of each fact table, reduced join overhead, and early filtering.
