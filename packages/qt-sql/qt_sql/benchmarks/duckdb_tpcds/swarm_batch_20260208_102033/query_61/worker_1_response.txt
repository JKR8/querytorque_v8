```sql
WITH filtered_dimensions AS (
  -- Pre-filter all dimension tables that are used in both subqueries
  -- This reduces the data volume early
  SELECT 
    s_store_sk,
    s_gmt_offset
  FROM store
  WHERE s_gmt_offset = -7
),
date_filter AS (
  SELECT 
    d_date_sk
  FROM date_dim
  WHERE d_year = 1999
    AND d_moy = 11
),
customer_address_filter AS (
  SELECT 
    ca_address_sk,
    c_customer_sk
  FROM customer_address
  JOIN customer ON ca_address_sk = c_current_addr_sk
  WHERE ca_gmt_offset = -7
),
item_filter AS (
  SELECT 
    i_item_sk
  FROM item
  WHERE i_category = 'Jewelry'
),
promotion_filter AS (
  SELECT 
    p_promo_sk
  FROM promotion
  WHERE p_channel_dmail = 'Y' 
    OR p_channel_email = 'Y' 
    OR p_channel_tv = 'Y'
),
base_sales AS (
  -- Create a base CTE with all necessary columns for both calculations
  -- This avoids scanning store_sales twice
  SELECT 
    ss_ext_sales_price,
    ss_promo_sk,
    ss_item_sk,
    ss_store_sk,
    ss_sold_date_sk,
    ss_customer_sk
  FROM store_sales
  WHERE ss_sold_date_sk IN (SELECT d_date_sk FROM date_filter)
    AND ss_store_sk IN (SELECT s_store_sk FROM filtered_dimensions)
    AND ss_customer_sk IN (SELECT c_customer_sk FROM customer_address_filter)
    AND ss_item_sk IN (SELECT i_item_sk FROM item_filter)
)
SELECT
  promotions,
  total,
  CAST(promotions AS DECIMAL(15, 4)) / CAST(total AS DECIMAL(15, 4)) * 100
FROM (
  SELECT
    SUM(ss_ext_sales_price) AS promotions
  FROM base_sales
  WHERE ss_promo_sk IN (SELECT p_promo_sk FROM promotion_filter)
) AS promotional_sales,
(
  SELECT
    SUM(ss_ext_sales_price) AS total
  FROM base_sales
) AS all_sales
ORDER BY
  promotions,
  total
LIMIT 100;
```

**Changes**: Consolidated shared dimension table filtering into CTEs and created a base sales CTE to scan store_sales only once. Applied early filtering on dimension tables to reduce join volume.

**Expected speedup**: ~1.5-2x by reducing duplicate work and applying dimension filters early before joins.
