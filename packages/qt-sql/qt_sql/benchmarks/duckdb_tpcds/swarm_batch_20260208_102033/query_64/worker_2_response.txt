```sql
WITH cs_ui AS (
  SELECT
    cs_item_sk,
    SUM(cs_ext_list_price) AS sale,
    SUM(cr_refunded_cash + cr_reversed_charge + cr_store_credit) AS refund
  FROM catalog_sales
  JOIN catalog_returns 
    ON cs_item_sk = cr_item_sk 
   AND cs_order_number = cr_order_number
  GROUP BY cs_item_sk
  HAVING SUM(cs_ext_list_price) > 2 * SUM(cr_refunded_cash + cr_reversed_charge + cr_store_credit)
),
-- Isolate date dimension filters
filtered_dates AS (
  SELECT d_date_sk, d_year
  FROM date_dim
  WHERE d_year IN (2001, 2002)  -- Only years needed in final query
),
-- Isolate item dimension filters
filtered_items AS (
  SELECT 
    i_item_sk,
    i_product_name,
    i_color,
    i_current_price
  FROM item
  WHERE i_color IN ('blanched', 'medium', 'brown', 'chocolate', 'burlywood', 'drab')
    AND i_current_price BETWEEN 24 AND 38  -- Combine both price conditions: 23+1 to 23+15
),
cross_sales AS (
  SELECT
    fi.i_product_name AS product_name,
    fi.i_item_sk AS item_sk,
    s.s_store_name AS store_name,
    s.s_zip AS store_zip,
    ad1.ca_street_number AS b_street_number,
    ad1.ca_street_name AS b_street_name,
    ad1.ca_city AS b_city,
    ad1.ca_zip AS b_zip,
    ad2.ca_street_number AS c_street_number,
    ad2.ca_street_name AS c_street_name,
    ad2.ca_city AS c_city,
    ad2.ca_zip AS c_zip,
    fd1.d_year AS syear,
    d2.d_year AS fsyear,
    d3.d_year AS s2year,
    COUNT(*) AS cnt,
    SUM(ss.ss_wholesale_cost) AS s1,
    SUM(ss.ss_list_price) AS s2,
    SUM(ss.ss_coupon_amt) AS s3
  FROM store_sales ss
  JOIN store_returns sr 
    ON ss.ss_item_sk = sr.sr_item_sk 
   AND ss.ss_ticket_number = sr.sr_ticket_number
  JOIN cs_ui 
    ON ss.ss_item_sk = cs_ui.cs_item_sk
  JOIN filtered_dates fd1 
    ON ss.ss_sold_date_sk = fd1.d_date_sk
  JOIN filtered_items fi 
    ON ss.ss_item_sk = fi.i_item_sk
  JOIN store s 
    ON ss.ss_store_sk = s.s_store_sk
  JOIN customer c 
    ON ss.ss_customer_sk = c.c_customer_sk
  JOIN customer_demographics cd1 
    ON ss.ss_cdemo_sk = cd1.cd_demo_sk
  JOIN customer_demographics cd2 
    ON c.c_current_cdemo_sk = cd2.cd_demo_sk
  JOIN household_demographics hd1 
    ON ss.ss_hdemo_sk = hd1.hd_demo_sk
  JOIN household_demographics hd2 
    ON c.c_current_hdemo_sk = hd2.hd_demo_sk
  JOIN customer_address ad1 
    ON ss.ss_addr_sk = ad1.ca_address_sk
  JOIN customer_address ad2 
    ON c.c_current_addr_sk = ad2.ca_address_sk
  JOIN promotion p 
    ON ss.ss_promo_sk = p.p_promo_sk
  JOIN income_band ib1 
    ON hd1.hd_income_band_sk = ib1.ib_income_band_sk
  JOIN income_band ib2 
    ON hd2.hd_income_band_sk = ib2.ib_income_band_sk
  JOIN date_dim d2 
    ON c.c_first_sales_date_sk = d2.d_date_sk
  JOIN date_dim d3 
    ON c.c_first_shipto_date_sk = d3.d_date_sk
  WHERE cd1.cd_marital_status <> cd2.cd_marital_status
    -- Item filters already applied in filtered_items CTE
  GROUP BY
    fi.i_product_name,
    fi.i_item_sk,
    s.s_store_name,
    s.s_zip,
    ad1.ca_street_number,
    ad1.ca_street_name,
    ad1.ca_city,
    ad1.ca_zip,
    ad2.ca_street_number,
    ad2.ca_street_name,
    ad2.ca_city,
    ad2.ca_zip,
    fd1.d_year,
    d2.d_year,
    d3.d_year
)
SELECT
  cs1.product_name,
  cs1.store_name,
  cs1.store_zip,
  cs1.b_street_number,
  cs1.b_street_name,
  cs1.b_city,
  cs1.b_zip,
  cs1.c_street_number,
  cs1.c_street_name,
  cs1.c_city,
  cs1.c_zip,
  cs1.syear,
  cs1.cnt,
  cs1.s1 AS s11,
  cs1.s2 AS s21,
  cs1.s3 AS s31,
  cs2.s1 AS s12,
  cs2.s2 AS s22,
  cs2.s3 AS s32,
  cs2.syear,
  cs2.cnt
FROM cross_sales cs1
JOIN cross_sales cs2
  ON cs1.item_sk = cs2.item_sk
 AND cs1.store_name = cs2.store_name
 AND cs1.store_zip = cs2.store_zip
WHERE cs1.syear = 2001
  AND cs2.syear = 2002  -- 2001 + 1
  AND cs2.cnt <= cs1.cnt
ORDER BY
  cs1.product_name,
  cs1.store_name,
  cs2.cnt,
  cs1.s1,
  cs2.s1;
```

**Changes**: Isolated date and dimension filters into separate CTEs (`filtered_dates`, `filtered_items`) to enable predicate pushdown and reduce fact table scans. Converted implicit joins to explicit JOIN syntax for better optimizer control. Combined redundant `i_current_price` filters into a single range (24-38). The main `cross_sales` CTE now joins with pre-filtered dimension CTEs instead of full tables.

**Expected speedup**: ~1.5-2x from reduced fact table scans and better predicate pushdown through isolated dimension filters.
