```sql
WITH filtered_date AS (
    SELECT d_date_sk
    FROM date_dim
    WHERE d_dow = 1
      AND d_year IN (1998, 1999, 2000)
),
filtered_store AS (
    SELECT s_store_sk, s_city
    FROM store
    WHERE s_number_employees BETWEEN 200 AND 295
),
filtered_household AS (
    SELECT hd_demo_sk
    FROM household_demographics
    WHERE hd_dep_count = 5
       OR hd_vehicle_count > 4
),
aggregated_sales AS (
    SELECT
        ss_ticket_number,
        ss_customer_sk,
        fs.s_city,
        SUM(ss_coupon_amt) AS amt,
        SUM(ss_net_profit) AS profit
    FROM store_sales ss
    JOIN filtered_date fd ON ss.ss_sold_date_sk = fd.d_date_sk
    JOIN filtered_store fs ON ss.ss_store_sk = fs.s_store_sk
    JOIN filtered_household fh ON ss.ss_hdemo_sk = fh.hd_demo_sk
    GROUP BY ss_ticket_number, ss_customer_sk, ss_addr_sk, fs.s_city
)
SELECT
    c.c_last_name,
    c.c_first_name,
    SUBSTRING(agg.s_city, 1, 30),
    agg.ss_ticket_number,
    agg.amt,
    agg.profit
FROM aggregated_sales agg
JOIN customer c ON agg.ss_customer_sk = c.c_customer_sk
ORDER BY
    c.c_last_name,
    c.c_first_name,
    SUBSTRING(agg.s_city, 1, 30),
    agg.profit
LIMIT 100;
```

```
Changes: Created CTEs for filtered dimension tables (date_dim, store, household_demographics) to push down filters early, then joined with store_sales for aggregation before final customer join. Converted implicit joins to explicit JOIN syntax.
Expected speedup: 1.5-2x from reduced fact table scanning and early dimension filtering.
```
