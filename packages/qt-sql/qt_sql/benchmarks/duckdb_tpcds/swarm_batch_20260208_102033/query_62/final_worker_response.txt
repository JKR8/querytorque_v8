```sql
WITH date_range AS (
    SELECT d_date_sk
    FROM date_dim
    WHERE d_month_seq BETWEEN 1194 AND 1194 + 11
), 
filtered_sales AS (
    SELECT 
        ws_warehouse_sk,
        ws_ship_mode_sk,
        ws_web_site_sk,
        ws_ship_date_sk - ws_sold_date_sk AS shipping_delay,
        CASE 
            WHEN ws_ship_date_sk - ws_sold_date_sk <= 30 THEN 1
            ELSE 0
        END AS delay_30,
        CASE 
            WHEN ws_ship_date_sk - ws_sold_date_sk > 30 
                 AND ws_ship_date_sk - ws_sold_date_sk <= 60 THEN 1
            ELSE 0
        END AS delay_31_60,
        CASE 
            WHEN ws_ship_date_sk - ws_sold_date_sk > 60 
                 AND ws_ship_date_sk - ws_sold_date_sk <= 90 THEN 1
            ELSE 0
        END AS delay_61_90,
        CASE 
            WHEN ws_ship_date_sk - ws_sold_date_sk > 90 
                 AND ws_ship_date_sk - ws_sold_date_sk <= 120 THEN 1
            ELSE 0
        END AS delay_91_120,
        CASE 
            WHEN ws_ship_date_sk - ws_sold_date_sk > 120 THEN 1
            ELSE 0
        END AS delay_over_120
    FROM web_sales
    WHERE ws_ship_date_sk IN (SELECT d_date_sk FROM date_range)
),
aggregated_sales AS (
    SELECT 
        ws_warehouse_sk,
        ws_ship_mode_sk,
        ws_web_site_sk,
        SUM(delay_30) AS "30 days",
        SUM(delay_31_60) AS "31-60 days",
        SUM(delay_61_90) AS "61-90 days",
        SUM(delay_91_120) AS "91-120 days",
        SUM(delay_over_120) AS ">120 days"
    FROM filtered_sales
    GROUP BY ws_warehouse_sk, ws_ship_mode_sk, ws_web_site_sk
)
SELECT
    SUBSTRING(w_warehouse_name, 1, 20),
    sm_type,
    web_name,
    "30 days",
    "31-60 days",
    "61-90 days",
    "91-120 days",
    ">120 days"
FROM aggregated_sales
JOIN warehouse ON ws_warehouse_sk = w_warehouse_sk
JOIN ship_mode ON ws_ship_mode_sk = sm_ship_mode_sk
JOIN web_site ON ws_web_site_sk = web_site_sk
ORDER BY
    SUBSTRING(w_warehouse_name, 1, 20),
    sm_type,
    web_name
LIMIT 100;
```

Changes: Implemented three-stage pipeline: (1) filter web_sales by date range using derived table, computing delay buckets immediately; (2) pre-aggregate by integer foreign keys before joining to dimensions; (3) final lightweight join to dimension tables for descriptive columns only.

Expected speedup: 2-3x by reducing early data movement, deferring string operations, and minimizing join cardinality through early aggregation.
