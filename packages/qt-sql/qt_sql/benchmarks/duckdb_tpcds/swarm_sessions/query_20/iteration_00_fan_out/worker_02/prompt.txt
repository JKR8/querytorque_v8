 from customer_address a
     ,customer c
     ,store_sales s
     ,date_dim d
     ,item i

 	and c.c_customer_sk = s.ss_customer_sk
 	and s.ss_sold_date_sk = d.d_date_sk
 	and s.ss_item_sk = i.i_item_sk

 	     (select distinct (d_month_seq)
