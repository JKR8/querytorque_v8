```sql
WITH filtered_item AS (
  SELECT 
    i_item_sk,
    i_brand_id,
    i_class_id,
    i_category_id,
    i_manufact_id
  FROM item
  WHERE i_category = 'Home'
),
filtered_date AS (
  SELECT 
    d_date_sk,
    d_year
  FROM date_dim
  WHERE d_year IN (1998, 1999)
),
catalog_sales_pre AS (
  SELECT 
    d.d_year,
    i.i_brand_id,
    i.i_class_id,
    i.i_category_id,
    i.i_manufact_id,
    cs_quantity - COALESCE(cr_return_quantity, 0) AS sales_cnt,
    cs_ext_sales_price - COALESCE(cr_return_amount, 0.0) AS sales_amt
  FROM catalog_sales cs
  JOIN filtered_item i ON cs.cs_item_sk = i.i_item_sk
  JOIN filtered_date d ON cs.cs_sold_date_sk = d.d_date_sk
  LEFT JOIN catalog_returns cr ON (
    cs.cs_order_number = cr.cr_order_number 
    AND cs.cs_item_sk = cr.cr_item_sk
  )
),
store_sales_pre AS (
  SELECT 
    d.d_year,
    i.i_brand_id,
    i.i_class_id,
    i.i_category_id,
    i.i_manufact_id,
    ss_quantity - COALESCE(sr_return_quantity, 0) AS sales_cnt,
    ss_ext_sales_price - COALESCE(sr_return_amt, 0.0) AS sales_amt
  FROM store_sales ss
  JOIN filtered_item i ON ss.ss_item_sk = i.i_item_sk
  JOIN filtered_date d ON ss.ss_sold_date_sk = d.d_date_sk
  LEFT JOIN store_returns sr ON (
    ss.ss_ticket_number = sr.sr_ticket_number 
    AND ss.ss_item_sk = sr.sr_item_sk
  )
),
web_sales_pre AS (
  SELECT 
    d.d_year,
    i.i_brand_id,
    i.i_class_id,
    i.i_category_id,
    i.i_manufact_id,
    ws_quantity - COALESCE(wr_return_quantity, 0) AS sales_cnt,
    ws_ext_sales_price - COALESCE(wr_return_amt, 0.0) AS sales_amt
  FROM web_sales ws
  JOIN filtered_item i ON ws.ws_item_sk = i.i_item_sk
  JOIN filtered_date d ON ws.ws_sold_date_sk = d.d_date_sk
  LEFT JOIN web_returns wr ON (
    ws.ws_order_number = wr.wr_order_number 
    AND ws.ws_item_sk = wr.wr_item_sk
  )
),
all_sales_combined AS (
  SELECT * FROM catalog_sales_pre
  UNION ALL
  SELECT * FROM store_sales_pre
  UNION ALL
  SELECT * FROM web_sales_pre
),
all_sales_agg AS (
  SELECT
    i_brand_id,
    i_class_id,
    i_category_id,
    i_manufact_id,
    SUM(CASE WHEN d_year = 1999 THEN sales_cnt ELSE 0 END) AS sales_cnt_1999,
    SUM(CASE WHEN d_year = 1999 THEN sales_amt ELSE 0 END) AS sales_amt_1999,
    SUM(CASE WHEN d_year = 1998 THEN sales_cnt ELSE 0 END) AS sales_cnt_1998,
    SUM(CASE WHEN d_year = 1998 THEN sales_amt ELSE 0 END) AS sales_amt_1998
  FROM all_sales_combined
  GROUP BY
    i_brand_id,
    i_class_id,
    i_category_id,
    i_manufact_id
)
SELECT
  1998 AS prev_year,
  1999 AS year,
  i_brand_id,
  i_class_id,
  i_category_id,
  i_manufact_id,
  sales_cnt_1998 AS prev_yr_cnt,
  sales_cnt_1999 AS curr_yr_cnt,
  sales_cnt_1999 - sales_cnt_1998 AS sales_cnt_diff,
  sales_amt_1999 - sales_amt_1998 AS sales_amt_diff
FROM all_sales_agg
WHERE
  sales_cnt_1998 > 0
  AND CAST(sales_cnt_1999 AS DECIMAL(17, 2)) / CAST(sales_cnt_1998 AS DECIMAL(17, 2)) < 0.9
ORDER BY
  sales_cnt_diff,
  sales_amt_diff
LIMIT 100
```

Changes: Created prefetch CTEs for filtered dimension tables (item and date_dim), then pre-joined each fact table with these filtered dimensions in dedicated CTEs. Consolidated the two year aggregations into a single pass using conditional aggregation, eliminating the self-join on the large `all_sales` CTE.

Expected speedup: 2-3x by reducing fact table scans through early filtering and eliminating the expensive self-join.
